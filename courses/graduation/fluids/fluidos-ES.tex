\begin{frame}{O que é um fluido?}
	\begin{block}{Fluido}
		\justifying{Líquido incompressível que assume a forma do recipiente que o contém.}
	\end{block}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{itemize}
				\item [\ding{51}] Os sólidos são objetos que possuem forma definida, portanto não são fluidos;
				\item [\ding{51}] O Volume de um fluido não se altera independente da temperatura que se encontra e do recipiente que o contém.
				\item [\ding{51}] Conceitos como densidade e pressão são usados ao invés de massa e força.
			\end{itemize}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}[c]
				\includegraphics[scale=0.25]{figuras/fluido.jpg}
				\caption*{\footnotesize Formato de um fluido em diversos recipientes.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Densidade}
	\begin{columns}
		\begin{column}{0.6\textwidth}
				Definimos densidade $\rho$ de um objeto como a taxa da quantidade infinitesimal de massa $dm$ por elemento $dV$ de seu volume,
				\begin{align*}
					\Aboxed{\rho=\lim\limits_{\Delta V\rightarrow 0}\frac{\Delta m}{\Delta V} = \frac{dm}{dV}.}
				\end{align*}
				\alert{Assim, a densidade fornece informações de como a massa está sendo distribuída ao longo do volume.}
		\end{column}

\begin{frame}{Força atuando em cada elemento de um fluido}
	\begin{columns}
		\begin{column}[c]{0.7\textwidth}
			Considere uma força $\Delta F$ atuando em um pequena área $\Delta A$ no fundo de um recipiente. Chamaremos de $\Delta F$ a força resultante atuando nessa área, onde podemos decompô-la em suas componentes tangencial $\Delta \vec{F}_t$ e perpendicular $\Delta \vec{F}_p$. No entanto, em um fluido em equilíbrio a força resultante deve ser zero, portanto obrigatoriamente deveremos ter $\Delta F_t=0$, restando apenas $\Delta \vec{F}_p$, que no caso deve ser equivalente ao peso da parte do fluido justamente acima, onde
			\begin{align*}
				\Delta F_p=\Delta F\cos\theta= P.
			\end{align*}

\begin{frame}{Pressão}
\begin{columns}
	\begin{column}[c]{0.7\textwidth}
		Definimos pressão p como a força distribuída por unidade de área perpendicular a superfície, ou seja,
		\begin{align*}
			\Aboxed{p =\lim\limits_{\Delta V\rightarrow 0}\frac{\Delta F_p}{\Delta A} = \frac{dF_p}{dA}.}
		\end{align*}
		No caso de uma força atuando obliquamente a área A temos que a pressão é definida como o produto escalar da força $\vec{F}$ pelo versor $\widehat{n}$ perpendicular a área A,
		\begin{align*}
			\Aboxed{p = \frac{d}{dA}\left(F\cos\theta\right)= \left(\frac{dF}{dA}\right)\cos\theta=\frac{d\vec{F}}{dA}\cdot\widehat{n}.}
		\end{align*}
	\end{column}

\begin{frame}{Variação da pressão com altitude e profundidade}
\begin{multicols}{2}
	Sabemos que a força atuando em cima e embaixo do cilindro é dado por $F = p A$, onde $A$ é a área do cilindro, portanto a força em cima e embaixo são
	\begin{align*}
	F_1 & = p_1A,\\
	F_2 & = p_2A.
	\end{align*}
	Se o cilindro está em repouso, pela segunda Lei de Newton a força resultante deve ser zero, ou seja,
	\begin{align*}
	F_2 - F_1 = P = (\Delta m)g.
	\end{align*}
	Pela definição de densidade, $\Delta m = \rho \Delta V$, e sabendo que o volume do cilindro é a base A vezes a altura h temos
	\begin{align*}
	p_2\bcancel{A} & = p_1\bcancel{A} + \rho gh\bcancel{A},\\
	\Aboxed{p_2 & = p_1 + \rho g h.}
	\end{align*}
\end{multicols}
\end{frame}

\begin{frame}{Variação da pressão com a profundidade}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{block}{Lei de Stevin}
				Se a superfície de um fluido, cuja densidade é $\rho$, está submetida a uma pressão $p_{atm}$, a pressão p, no interior desse líquido, a uma profundidade h, é dada por
				\[p=p_{atm}+\rho gh\]
			\end{block}
			\begin{corollary}
				A força da gravidade puxa o fluido para baixo causando uma pressão na base e nas paredes do recipiente.
			\end{corollary}
		\end{column}

\begin{frame}{Experiência de Torricelli}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\justifying Coloca-se mercúrio cuja densidade é conhecida num tubo fino e vira-o de cabeça para baixo. O líquido irá descer e irá preencher o recipiente da parte de baixo. A parte de cima como estava fechada não entrou ar e com a descida do líquido criou-se um vácuo, portanto a pressão da parte de cima será zero. Pela Lei de Stevin temos que a pressão da parte de baixo é dado por
			\begin{align*}
				p_{atm} = \rho g h,
			\end{align*}
			onde h é a coluna de mercúrio (se for medido ao nível do mar h=760mm).

\begin{frame}{Relação entre a densidade do fluido, do objeto e o princípio de Arquimedes}
	\begin{columns}[t]
		\begin{column}{0.55\textwidth}
		Pela definição de empuxo E podemos dizer que
		\begin{align*}
			E=m_{fluido}g,
		\end{align*}
		mas pela definição de densidade temos $m_{fluido}=\rho_{fluido}V$, portanto
		\begin{align*}
			E = \rho_{fluido}Vg
		\end{align*}
		O peso P do objeto mergulhado no fluido é dado por $P=m_{obj}g$, portanto se o empuxo for igual ao peso do objeto temos
		\end{column}
		\begin{column}{0.45\textwidth}
		\begin{align*}
			m_{obj}g & = \rho_{fluido}Vg,\\
			\rho_{obj}\bcancel{Vg} & = \rho_{fluido}\bcancel{Vg}.
		\end{align*}
			\begin{corollary}
				\begin{itemize}
					\small
					\item Se $\rho_{fluido}<\rho_{obj}$, o corpo afundará;
					\item Se $\rho_{fluido}=\rho_{obj}$, o corpo ficará em equilíbrio;
					\item Se $\rho_{fluido}>\rho_{obj}$, o corpo irá flutuar na superfície;
				\end{itemize}
			\end{corollary}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Fluidos ideias em movimento}
	Um fluido ideal em movimento satisfaz as seguintes condições:
		\begin{enumerate}
			\item[\ding{51}] Escoamento laminar: a velocidade do fluido em ponto qualquer não varia com o tempo;
			\item[\ding{51}] Escoamento incompressível: a massa específica tem o mesmo valor em todos os pontos do fluido e em qualquer instante de tempo.
			\item[\ding{51}] Escoamento não viscoso: o fluido é não viscoso, ou seja, não apresenta resistência ao escoamento.
			\item[\ding{51}] Escoamento irrotacional: o movimento do fluido é apenas linear.
		\end{enumerate}
\end{frame}

			\begin{align*}
				\tau_1 & = p_1V,\\
				\tau_2 & = -p_2V.

		\end{column}
		\begin{column}[t]{0.55\textwidth}
			Substituindo na equação acima temos
			\begin{align*}
				\alert{p_1V} + \frac{1}{2}mv^2_1+mgh_1 & = -\left(\alert{-p_2V}\right) + \\
				& + \frac{1}{2}mv^2_2 + mgh_2,
			\end{align*}
			mas sabemos que $V=\frac{m}{\rho},$
			\begin{align*}
				p_1\alert{\frac{\bcancel{m}}{\rho}} + \frac{1}{2}\bcancel{m}v^2_1+\bcancel{m}gh_1 & = p_2\alert{\frac{\bcancel{m}}{\rho}} + \frac{1}{2}\bcancel{m}v^2_2 + \bcancel{m}gh_2,\\
				\frac{p_1}{\rho} + \frac{v^2_1}{2}+gh_1 & = \frac{p_2}{\rho} + \frac{v^2_2}{2} + gh_2.
			\end{align*}
		\end{column}
	\end{columns}

\begin{frame}{Pressão, velocidade e altura de um fluido em duas regiões distintas (continuação)}
	\begin{block}{Equação de Bernoulli}
		Para um fluido não viscoso com escoamento laminar a soma das parcelas hidrostáticas e hidrodinâmicas é a mesma em cada ponto do fluido, no qual vale a equação
		\begin{align*}
			\frac{p_1}{\rho} +gh_1 + \frac{v^2_1}{2} & = \frac{p_2}{\rho} +gh_2 + \frac{v^2_2}{2} = \cdots,\\
			\frac{p}{\rho} +gh + \frac{v^2}{2} & = \text{constante}.
		\end{align*}
	\end{block}
	\begin{corollary}
		A equação de Bernoulli se baseia na lei de conservação da energia na mecânica.
	\end{corollary}
\end{frame}

\begin{frame}{Analisando os termos da equação de Bernoulli}
	Supondo a densidade constante ao longo de todo o fluido, podemos multiplicar todos os termos da equação por $\rho$ e obter a relação
	\begin{align*}
		\underbracket{\alert{\bcancel{\rho}}\frac{p}{\bcancel{\rho}} +\alert{\rho}gh}_{\text{Lei de Stevin}} + \alert{\rho}\frac{v^2}{2} = \text{constante}.
	\end{align*}
	\begin{corollary}
		\begin{itemize}
			\item Parcela hidrostática \alert{$\left(p +\rho gh\right)$}: Corresponde a pressão hidrostática no fluido;
			\item Parcela fluidodinâmica \alert{$\left(\rho\frac{v^2}{2}\right)$}: Corresponde a pressão hidrodinâmica;
			\item Se o fluido está em repouso $\frac{\rho v^2}{2}=0$ temos a Lei de Stevin.
		\end{itemize}
	\end{corollary}
\end{frame}

\begin{frame}{Lei de Stevin para um fluido não homogêneo}
	\begin{multicols}{2}
		Sabemos que a diferença das forças atuando em cima e embaixo do cilindro é dado por $F = p A$, onde $A$ é a área do cilindro, portanto a força resultante será
		\begin{align*}
			\Delta F & = F_2-F_1,\\
			\Delta F & = \rho g\Delta V=p\Delta A,
		\end{align*}
		onde para o cilindro de altura dz teremos $\Delta V=A\Delta z$. Substituindo
		\begin{align*}
			p\bcancel{A} = -\rho g\bcancel{A}dz,
		\end{align*}
		ou seja,
		\begin{align*}
			\frac{dp}{dz} = -\rho(z) g.
		\end{align*}
		Integrando teremos
		\begin{align*}
			\Aboxed{\Delta p = -\int\limits_{z_0}^{h}g\rho(z)dz.}
		\end{align*}
	\end{multicols}
\end{frame}

\begin{frame}{Variação da pressão com a altitude (em construção)}
	\begin{multicols}{2}
		No caso anterior, se $\rho = cte$ chegamos na Lei de Stevin como caso especial. Entretanto, se tivermos $\rho (z)=\frac{\rho_{atm}}{p_{atm}}p(z)$, teremos
		\begin{align*}
			\frac{dp}{dz} & = -g\left[\frac{\rho_{atm}}{p_{atm}}p(z)\right],\\
			dp & = -\left[g\frac{\rho_{atm}}{p_{atm}}p(z)\right]dz,\\
			\frac{dp}{p} & = -\left(g\frac{\rho_{atm}}{p_{atm}}\right)dz.
		\end{align*}
		Integrando teremos
		\begin{align*}
			\int\limits_{p_{atm}}^{p} \frac{dp}{p} & = -\int\limits_{0}^{z}g\frac{\rho_{atm}}{p_{atm}}dz,\\
			\ln p(z)-\ln p_{atm} & = -\frac{\rho_{atm}}{p_{atm}}z,\\
			\ln \left[\frac{p(z)}{p_{atm}}\right] & = -\frac{\rho_{atm}}{p_{atm}}z,\\
			\Aboxed{p(z) & = p_{atm}e^{-\left(\frac{\rho_{atm}}{p_{atm}}\right)z}.}
		\end{align*}
	\end{multicols}
\end{frame}

\begin{frame}{Propriedades que definem um gás ideal}
	\begin{enumerate}
		\item[\ding{51}] Entre as partículas dele, não há qualquer tipo de interação, como forças atrativas ou repulsivas (\alert{somente colisões entre esferas duras}).
		\item[\ding{51}] As colisões são perfeitamente elástica (\alert{não há perdas de energia cinética, portanto o gás mantém a sua temperatura se o volume e a pressão não mudarem}).
		\item[\ding{51}] O volume de cada molécula é praticamente desprezível comparado com o volume do gás (\alert{assim o espaço que as moléculas podem se mover é praticamente o volume do recipiente}).
	\end{enumerate}
\end{frame}

\begin{frame}{Transformação isotérmica}

				Numa transformação termodinâmica do estado 1 para o estado 2, se a temperatura T de uma dada massa gasosa for mantida constante, o volume V desse gás será inversamente proporcional à pressão p exercida sobre ele,

					pV & = \text{constante},\;(T=\text{constante}).

\section{Equação do gás ideal}

\subsection{}

\begin{frame}{Equação do gás ideal}
	\begin{columns}[t]
		\begin{column}{0.5\textwidth}
			Reunindo as Leis de Boyle, Gay-Lussac e Avogadro de um gás a pressão p, temperatura T e massa molar M na forma
			\begin{align*}
				\rho\; & \alpha\;p,\\
				\rho\; & \alpha\;\frac{1}{T},\\
				\rho\; & \alpha\;M.
			\end{align*}
			Reunindo em uma única relação
			\begin{equation*}
				\rho\;\alpha\;\frac{pM}{T}.
			\end{equation*}
		\end{column}
		\begin{column}{0.5\textwidth}
			Mas sabemos que $\rho=\frac{m_{\text{Total}}}{V}$, portanto
			\begin{align*}
				\frac{m_{\text{Total}}}{V}\; & \alpha\;\frac{pM}{T},\\
				pV & \alpha\;\left(\frac{m}{M}\right)T.
			\end{align*}
			Mas $n=m_{\text{Total}}/M$, resultando na relação entre a pressão, volume e temperatura de uma massa gasosa contendo n mols,
			\begin{block}{Equação do gás ideal}
				\begin{equation*}
					pV = nRT.
				\end{equation*}
			\end{block}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Constante universal dos gases}
	\begin{multicols}{2}
	Verificamos anteriormente que $pV=RnT$, ou seja,
	\begin{align*}
		R=\frac{pV}{nT},
	\end{align*}
	onde R é a constante de proporcionalidade chamada \textcolor{red}{constante universal dos gases}. Experimentalmente, podemos verificar que para um mol de um gás ideal qualquer (n=1 mol), à temperatura de 273 K e à pressão de 1 atm, o gás irá ocupar um volume de 22,4 L.
	 Substituindo na equação acima
	\begin{align*}
		R=0,082\;\frac{\text{atm}\cdot{L}}{\text{mol}\cdot{K}}.
	\end{align*}
	No SI a pressão é medida em $N/m^2$ e o volume em $m^3$, portanto após converter as unidades de medida temos
	\begin{align*}
		\boxed{R=8,31\;\frac{\text{J}}{\text{mol}\cdot{K}}.}
	\end{align*}
	\end{multicols}
\end{frame}

\section{Teoria cinética dos gases}

\begin{frame}{Cálculo da pressão de um gás}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			A pressão de um gás se deve a colisões contínuas das moléculas de massa m contra as paredes do recipiente, na forma
			\begin{align*}
				p=\frac{1}{3}\frac{N}{V}m\left<v^2\right>.
			\end{align*}
			\begin{itemize}
				\item N: Número total de moléculas;
				\item V: Volume do recipiente;
				\item m: Massa de cada molécula;
				\item $\left<v^2\right>$: Média dos quadrados das velocidades.
			\end{itemize}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.35]{figuras/gas_molecules.png}
				\caption*{\footnotesize Recipiente contendo gás ideal.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Energia interna de um gás}
	\begin{multicols}{2}
		Do cálculo da pressão de um gás contendo n mols de moléculas podemos ter
		\begin{align*}
			pV = \frac{1}{3}(nN_0)m\left<v^2\right>.
		\end{align*}
		Comparando com a equação do gás ideal temos
		\begin{align*}
			\frac{1}{3}\bcancel{n}N_0m\left<v^2\right> &=\bcancel{n}RT,\\
			\frac{1}{2}m\left<v^2\right> &=\frac{3}{2}\underbracket{\left(\frac{R}{N_0}\right)}_{\alert{k_B}}T.
		\end{align*}
	\end{multicols}
	onde $k_B$ é chamado \alert{constante de Boltzmann} ($k_B=1,38\times 10^{-23}\;\text{J/K}$).
	\begin{block}{Energia interna de um gás ideal}

\begin{frame}{Transformar um número em notação científica}
	\begin{corollary}
		\begin{enumerate}[leftmargin=1.75cm]
			\item[Passo 1:] Escrever o número incluindo a vírgula.
			\item[Passo 2:] Andar com a vírgula até que reste somente um número diferente de zero no lado esquerdo.
			\item[Passo 3:] Colocar no expoente da potência de 10 o número de casas decimais que tivemos que "andar" com a vírgula. Se ao andar com a vírgula o valor do número diminuiu, o expoente ficará positivo, se aumentou o expoente ficará negativo.
		\end{enumerate}
	\end{corollary}
	\begin{exampleblock}{Exemplo}
		\begin{equation*}
			6\; 590\; 000\; 000\; 000\; 000,0 = \SI{6,59e15}{}		
		\end{equation*}
	\end{exampleblock}
\end{frame}

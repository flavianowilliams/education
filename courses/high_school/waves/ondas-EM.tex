\begin{frame}{Ondas progressivas numa corda (continuação)}
	Uma onda é formada a partir de uma série de pulsos propagando-se na mesma direção e sentido a cada intervalo de tempo igual ao período de oscilação T.
	\begin{figure}
		\includegraphics[scale=0.27]{figuras/figura-1.png}
				\caption*{\href{https://www.youtube.com/watch?v=0EFK_vZTpio&t=42s}{\textcolor{blue}{\footnotesize Pulsos se propagando numa corda.}}}
	\end{figure}
	\begin{corollary}
		Uma onda mecânica necessita de um meio material para se propagar.
	\end{corollary}
\end{frame}

\begin{frame}{Tipos de ondas progressivas}
	\begin{columns}[c]
		\begin{column}[c]{0.5\textwidth}
			\begin{figure}
				\includegraphics[scale=0.7]{figuras/ondas.jpeg}
				\caption*{\footnotesize Diferença entre onda longitudinal e transversal \cite{ondas}.}
			\end{figure}
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{itemize}
				\item[\ding{51}] Numa onda longitudinal, a direção de oscilação é paralela a direção de propagação dessa onda.
				\item[\ding{51}] Numa onda transversal, a direção de oscilação é perpendicular a direção de propagação dessa onda.
			\end{itemize}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Velocidade de uma onda progressiva}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Se cada pulso percorrer uma distância $\Delta S$ num intervalo de tempo $\Delta t$, a velocidade v desse pulso será
			\begin{equation*}
				v=\frac{\Delta S}{\Delta t}.
			\end{equation*}
			Por exemplo, se $\Delta S=6,0\;m$ e $\Delta t=1,5\;\text{seg}$, a velocidade dessa onda será 4 m/s.
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}
				\includegraphics[scale=0.3]{figuras/onda-velocidade.png}
			\end{figure}
		\end{column}
	\end{columns}	
	\begin{corollary}
		A velocidade de propagação de uma onda numa corda depende da densidade (\alert{$\frac{\text{massa}}{\text{comprimento}}$}) dessa corda e da tensão aplicada sobre ela.
	\end{corollary}
\end{frame}

\begin{frame}{Propriedades de uma onda}
	\begin{columns}[c]
		\begin{column}[c]{0.5\textwidth}
			\begin{figure}
				\includegraphics[scale=0.35]{figuras/onda-lambda.png}
			\end{figure}
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{itemize}
				\item[\ding{51}] \alert{Amplitude da onda}: A mesma amplitude da oscilação vertical.
				\item[\ding{51}] \alert{Frequência da onda}: A mesma frequência da oscilação vertical.
				\item[\ding{51}] \alert{Comprimento de onda $(\lambda)$}: Distância que a onda percorre durante um período T.
			\end{itemize}
		\end{column}
	\end{columns}
	\begin{corollary}
		Sendo v a velocidade de propagação da onda e T o período, o comprimento de onda é dado por
		\begin{equation*}
			\lambda = vT
		\end{equation*}
	\end{corollary}
\end{frame}

\begin{frame}{Interferência destrutiva e a experiência de Young}
	Teremos interferência destrutiva se a diferença dos caminhos percorridos pelas duas ondas for um valor semi-inteiro do comprimento de onda, ou seja,	
	\begin{columns}[c]
		\begin{column}[c]{0.6\textwidth}
			\begin{equation*}
				r_1-r_2 = \left(n+\frac{1}{2}\right)\lambda,\; n=\pm 0,\pm 1,\pm 2,\dots
			\end{equation*}
			\begin{corollary}
				\begin{align*}
					r_1-r_2 & = dsen\theta = 
					\begin{cases}
						n\lambda, & \textrm{(Construtiva)}\\											\left(n+\frac{1}{2}\right)\lambda, & \textrm{(Destrutiva)}
					\end{cases}\\
					n & = \pm 0,\pm 1,\pm 2,\pm 3,\pm 4,\dots
				\end{align*}
			\end{corollary}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}[c]

\end{frame}

\section{Ondas sonoras}

\subsection{}

\begin{frame}{O que é som}
	O som é uma onda mecânica longitudinal que se propaga ao longo do meio material (sólido, líquido ou gás), numa frequência entre 20 a 20000 Hz.
	\begin{columns}
		\begin{column}[t]{0.6\textwidth}
			\begin{figure}
				\Centering
				\includegraphics[scale=0.3]{figuras/som.jpg}
				\caption*{\footnotesize Som se propagando da fonte até o ouvido.}
			\end{figure}
		\end{column}
		\begin{column}[t]{0.4\textwidth}
			\begin{corollary}
				 Cada pulso se propaga através da variação de pressão entre as moléculas que compõem a matéria (as moléculas que estão atrás empurram as que estão justamente a frente, e assim a energia do pulso se propaga no meio).
			\end{corollary}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Propriedades de uma onda sonora}
%	Podemos determinar as propriedades da onda através da amplitude, frequência e comprimento de onda.
	\begin{block}{Qualidades usadas para distinguir o som}
		\begin{itemize}
			\item[\ding{51}] \alert{Intensidade}: Energia média transmitida por unidade de tempo e área (depende da amplitude da onda sonora).
			\item[\ding{51}] \alert{Altura}: Sensação que permite distinguir entre sons graves e agudos (diretamente associado com a frequência da onda).
			\item[\ding{51}] \alert{Timbre}: Capacidade de distinguir sons de mesma altura (frequência) e intensidade em instrumentos diferentes (coloração do som).
		\end{itemize}	
	\end{block}
	\begin{corollary}
		A velocidade da onda sonora no ar é 340 m/s e aumenta com a temperatura, além do mais, já é conhecido que $v_{\text{sólido}}>v_{\text{líquido}}>v_{\text{gás}}$.
	\end{corollary}
\end{frame}

\begin{frame}{Altura}
	A característica física de uma onda sonora associada a altura é a frequência.
	\begin{table}
		\caption*{Notas musicais (Hz).}
		\begin{tabular}{ccccccccc}
			\hline
			\hline
			Nota & Dó & Ré & Mi & Fá & Sol & Lá & Si & Dó\\
			Freq. & 264 & 297 & 330 & 352 & 396 & 440 & 495 & 528\\
			\hline		
		\end{tabular}	
	\end{table}
	\begin{corollary}
		\begin{itemize}
			\item[\ding{51}] Um som grave possui baixa frequência (som baixo), enquanto que um som alto possui alta frequência (som alto);
			\item[\ding{51}] Infrassons são ondas sonoras cuja frequência estão abaixo de 20 Hz enquanto que ultrasons estão acima de 20000 Hz.
		\end{itemize}				
	\end{corollary}
\end{frame}

\begin{frame}{Intensidade de uma onda sonora}
	\begin{columns}[b]
		\begin{column}[b]{0.5\textwidth}
			A intensidade de uma onda é definida como
			\begin{align*}
				I = \frac{1}{A}\left(\frac{\Delta E}{\Delta t}\right).
			\end{align*}
			No SI usa-se a unidade $\left[I\right]=\frac{W}{m^{2}}$ mas na prática usa-se decibéis (dB),
			\begin{equation*}
				\boxed{\beta=10Log_{10}(I/I_{0})\; \textrm{dB}.}
			\end{equation*}
		\end{column}
		\begin{column}[b]{0.5\textwidth}
			\begin{figure}[c]
				\Centering
				\includegraphics[scale=0.22]{figuras/intensidade-sonora.png}
				\caption*{\footnotesize Intensidades sonoras de alguns objetos.}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		$I_0$ é a mínima intensidade perceptível por uma pessoa $(I_{0}=10^{-12}W/m^{2})$.
	\end{corollary}
\end{frame}

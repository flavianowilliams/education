\begin{frame}{Trabalho realizado por um gás}
	\begin{columns}
		\begin{column}{0.7\textwidth}
			A força que o gás exerce no pistão realiza um trabalho $\tau$ sobre ele, deslocando-o para cima por uma distância $\Delta S$, segundo a relação
			\begin{equation*}
				\tau = F\cdot\Delta S.
			\end{equation*}
			Mas $F=pA$, sendo A a área do pistão, portanto
			\begin{align*}
				\tau & = p\overbracket{A\Delta S}^{\alert{\Delta V}},\\
				\Aboxed{\tau & = p\Delta V.}
			\end{align*}
		\end{column}
		\begin{column}{0.3\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.3]{figuras/figura-1.png}
				\caption*{\footnotesize Aumento $\Delta V$ do volume do câmara devido ao trabalho $\tau$ realizado pelo gás.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Convenção de sinais de calor e trabalho na termodinâmica}
	\begin{block}{Convenção de sinais do calor}
		\begin{itemize}
			\item Se o sistema recebe calor da vizinhança então Q é positivo.
			\item Se o sistema cede calor para a vizinhança então Q é negativo.
		\end{itemize}	
	\end{block}
	\begin{block}{Convenção de sinais do trabalho}
		\begin{itemize}
			\item Se o trabalho está sendo realizado sobre o sistema então $\tau$ é negativo.
			\item Se o sistema realiza trabalho sobre a vizinhança então $\tau$ é positivo.
		\end{itemize}	
	\end{block}
\end{frame}

\begin{frame}{Primeira Lei da Termodinâmica e a Lei da conservação da energia}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			Quando um sistema vai de um estado i para o estado f e troca energia com a vizinhança, a sua energia interna aumenta ou diminui e a sua variação é dado por
			\begin{equation*}
				\Delta U = U_f-U_i=Q.
			\end{equation*}
			Se ele ao mesmo tempo realizar trabalho $\tau$, ou trabalho for feito sobre ele, a quantidade de energia interna que ele recebe ou cede é dado por
			\begin{equation*}
				\boxed{\Delta U = Q-\tau.}
			\end{equation*}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.35]{figuras/figura-3.png}
				\caption*{\footnotesize Exemplo de aplicação da primeira Lei da Termodinâmica.}
			\end{figure}
		\end{column}
	\end{columns}	
\end{frame}

\section{Aplicações da Primeira Lei da Termodinâmica}

\begin{frame}{Transformação adiabática}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			Na transformação adiabática \alert{o sistema não troca calor com a vizinhança}, portanto Q = 0. A variação da energia interna do gás é dado por
			\begin{align*}
				\Delta U & = Q - \tau,\\
				\Aboxed{\Delta U & = -\tau.}
			\end{align*}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.35]{figuras/figura-4.png}
				\caption*{\footnotesize Exemplo de transformação adiabática.}
			\end{figure}
		\end{column}
	\end{columns}	
\end{frame}

\begin{frame}{Transformação isotérmica}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			Na transformação isotérmica, a temperatura do sistema não muda e \alert{a energia de um gás depende somente da temperatura T}, ou seja,
			\begin{equation*}
				U(T)= \frac{3}{2}Nk_BT,
			\end{equation*}
			portanto $\Delta U=0$. Pela Primeira Lei da Termodinâmica temos
			\begin{align*}
				Q-\tau & = \Delta U,\\
				Q - \tau & = 0,\\
				\Aboxed{Q & = \tau.}
			\end{align*}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.45]{figuras/figura-5.png}
				\caption*{\footnotesize Exemplo de transformação isotérmica.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Transformação isovolumétrica}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			Na transformação isovolumétrica (ou isocórica), o volume do sistema não muda, portanto
			\begin{equation*}
				\Delta V = 0.
			\end{equation*}
			Mas o trabalho associado a um gás é igual a $p\Delta V$, portanto $\tau=0$. Pela Primeira Lei da Termodinâmica temos
			\begin{align*}
				 \Delta U & = Q-\overbracket{\tau}^{\alert{0}},\\
				\Aboxed{\Delta U & = Q.}
			\end{align*}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.4]{figuras/figura-7.png}
				\caption*{\footnotesize Exemplo de transformação isocórica.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

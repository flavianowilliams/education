\section{Lei de Biot-Savart}

\subsection{}

\begin{frame}{Observação experimental do campo magnético ao redor de um fio retilíneo}
	\begin{columns}
		\begin{column}{0.55\textwidth}
			De acordo com a experiência, a corrente que percorre um fio condutor produz um campo magnético circular $\vec{B}$, que obedece a seguinte relação de proporcionalidade
			\begin{equation*}
				B\; \alpha\; \frac{i}{R}.
			\end{equation*}
			Como será mostrado a seguir, a relação de proporcionalidade nos fornece
			\begin{align*}
				B=\frac{\mu_0i}{2\pi R}
			\end{align*}
		\end{column}
		\begin{column}{0.45\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.27]{figuras/magnetismo-fio.png}
				\caption*{\footnotesize Campo magnético de limalhas de ferro ao redor de um fio por onde atravessa uma corrente elétrica i.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Campo magnético produzido por uma corrente}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			Um elemento de corrente $id\vec{s}$ produz um elemento de campo magnético $d\vec{B}$ no ponto P, segundo a relação abaixo
			\begin{align*}
				d\vec{B} = \frac{\mu_0}{4\pi}\frac{id\vec{s}\times \vec{r}}{r^2}.
			\end{align*}
			Na figura ao lado, o campo magnético possui a orientação de estar entrando na tela. $\mu_0$ é chamada permeabilidade magnética no vácuo.
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.4]{figuras/biot-savart.png}
				\caption*{\footnotesize Campo magnético $d\vec{B}$ produzido pelo elemento de corrente $id\vec{s}$ \cite{halliday}.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Campo magnético de um fio retilíneo à partir da Lei de Biot-Savart}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			O módulo do campo magnético $d\vec{B}$ no ponto P mostrado na figura ao lado é dado por
			\begin{align*}
				dB = \frac{\mu_0}{4\pi}\frac{ids\sin\theta}{r^2}.
			\end{align*}
			Supondo um fio retilíneo infinito, podemos somar a contribuição de cada elemento $d\vec{s}$ do fio no ponto P integrando a equação acima,
			\begin{align*}
				B = \int\limits_{0}^{\infty} \frac{\mu_0 i}{4\pi} \frac{\sin \theta ds}{r^2}.
			\end{align*}
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.3]{figuras/fio-retilineo.png}
				\caption*{\footnotesize Campo magnético $d\vec{B}$ no ponto P devido ao elemento $d\vec{s}$ de um fio retilíneo \cite{halliday}.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Campo magnético de um fio retilíneo à partir da Lei de Biot-Savart (continuação)}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			A relação entre as variáveis r e $\theta$ com o comprimento s é dado por
			\begin{align*}
				r & = \sqrt{s^2+R^2},\\
				\sin\theta & = \frac{R}{\sqrt{s^2+R^2}}.
			\end{align*}
			Substituindo temos
			\begin{align*}
				B & = \frac{\mu_0 i}{4\pi} \int\limits_{0}^{\infty} \frac{R ds}{(s^2+R^2)^{3/2}},
			\end{align*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			Fazendo a substituição $s=R\tan\phi$ e $ds=R\sec^2\phi d\phi$ temos
			\begin{align*}
				B & = \frac{\mu_0 iR}{4\pi} \int^{\frac{\pi}{2}}_{0} \frac{R\sec^2\phi d\phi}{R^3\sec^3\phi},\\
				B & = \frac{\mu_0 i}{4\pi R} \int^{\frac{\pi}{2}}_{0}\cos\phi d\phi.
			\end{align*}
			Aplicando os limites de integração temos
			\begin{align*}
				\Aboxed{B = \frac{\mu_0i}{4\pi R}.}
			\end{align*}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Força magnética entre dois fios}
A corrente $i_1$ produzirá um campo magnético $B_1$ no fio 1. A força magnética atuando no fio 2 é $F_m=BiL$. No caso do fio retilíneo temos $B=\frac{\mu_0i}{2\pi r}$, portanto
\begin{columns}
	\begin{column}{0.6\textwidth}
		\begin{align*}
		F_m & = BiL,\\
		F_m & = \left(\frac{\mu_0i}{2\pi r}\right)iL.
		\end{align*}
		Se r=1m, i=1A, e L=1m a força $F_m$ encontrada equivale a \SI{2e-7}{N}. Substituindo temos
		\begin{equation*}
		\boxed{\mu_0 = 4\pi\times 10^{-7}\;\frac{N}{A^2}.}
		\end{equation*}
	\end{column}
	\begin{column}{0.4\textwidth}
		\begin{figure}
			\centering
			\includegraphics[scale=0.7]{figuras/forca-magnetica.jpg}
			\caption*{\footnotesize Força magnética $F_m$ atuando entre dois fios retilíneo \cite{osfundamentosdafisica}.}
		\end{figure}
	\end{column}
\end{columns}
\end{frame}

\section{A Lei de Ampère}

\subsection{}

\begin{frame}{Integral do campo magnético circular}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			Sabemos que a corrente que percorre um fio retilíneo produz campos magnéticos circulares. Portanto é de se esperar que a integral de $\vec{B}$ em um caminho fechado $C$ deve ser diferente de zero,
			\begin{align*}
				\oint_C \vec{B}\cdot d\vec{r}\neq 0.
			\end{align*}
		\end{column}
		\begin{column}{0.5\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.48]{figuras/linhas-campo-2.png}
				\caption*{\footnotesize Linhas de campo formando caminhos fechados.}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		A inexistência de monopolos magnéticos explica o fato das linhas de campo magnético serem obrigatoriamente circulares.
	\end{corollary}
\end{frame}

\begin{frame}{A Lei de Ampère}
	\begin{columns}
		\begin{column}[c]{0.65\textwidth}
			A experiência nos diz que a integral do campo magnético em um caminho fechado $C$ (\alert{chamada amperiana}) é proporcional à intensidade de corrente estacionária i envolvida pelo caminho $C$,
			\begin{align*}
				\oint_C \vec{B}\cdot d\vec{r}\;\alpha\; i.
			\end{align*}
			A constante de proporcionalidade é justamente a constante de permeabilidade magnética $\mu_0$, portanto
			\begin{align*}
				\Aboxed{\oint_C \vec{B}\cdot d\vec{r}=\oint_C B\cos\theta dr=\mu_0i.}
			\end{align*}		
		\end{column}
		\begin{column}[c]{0.35\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.35]{figuras/lei-ampere.png}
				\caption*{\footnotesize Aplicação da lei de Ampère a uma amperiana que envolve dois fios retilíneos longo \cite{halliday}.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Campo magnética induzido por um fio retilíneo}
	\begin{columns}
		\begin{column}{0.7\textwidth}
			Aplicando a lei de Ampere para uma amperiana circular de raio R que envolve um fio retilíneo longo por onde passa uma corrente i teremos
			\begin{equation*}
				\oint_{C} B\cos\theta ds=\mu_0i.
			\end{equation*}
			Pela simetria do problema, percebe-se que $B$ é constante para cada elemento $d\vec{s}$ da amperiana, e convenientemente escolhemos $\theta=\SI{0}{\degree}$, de modo que iremos calcular o campo magnético circular ao redor do fio.
		\end{column}
		\begin{column}{0.3\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.3]{figuras/lei-ampere-fio.png}
				\caption*{\footnotesize Uso da lei de Ampere para determinar o campo magnético ao redor de um fio retilíneo longo \cite{halliday}.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Campo magnética induzido por um fio retilíneo (continuação)}
	\begin{multicols}{2}
		Dessa maneira, podemos dizer que
		\begin{align*}
			\oint_{C} B\cos\theta ds & = \mu_0i,\\
			B \oint_{C} ds & = \mu_0i.
		\end{align*}
		Sabendo que $C=2\pi r$ temos
		\begin{align*}
			B(2\pi r) & =\mu_0 i,\\
			\Aboxed{B & = \frac{\mu_0i}{2\pi r}.}
		\end{align*}
	\end{multicols}
	\begin{corollary}
		A solução acima corresponde ao dobro do resultado obtido pela lei de Biot-Savart para a solução do campo magnético ao redor de um fio semi-infinito, devido ao fato da lei de Ampère ser aplicada para o caso de um fio infinito.
	\end{corollary}
\end{frame}

\begin{frame}{Campo magnético induzido por um cilindro}
\begin{columns}
	\begin{column}[c]{0.6\textwidth}
		Para o caso de um cilindro circular de densidade de corrente $\vec{J}$, onde $i=\pi r^2J$, se a amperiana for uma circunferência de raio r, onde $r<R$ teremos
		\begin{align*}
			B\oint_{C} ds & = \mu_0\overbracket{\pi r^2J}^{\tiny \alert{i}},\\
			B(2\pi r) & = \mu_0\pi r^2J.
		\end{align*}
		sendo $J=\frac{i}{\pi R^2}$, onde R é o raio do cilindro, temos
		\begin{align*}
			\Aboxed{B = \frac{\mu_0i}{2\pi R^2}r.}
		\end{align*}
	\end{column}
	\begin{column}[c]{0.4\textwidth}
		\begin{figure}
			\centering
			\includegraphics[scale=0.3]{figuras/lei-ampere-cilindro.png}
			\caption*{\footnotesize Uso da lei de Ampere para determinar o campo magnético ao redor de um fio retilíneo longo \cite{halliday}.}
		\end{figure}
	\end{column}
\end{columns}
\end{frame}

\begin{frame}{Campo magnético induzido por um cilindro (continuação)}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Podemos resumir a solução encontrada para o cilindro maciço na forma abaixo,
			\begin{equation*}
				\boxed{B(r) = 
				\begin{cases}
					\frac{\mu_0i}{2\pi R^2}r,\quad & (r<R),\\
					\frac{\mu_0i}{2\pi r},\quad & (r\ge R).

					\frac{\mu_0i}{2\pi r},\quad & (r\ge R).

\section{Aplicações}

\subsubsection{}

\begin{frame}{Campo magnético em uma espira circular}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Considere uma espira circular de raio R onde o objetivo é determinar o campo magnético $\vec{B}$ a uma altura $z$ do centro da espira. Pela simetria do problema é conveniente usarmos a lei de Biot-Savart ao invés da lei de Ampère. Pela figura podemos ver que o campo $d\vec{B}$ produzido pelo elemento de corrente $d\vec{s}$ é dado por
			\begin{align*}
				dB = \frac{\mu_0}{4\pi}\frac{ids\sin\frac{\pi}{2}}{r^2}=\frac{\mu_0}{4\pi}\frac{ids}{r^2}.
			\end{align*}
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.25]{figuras/campo-espira.png}
				\caption*{\footnotesize Campo magnético $\vec{B}$ no ponto P a uma distância z do centro de uma espira circular de raio R \cite{halliday}.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Campo magnético em uma espira circular (continuação)}
	\begin{multicols}{2}
		Podemos ver que as componentes $d\vec{B}$ se cancelam mutuamente, restando apenas a componente $d\vec{B}$, onde
		\begin{align*}
			dB = dB\cos\alpha,
		\end{align*}
		Analisando a figura podemos ver que
		\begin{align*}
			r^2 & = z^2+R^2,\\
			\cos\alpha & = \frac{R}{\sqrt{z^2+R^2}}.
		\end{align*}
		Substituindo na integral temos
		\begin{align*}
			dB & = \left(\frac{\mu_0}{4\pi}\frac{ids}{r^2}\right)\left(\cos\alpha\right),\\
			dB & = \left(\frac{\mu_0}{4\pi}\frac{ids}{\left(z^2+R^2\right)}\right)\left(\frac{R}{\sqrt{z^2+R^2}}\right),\\
			dB & = \underbracket{\frac{\mu_0iR}{4\pi\left(z^2+R^2\right)^{3/2}}}_{\text{\alert{constante}}}ds.
		\end{align*}
	\end{multicols}
\end{frame}

\begin{frame}{Campo magnético em uma espira circular (continuação)}
%	\begin{multicols}{2}
	\begin{columns}
		\begin{column}[c]{0.7\textwidth}
		Integrando a equação anterior teremos
		\begin{align*}
			B & = \left(\frac{\mu_0iR}{4\pi\left(z^2+R^2\right)^{3/2}}\right)\overbracket{(2\pi R)}^{\tiny \alert{s}}.
		\end{align*}
		Se $z\gg R$ podemos dizer que $z^2+R^2\approx z^2$, assim
		\begin{align*}
			B & = \frac{\mu_0}{2\pi}\frac{i\pi R^2}{z^3},\\
			\Aboxed{\vec{B} & = \frac{\mu_0}{2\pi}\frac{\vec{\mu}}{z^3}.}
		\end{align*}
		onde $\mu=i\pi R^2$ é o momento magnético da espira.
	\end{column}
	\begin{column}[c]{0.3\textwidth}
		\begin{figure}
			\centering
			\includegraphics[scale=0.3]{figuras/dipolo-magnetico.png}
			\caption*{\footnotesize Dipolo magnético produzido por uma espira circular \cite{halliday}.}
		\end{figure}
	\end{column}
%	\end{multicols}
\end{columns}
\end{frame}

	Podemos representar um solenóide como uma bobina helicoidal formada por espiras circulares muito próximas uma da outra. Sabemos que cada espira circular irá produzir um campo magnético ao seu redor, o campo resultante em cada ponto no espaço será a somatória do campo produzido por cada espira,

		\hspace*{1cm}
		\begin{subfigure}{0.45\textwidth}
			\centering
			\includegraphics[scale=0.2]{figuras/solenoide-2.jpg}

\begin{frame}{Campo magnético de um solenóide}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			Considere uma amperiana retangular como mostra a figura ao lado, aplicando a lei de Ampere podemos separar a contribuição da integral em quatro partes distintas,
			\begin{align*}
				\oint_C \vec{B}\cdot d\vec{s} & = \int_{a}^{b} \vec{B}\cdot d\vec{s} + \int_{b}^{cb} \vec{B}\cdot d\vec{s} +\\
				& + \int_{c}^{d} \vec{B}\cdot d\vec{s} + \int_{d}^{a} \vec{B}\cdot d\vec{s}.
			\end{align*}
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.5]{figuras/solenoide-3.png}
				\caption*{\footnotesize Aplicação da lei de Ampère a um solenóide usando como amperiana o retângulo abcda \cite{halliday}.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Campo magnético de um solenóide (continuação)}
	\begin{multicols}{2}
		A primeira integral fornece
		\begin{align*}
			Bh = \mu_0Ni,
		\end{align*}
		onde N é o número de espiras envolvidas pela amperiana. A segunda e a quarta integrais são nulas porque $\vec{B}$ é perpendicular a $d\vec{s}$, portanto $\vec{B}\cdot d\vec{s}=0$. A terceira integral está associada ao fato que o campo magnético fora do solenóide é zero, pois o campo magnético produzido na parte inferior praticamente anula o campo magnético produzido na parte superior. Portanto, resta-nos apenas a seguinte equação
		\begin{align*}
			Bh = \mu_0 Ni.
		\end{align*}
		Definindo n o número de espiras por unidade de comprimento do solenóide, podemos dizer que $N=nh$, assim
		\begin{align*}
			B\bcancel{h} & = \mu_0 n\bcancel{h}i.\\
			\Aboxed{B & = \mu_0 ni.}
		\end{align*}
	\end{multicols}
\end{frame}

\begin{frame}{Campo magnético de um toróide}
	\begin{columns}
		\begin{column}[c]{0.7\textwidth}
			Podemos imaginar um toróide como um solenóide que foi encurvado até as extremidades se tocarem, formando um anel. Assim como no caso do solenóide, o campo magnético está todo confinado em seu interior. Pela simetria do problema, é conveniente usarmos a lei de Ampère para o cálculo o campo. Considerando uma amperiana de raio r que envolve todas as N espiras do toróide temos
			\begin{align*}
				B(2\pi r) & = \mu_0 Ni,\\
				\Aboxed{B & = \frac{\mu_0 Ni}{2\pi r}.}
			\end{align*}
		\end{column}
		\begin{column}[c]{0.3\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.3]{figuras/toroide.png}
				\caption*{\footnotesize Seção reta horizontal de um toróide \cite{halliday}.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{A antiga teoria quântica}
	\begin{enumerate}
		\item [\ding{51}] \textit{Numa reunião da sociedade alemã de física em 1900, Max Planck apresentou o seu artigo "Sobre a teoria da lei de distribuição de energia do espectro normal". \alert{Esse dia marca o nascimento da física quântica}.}
		\item[\ding{51}] \textit{Até o surgimento da equação de Schroedinger, diversos estudos foram desenvolvidos demonstrando falhas na física clássica. Esses estudos, \alert{chamados de antiga teoria quântica}, marcam os fundamentos da física quântica atual.}
		\item[\ding{51}] \textit{Assim como a teoria da relatividade, a física quântica representa uma generalização da física clássica, que inclui as leis clássicas como casos especiais.}
	\end{enumerate}
	\begin{corollary}
		Os fenômenos ligados a física quântica abrangem todas as áreas da física clássica: mecânica, termodinâmica, ondas, mecânica estatística e eletromagnetismo.
	\end{corollary}
\end{frame}

\begin{frame}{Lei de Stefan-Boltzman}
	\begin{enumerate}
		\item [\ding{51}] \textit{A radiação (\alert{intensidade}) incidente aumenta a vibração dos átomos, aumentando a temperatura do corpo negro;}
		\item [\ding{51}] \textit{A radiação emitida somente depende da temperatura do corpo negro;}
		\item [\ding{51}] \textit{De acordo com a teoria clássica, a radiação aumenta indefinidamente com a frequência da radiação emitida.}
	\end{enumerate}
	Chegando a lei de Stefan-Boltzman que relaciona a radiação emitida por um objeto com a sua temperatura T,
	\begin{align*}
		R(T) = \sigma T^4.
	\end{align*}
	onde $\sigma=\SI{5,6704e-8}{\watt\per\square\meter\kelvin}$
\end{frame}

\begin{frame}{Radiância e radiação emitida pelo material}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
		Pela teoria do eletromagnetismo, a \alert{radiância}, ou seja, radiação emitida por cada onda de frequência $\nu$ ($R(\nu)$), que sai da cavidade de um corpo negro é dado por
		\begin{align*}
			R(\nu) = \frac{c}{4} u(\nu),
		\end{align*}
		onde $u(\nu)$ é a energia da radiação emitida por volume. Neste caso, a densidade de energia na faixa de radiação com frequências entre $\nu$ e $\nu+d\nu$ vale $u(\nu)d\nu$. Portanto, a quantidade de radiação dR
		\end{column}
		\begin{column}[t]{0.5\textwidth}	
			 neste intervalo vale
	 		\begin{align*}
				dR & = R(\nu)d\nu,\\
				dR & = \frac{c}{4}u(\nu)d\nu.
			\end{align*}
			Para determinar a radiação total emitida pelo corpo negro, integramos a equação acima,
			\begin{align*}
				R = \frac{c}{4}\int\limits_{0}^{\infty}u(\nu)d\nu.
			\end{align*}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Valor médio da energia para uma distribuição contínua de energia}
	\begin{multicols*}{2}
		Para determinar o valor médio da energia $\left<E\right>$, usamos a expressão abaixo
		\begin{align*}
			\left<E\right> & = \frac{\int_{0}^{\infty}Ef(E)}{\int_{0}^{\infty}f(E)}dE,\\
			f(E) & = Ae^{-E/k_BT}dE.
		\end{align*}
		onde $f(E)$ é conhecida como distribuição de Maxwell-Boltzmann e $k_B$ é chamado constante de Boltzmann.	Substituindo na expressão teremos
		\begin{align*}
			\left<E\right> & = \frac{\int_{0}^{\infty}\bcancel{A}Ee^{-E/k_BT}dE}{\int_{0}^{\infty}\bcancel{A}e^{-E/k_BT}dE},\\
			\left<E\right> & = \frac{\int_{0}^{\infty}Ee^{-E/k_BT}dE}{\int_{0}^{\infty}e^{-E/k_BT}dE}.
		\end{align*}
		É possível provar que o resultado da equação acima equivale a
		\begin{align*}
			\Aboxed{\left<E\right> = k_BT.}
		\end{align*}
	\end{multicols*}
\end{frame}

\begin{frame}{A equação de Rayleigh-Jeans}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Considerando uma distribuição contínua de energia para a radiação emitida pelo corpo negro, teremos que a densidade de energia $u(\nu)$ é dado por
			\begin{align*}
				u(\nu) & = \left<E\right>n(\nu);\\
				u(\nu) & =\left(k_BT\right)\left(\frac{8\pi\nu^2}{c^3}\right).
			\end{align*}
			Substituindo na fórmula da radiação,
			\begin{align*}
				R = \frac{c}{4}\int_{0}^{\infty}u(\nu)d\nu.
			\end{align*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
		\begin{align*}
			R & = \frac{\bcancel{c}}{4}\int_{0}^{\infty}\frac{8\pi k_BT}{\bcancel{c}^3}\nu^2d\nu,\\
			R & = \frac{8\pi k_BT}{c^2}\int_{0}^{\infty}\nu^2d\nu,\\
			R & = \left.\frac{8\pi k_BT}{3c^2}\nu^3\right|_{0}^{\infty}=\infty.
		\end{align*}
		Vemos que se considerarmos uma distribuição contínua de energia para $u(\nu)$, quando $\nu\rightarrow \infty$ teremos $R\rightarrow\infty$, o que resulta na catástrofe do ultravioleta.
		\end{column}
	\end{columns}
%	\end{multicols}
\end{frame}

\begin{frame}{A lei de Planck e o nascimento da física quântica}
	Para explicar a catástrofe do ultravioleta, duas causas são possíveis
	\begin{enumerate}
		\item [\ding{51}] \textit{Contagem errada do número de estados $n(\nu)$;}
		\item[\ding{51}] \textit{Valor da energia para cada modo vibracional está errado.}
	\end{enumerate}
	\begin{block}{Hipótese de Planck}
		Como opção, Planck sugeriu que a energia das cargas oscilantes ao invés de assumir qualquer valor, ela deverá ter valores discretos bem definidos, e também deverá ser proporcional a frequência da radiação emitida,
		\begin{align*}
			E_n = nh\nu,\; n = 0,1,2,\cdots
		\end{align*}
		h é conhecido como a \alert{constante de Planck}.
	\end{block}
\end{frame}

\begin{frame}{Valor médio da energia para uma distribuição discreta de energia}
%	\begin{multicols*}{2}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
		O valor médio da energia $\left<E\right>$ para uma distrição discreta de energias é dado por
		\begin{align*}
			\left<E\right> & = \frac{\sum_{n=0}^{\infty} E\left(\bcancel{A}e^{-E/k_BT}\right)}{\sum_{n=0}^{\infty}\left(\bcancel{A}e^{-E/k_BT}\right)},\\
			\left<E\right> & = \frac{\sum_{n=0}^{\infty} Ee^{-E/k_BT}}{\sum_{n=0}^{\infty}e^{-E/k_BT}}.
		\end{align*}
		Como temos valores discretos de energia, a integral é substituída por uma somatória de valores de energia $E_n$. É possível provar que o resultado equivale a
		\end{column}
		\begin{column}[c]{0.5\textwidth}
		\begin{align*}
			\Aboxed{\left<E\right> = \frac{h\nu}{e^{\frac{h\nu}{k_BT}}-1}.}
		\end{align*}
		Substituindo em $u(\nu)$ teremos
		\begin{align*}
			\Aboxed{u(\nu) = \frac{8\pi h\nu^3}{c^3\left(e^{\frac{h\nu}{k_BT}}-1\right)}.}
		\end{align*}
		O resultado acima em $R(\nu)$ reproduz perfeitamente os dados experimentais.
		\end{column}
	\end{columns}
%	\end{multicols*}
\end{frame}

\begin{frame}{Lei de Planck para baixas frequências}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
		Pela teoria clássica temos que a teoria se aproxima do resultado experimental quando $\nu <<1$, usando a expansão
		\begin{align*}
			e^{\frac{h\nu}{k_BT}} = 1+\frac{h\nu}{k_BT}+\frac{1}{2}\left(\frac{h\nu}{k_BT}\right)^2+\cdots,
		\end{align*}
		podemos dizer que $e^{\frac{h\nu}{k_BT}}\approx 1$ se $h\nu <<k_BT$. Substituindo em $u(\nu)$ resulta em
		\end{column}
		\begin{column}[t]{0.5\textwidth}
		\begin{align*}
			u(\nu) & \approx \frac{8\pi \bcancel{h}\nu^{\bcancel{3}}}{c^3\left(\bcancel{1}+\frac{\bcancel{h}\bcancel{\nu}}{k_BT}-\bcancel{1}\right)},\\
			u(\nu) & \approx \frac{8\pi k_BT\nu^2}{c^3},
		\end{align*}
		o que corresponde a lei de Rayleigh-Jeans da teoria clássica da radiação.
		\end{column}
	\end{columns}
	\begin{corollary}
		Para $h\nu/k_BT<<1$ a equação de Planck se resume na lei de Rayleigh-Jeans.
	\end{corollary}
\end{frame}

\begin{frame}{Efeito fotoelétrico}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Vamos considerar um equipamento por onde incide uma luz de determinada frequência no alvo T de um metal específico. A experiência mostra que os elétrons são ejetados do material gerando uma corrente i que pode ser registrada pelo amperímetro A. \\
			Uma diferença de potencial V é ajustada entre os terminais do aparelho com a intenção de frear os elétrons até pararem, registrando assim uma corrente zero no amperímetro. Dessa maneira, a energia cinética máxima K deve ser igual a $eV_F$.
		\end{column}

\begin{frame}{O que era esperado pela teoria clássica}
%	\begin{multicols}{2}
		De acordo com a teoria do eletromagnetismo, a intensidade da onda I eletromagnética é dado por $I=\frac{E^2_m}{2\mu_0c}$, ou seja, depende somente da amplitude do campo elétrico $E_m$ e não da frequência da luz. Além do mais, como a intensidade é potência por área, era de se esperar que o metal absorvesse cada vez ao longo do tempo. Assim o elétron teria energia cinética o suficiente para escapar do material. Portanto, era de se esperar que
		\begin{enumerate}
			\item [\ding{51}] \textit{a energia cinética dos elétrons deveria depender da intensidade da onda;}
			\item [\ding{51}] \textit{o efeito fotoelétrico deveria ocorrer com a luz de qualquer frequência;}
			\item [\ding{51}] \textit{deveria haver um retardo de tempo, de modo que o elétron absorvesse energia do feixe continuamente.}
		\end{enumerate}
%	\end{multicols}
\end{frame}

\begin{frame}{O que foi observado experimentalmente}
	\begin{columns}
		\begin{column}[c]{0.65\textwidth}
			Foi observado que a energia cinética K, onde
			\begin{align*}
				K = eV_F,
			\end{align*}
			independe da luz incidente. Aumentando a intensidade, apenas aumenta a corrente no circuito, mas o potencial de corte $V_F$ permanece o mesmo.
		\end{column}
		\begin{column}[c]{0.35\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.5]{figuras/efeito-fotoeletrico-1.png}
				\caption*{\footnotesize Variação da corrente com o potencial.}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		O efeito fotoelétrico independe da intensidade da luz incidente.
	\end{corollary}
\end{frame}

\begin{frame}{O que foi observado experimentalmente}
	\begin{columns}
		\begin{column}[c]{0.65\textwidth}
			Foi observado que para cada material existe um limiar de frequência $\nu_0$. Caso a frequência da luz incidente for menor que $\nu_0$, o efeito fotoelétrico não ocorre para aquele material. A figura apresenta um metal alcalino, onde para cada luz incidente, existe um potencial de corte. Nesse caso, o efeito fotoelétrico deixaria de ocorrer para a luz vermelha, que possui frequência menor que a amarela ($\nu_{verm}=\SI{0,4e15}{\per\second}$).
		\end{column}
		\begin{column}[c]{0.35\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.5]{figuras/efeito-fotoeletrico-2.png}
				\caption*{\footnotesize Variação da corrente para diversos valores da frequência da luz.}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		Para frequências menores que $\nu_0$ o efeito fotoelétrico não ocorre, qualquer que seja a intensidade da iluminação.
	\end{corollary}
\end{frame}

\begin{frame}{Hipótese de Einstein}
	Para explicar as divergências observadas no efeito fotoelétrico, Einstein propôs que a luz é constituídas por pacotes de energia chamada fóton, onde cada fóton carrega a quantidade de energia
	\begin{align*}
		\Aboxed{E = h\nu.}
	\end{align*}
	Assim, a energia cinética K dos elétrons que saem do material é dado por
	\begin{align*}
		K = eV_F = h\nu-\phi.
	\end{align*}
	$\phi$ é denominado função trabalho, que representa a energia necessária para remover o elétron do material.
\end{frame}

\begin{frame}{Explicações plausíveis para o efeito fotoelétrico}
	\begin{enumerate}
		\item [\ding{51}] \textit{O efeito fotoelétrico independe da intensidade da luz incidente.} \alert{Um aumento na intensidade significa mais fótons com a mesma energia $h\nu$ colidindo com elétrons diferentes, o que justifica o aumento na corrente elétrica. Mas se a energia de cada fóton não equivaler a função trabalho, os elétrons não conseguem escapar do material independente da quantidade fótons.}
		\item [\ding{51}] \textit{Para frequências menores que $\nu_0$ o efeito fotoelétrico não ocorre, qualquer que seja a intensidade da iluminação.} \alert{Na colisão dos fótons com os elétrons, uma energia equivalente a $h\nu$ é absorvida pelo elétron. Se essa energia não equivaler a função trabalho, o elétron não consegue escapar do material.}
		\item [\ding{51}] \textit{Assim que a luz incide no metal, os elétrons são imediatamente removidos, não havendo um retardo de tempo.} \alert{Na colisão, a energia dos fótons é imediatamente absorvida, não havendo a necessidade de mais colisões.}
	\end{enumerate}
\end{frame}

\begin{frame}{Como obter a constante de Planck e o limiar de frequência?}
	\begin{columns}
		\begin{column}[c]{0.7\textwidth}
			Isolando o potencial de corte $V_F$ anteriormente teremos
			\begin{align*}
				eV_F & = h\nu - \phi,\\
				V_F & = \frac{h}{e}\nu-\frac{\phi}{e}.
			\end{align*}
			Considerando $V_F$ como função da frequência da luz incidente, podemos representá-la em um gráfico $V_F$ versus $\nu$, onde o coeficiente angular da reta representa o valor da constante de Planck,

				\Aboxed{h = \SI{6,57e-34}{\joule\second}.}
			\end{align*}

\section{Modelo cinético}

\subsection{}

\begin{frame}{O gás ideal}
	\begin{enumerate}
		\item [\ding{51}] O gás é constituído de moléculas de massa e volume desprezíveis em relação à distância de separação das mesmas;
		\item[\ding{51}] O número de moléculas no gás é muito grande e elas estão distribuídas uniformemente em um elemento de volume $\Delta V$;
		\item[\ding{51}] O movimento das moléculas obedece às leis de Newton e tem movimento desordenado devido às colisões;
		\item[\ding{51}] O volume ocupado por cada molécula deve ser muito menor que o volume do recipiente, no qual elas são mantidas;
		\item[\ding{51}] As colisões entre as moléculas são elásticas, com duração e alcance desprezíveis;
		\item[\ding{51}] As direções e os sentidos das velocidades das moléculas estão distribuídas uniformemente.
	\end{enumerate}
\end{frame}

\begin{frame}{Pressão de um gás ideal}
	\begin{columns}
		\begin{column}[c]{0.7\textwidth}
			Cada partícula poderá colidir com as paredes de um recipiente que a contém. 			\begin{align*}
				F_{\text{médio}} = \frac{\Delta p}{\Delta t}=-\frac{2mv_x}{\Delta t}.
			\end{align*}
			Se a colisão acontece em um cubo de aresta L teremos
			\begin{align*}
				\Delta t = \frac{2L}{v_x}.

\begin{frame}{Pressão de um gás ideal}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			A pressão devido a cada colisão é dado por $p=\frac{F}{A}$, onde $A$ é a área do parede compreendido pelo cubo, assim
			\begin{align*}
				P & = \frac{1}{A}\frac{2mv_x}{\Delta t},\\
				P & = \frac{\bcancel{2}}{L^2}\left(\frac{mv^2_x}{\bcancel{2}L}\right).
			\end{align*}
			Para o caso de N partículas teremos
			\begin{align*}
				\sum P & = \sum\limits_{i=1}^{N}\frac{m_iv^2_x}{L^3} = \frac{m}{L^3}\sum\limits_{i=1}^{N}v^2_x.
			\end{align*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			Dividindo e multiplicando por N partículas temos o valor da pressão, dado por
			\begin{align*}
				\frac{\sum P}{\alert{N}} & = \frac{m}{L^3}\left(\frac{1}{\alert{N}}\sum\limits_{i=1}^{N}v^2_x\right) = \frac{m}{L^3}\left<v^2_x\right>.
			\end{align*}
			Considerando que $v^2_x\approx v^2_y\approx v^2_z\approx \frac{1}{3}v^2$, e que $\rho$ é a densidade do gás, teremos
			\begin{align*}
				\Aboxed{\left<P\right>=\frac{1}{3}\rho \left<v^2\right>.}
			\end{align*}
			onde $\rho$ é a densidade do gás.
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Valor médio da energia cinética de cada partícula}
%	\begin{multicols}{2}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Da equação do gás ideal temos para um gás \alert{monoatômico} que $PV=nRT$. Substituindo a expressão da pressão teremos
			\begin{align*}
				nRT & = V\left(\frac{1}{3}\rho \left<v^2\right>\right).\\
				nRT & = \bcancel{V}\left(\frac{1}{3}\frac{Nm}{\bcancel{V}} \left<v^2\right>\right),
			\end{align*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			\vspace*{-0.25cm}
			\begin{equation*}
				3nRT = Nm\left<v^2\right>.
			\end{equation*}
			Dividindo por 2 poderemos identificar o termo do lado direito como a energia cinética média, onde $\left<K\right> = \frac{m\left<v^2\right>}{2}.$ Assim
			\begin{equation*}
				\boxed{\left<K\right> = \frac{3nRT}{2N}.}
			\end{equation*}
		\end{column}
	\end{columns}
	\begin{corollary}
		Em uma dada temperatura T, as moléculas de qualquer gás ideal, independentemente da massa de cada uma delas, têm a mesma energia cinética média.
	\end{corollary}
%	\end{multicols}
\end{frame}

\begin{frame}{Energia interna}
	Se a energia associada a qualquer grau de liberdade é uma função quadrática da variável que especifica, o grau de liberdade, o valor médio da energia correspondente a este grau de liberdade é igual a
	\begin{equation*}
		\boxed{\text{f graus de liberdade e N moléculas}.}
	\end{equation*}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			A energia interna de um gás pode ser determinada multiplicando a energia cinética média pelo número de moléculas N,
			\begin{align*}
				U & = N\left<K\right>,\\
				U(T) & \Rightarrow \bcancel{N}\frac{nRT}{2\bcancel{N}} = \frac{nRT}{2}.
			\end{align*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			Considerando os graus f de liberdade de cada molécula e sabendo que $n=NN_0$, sendo $N_0$ o número de Avogadro e $k_B=N_0R$, temos
			\begin{equation*}
				\boxed{U(T) =\frac{f}{2}nRT= \frac{f}{2}Nk_BT.}
			\end{equation*}
		\end{column}
	\end{columns}
\end{frame}

\subsection{}

\begin{frame}{Energia interna de um gás diatômico}
	Uma molécula diatômica pode apresentar os seguintes graus de liberdade:
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			\begin{enumerate}
				\item [\ding{51}] Translação:
				\begin{equation*}
					K_{\text{trans}}=\frac{1}{2}mv_x^2+\frac{1}{2}mv_y^2+\frac{1}{2}mv_z^2=\frac{3}{2}Nk_BT.
				\end{equation*}
				\item [\ding{51}] Rotação: 
				\begin{equation*}
					K_{\text{rot}}=\frac{1}{2}I_x\omega_x^2+\frac{1}{2}I_y\omega_y^2=\frac{\bcancel{2}}{\bcancel{2}}Nk_BT.
				\end{equation*}
				\item [\ding{51}] Vibração: $K_{\text{vib}}=\frac{k}{2}r^2=\frac{1}{2}Nk_BT$.
			\end{enumerate}
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.25]{figuras/graus-liberdade.png}
				\caption*{\footnotesize Graus de liberdade da molécula diatômica \cite{halliday}.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Calor específico a volume constante}
	\begin{columns}
		\begin{column}[c]{0.7\textwidth}
			É sabido experimentalmente que a temperatura de um objeto de massa m aumenta com a quantidade de calor que ele absorve ($dQ=ncdT$). Se a transformação acontece a volume constante implica que não há trabalho envolvido, e todo o calor é convertido em energia do gás ($dU=dQ_V$).
			\begin{equation*}
				dQ_V = dU = nc_VdT,
			\end{equation*}
			onde $c_V$ é calor específico a volume constante. Sabemos que $dU=\frac{f}{2}nRdT$,
			\begin{equation*}
				\boxed{c_V = \frac{f}{2}R,}
			\end{equation*}
		\end{column}
		\begin{column}[c]{0.3\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.3]{figuras/capacidade-termica.png}
				\caption*{\footnotesize Aumento da energia cinética das moléculas devido ao acréscimo $dK=dQ$ \cite{halliday}.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Calor específico a pressão constante}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
%	\begin{multicols}{2}
			Considere um processo termodinâmico que o ocorre a pressão constante, onde o gás recebe uma quantidade de calor $dQ_P=nc_PdT$. Pela primeira lei da termodinâmica, sabemos que
			\begin{NodesList}[margin=1.25cm]
				\begin{align*}
					dU & = dQ_P-dW,\AddNode\\
					dU & = nc_PdT-PdV,\AddNode
				\end{align*}
				\LinkNodes{\tiny $dQ_P=nc_PdT$}
			\end{NodesList}
			Mas vimos anteriormente que $dU=nc_VdT$. No entanto, pela equação do gás ideal temos $PV=nRT$, onde
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			\vspace*{-0.75cm}
			\begin{align*}
				d(PV) & = nRdT,\\
				PdV+VdP & = nRdT,\\
				PdV & = nRdT-VdP.
			\end{align*}
			Substituindo $PdV$ e $dU$ teremos
			\begin{align*}
				nc_VdT & = nc_PdT-(nRdT-VdP)
			\end{align*}
			No entanto, como o processo ocorre a pressão constante $dP=0$, portanto
			\begin{equation*}
				\boxed{c_V=c_P-R.}
			\end{equation*}
		\end{column}
	\end{columns}
%	\end{multicols}
\end{frame}

\begin{frame}{Distribuição de Velocidades de Maxwell-Boltzmann}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
%	\begin{multicols}{2}
			A distribuição de velocidades é uma função que expressa a fração de moléculas, cujas velocidades estão entre v+dV. Maxwell e Boltzmann definiram essa distribuição como
			\begin{equation*}
				\boxed{n(v) = 4\pi N\left(\frac{m}{2\pi k_BT}\right)^{3/2}v^2e^{-mv^2/k_BT}.}
			\end{equation*}
			A distribuição de Maxwell-Boltzmann obedece a seguinte condição abaixo,
			\begin{equation*}
				\int_{-\infty}^{\infty}n(v)dv = 1.
			\end{equation*}		
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}

\begin{frame}{Valor quadrático médio da velocidade}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			Em mecânica estatística usamos a expressão abaixo para calcular o valor médio de uma grandeza que possui valores contínuos,
			\begin{equation*}
				\left<A\right> = \frac{\int_{-\infty}^{\infty}n(A)AdA}{\int_{-\infty}^{\infty}n(A)dA}.
			\end{equation*}
			Considerando v\textsuperscript{2} como o valor a ser determinado, usamos a distribuição de Maxwell-Boltzmann para representar $n(v)$. Portanto
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\vspace*{-0.25cm}
			\begin{equation*}
				\left<v^2\right> = \int_{-\infty}^{\infty}4\pi N\left(\frac{m}{2\pi k_BT}\right)v^4e^{-mv^2/k_BT}dv,
			\end{equation*}
			onde usamos a condição $\int_{-\infty}^{\infty}n(v)dv = 1$. A integral acima possui resultado analítico, cujo valor é
			\begin{align*}
				\left<v^2\right> & = 3\frac{2k_BT}{2m},\\
				\frac{m}{2}\left<v^2\right> & = \frac{3}{2}k_BT,
			\end{align*}
			que resulta na energia interna do gás.
		\end{column}
	\end{columns}
\end{frame}

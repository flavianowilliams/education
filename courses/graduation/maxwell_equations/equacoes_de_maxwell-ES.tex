\begin{frame}{Equações de Maxwell}
		As principais equações do eletromagnetismo são
		\begin{align*}
			\oint\vec{E}\cdot d\vec{A} & = \frac{q}{\varepsilon_{0}}, & \quad\textrm{(Lei de Gauss)}\\
			\oint\vec{B}\cdot d\vec{A} & = 0, & \quad\textrm{(Lei de Gauss do magnetismo)}\\
			\oint\vec{E}\cdot d\vec{s} & = -\frac{d\Phi_{B}}{dt}, & \quad\textrm{(Lei de Faraday)}\\
			\oint\vec{B}\cdot d\vec{s} & = \mu_0i, & \quad\textrm{(Lei de Ampère)}\
		\end{align*}
\end{frame}

\begin{frame}{Equações de Maxwell}
	As equações de Maxwell são formadas pela combinação das quatro equações do eletromagnetismo:
	\begin{align*}
		\oint\vec{E}\cdot d\vec{A} & = \frac{q}{\varepsilon_{0}}, & \quad\textrm{(Lei de Gauss)}\\
		\oint\vec{B}\cdot d\vec{A} & = 0, & \quad\textrm{(Lei de Gauss do magnetismo)}\\
		\oint\vec{E}\cdot d\vec{s} & = -\frac{d\Phi_{B}}{dt}, & \quad\textrm{(Lei de Faraday)}\\
		\oint\vec{B}\cdot d\vec{s} & = \mu_0i+\alert{\mu_{0}\varepsilon_{0}\frac{d\Phi_{E}}{dt}}, & \quad\textrm{(Lei de Ampère-Maxwell)}\
	\end{align*}
\end{frame}

\section{Ondas eletromagnéticas}

\subsection{}

\begin{frame}{Equações de Maxwell no vácuo}
		\justify
		No vácuo temos ausência de cargas elétricas, o que resulta q=0 e i=0,
		\begin{align*}
			\oint\vec{E}\cdot d\vec{A} & = 0,\\
			\oint\vec{B}\cdot d\vec{A} & = 0,\\
			\oint\vec{E}\cdot d\vec{s} & = -\frac{d\Phi_{B}}{dt},\\
			\oint\vec{B}\cdot d\vec{s} & = \mu_{0}\varepsilon_{0}\frac{d\Phi_{E}}{dt}\
		\end{align*}
\end{frame}

\begin{frame}{Equação da onda à partir da lei de Faraday-Lenz}
	\begin{multicols}{2}
		\justify
		O fluxo do campo magnético é dado por
		\begin{align*}
			\Phi_{B} & = (B)(\underbracket{hdx}_{\alert{cte}}),\\
			\frac{d}{dt}\Phi_{B} & = hdx\frac{dB}{dt}
		\end{align*}
		Sabendo que $\frac{d\Phi_{B}}{dt} = -hdE$ temos
		\begin{align*}
			\alert{h}dE & = -\alert{h}dx\frac{dB}{dt},\\
			dE & = -\left[\frac{dB}{dt}\right]dx.
		\end{align*}
		Usando a regra de diferencial
		\[\boxed{\frac{\partial E}{\partial x}=\alert{-}\frac{\partial B}{\partial t}}\]
		A única função para $E(x,t)$ e $B(x,t)$ que satisfaz a equação acima seria
		\begin{align*}
			E(x,t) & = E_{m}cos\left(kx\alert{-}\omega t\right),\\
			B(x,t) & = B_{m}cos\left(kx\alert{-}\omega t\right)
		\end{align*}		
	\end{multicols}

\begin{frame}[label=derivadas]{Equação da onda à partir da lei de Ampère-Maxwell}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			O fluxo do campo elétrico é dado por
			\begin{align*}
				\Phi_{E} & = (E)(\underbracket{hdx}_{\alert{cte}}),\\
				\frac{d\Phi_{E}}{dt} & = hdx\frac{dE}{dt}.
			\end{align*}
			Sabendo que $\frac{d\Phi_{E}}{dt} = -\frac{1}{\varepsilon_{0}\mu_{0}}hdB$ temos
			\begin{align*}
				\frac{1}{\varepsilon_{0}\mu_{0}}\bcancel{h}dB = -\bcancel{h}\left[\frac{dE}{dt}\right]dx,
			\end{align*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			Usando a regra de diferencial
			\begin{align*}
				\Aboxed{\frac{\partial B}{\partial x}=\alert{-}\varepsilon_{0}\mu_{0}\frac{\partial E}{\partial t}.}
			\end{align*}
			\begin{corollary}
				\begin{align*}
					\frac{\partial E}{\partial t} & = -\frac{1}{\varepsilon_{0}\mu_{0}}\frac{\partial B}{\partial x}\\
					\frac{\partial E}{\partial x} & = -\frac{\partial B}{\partial t}
				\end{align*}
			\end{corollary}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Amplitude dos campos elétrico e magnético}
	\begin{columns}[c]
		\begin{column}[c]{0.5\textwidth}
			Supondo uma solução que satisfaz a equação anterior,\hyperlink{derivadas}{\beamerreturnbutton{Retornar ao slide anterior}}
			\begin{align*}
				E(x,t) & = E_{m}cos\left(kx-\omega t\right),\\
				B(x,t) & = B_{m}cos\left(kx-\omega t\right)
			\end{align*}	
			Usando a equação $\frac{\partial E}{\partial x} = -\frac{\partial B}{\partial t}$ e derivando
			\begin{align*}
				\frac{\partial E}{\partial x} & = \alert{-E_{m}k}sen\left(kx-\omega t\right)\\
				\frac{\partial B}{\partial t} & = \alert{B_{m}\omega} sen\left(kx-\omega t\right)
			\end{align*}
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			O que resulta
			\[B_{m}\omega = E_{m}k\]
			mas $\frac{\omega}{k}=c$,
			\[\boxed{B_{m}  = \frac{E_{m}}{c}}\]
			\begin{corollary}
				A intensidade do campo elétrico é muito maior que a do campo magnético.
			\end{corollary}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Velocidade da onda eletromagnética}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			Usando a equação $\frac{\partial E}{\partial t} = -\frac{1}{\varepsilon_{0}\mu_{0}}\frac{\partial B}{\partial x}$ e derivando
			\begin{align*}
				\frac{\partial E}{\partial t} & = \alert{E_{m}\omega}sen\left(kx-\omega t\right)\\
				\frac{\partial B}{\partial x} & = -\alert{B_{m}k} sen\left(kx-\omega t\right)
			\end{align*}
			O que resulta
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{align*}
				E_{m}\omega & = \frac{1}{\varepsilon_{0}\mu_{0}}kB_{m},\\
				\frac{E_{m}}{B_{m}} & = \frac{1}{\varepsilon_{0}\mu_{0}}\frac{k}{\omega}
			\end{align*}
			mas $\frac{E_{m}}{B_{m}}=c$ e numa onda $c=\frac{\omega}{k}$, teremos
			\begin{equation*}
				\boxed{c  = \frac{1}{\sqrt{\varepsilon_{0}\mu_{0}}}.}
			\end{equation*}
		\end{column}
	\end{columns}
	\begin{corollary}
		\small Os campos elétrico e magnético propagam no vácuo a uma velocidade c.
	\end{corollary}
\end{frame}

\begin{frame}{A luz como onda eletromagnética}
	Maxwell percebeu que a velocidade c obtida à partir do eletromagnetismo é exatamente idêntica a velocidade da luz, já bem conhecida ná época através de diversas técnicas de medição.\\
	\textit{\small "A velocidade das ondas transversais em nosso meio hipotético, calculada a partir dos experimentos electromagnéticos dos Srs. Kohrausch e Weber, concorda tão exactamente com a velocidade da luz, calculada pelos experimentos óticos do Sr. Fizeau, que é difícil evitar a inferência de que a luz consiste nas ondulações transversais do mesmo meio que é a causa dos fenômenos eléctricos e magnéticos."}
	\begin{corollary}
		A luz é uma onda eletromagnética capaz de se \alert{propagar no vácuo} com a velocidade de aproximadamente $3\times 10^8\;m/s$.% Isso é possível, pois podemos ter variações dos campos elétrico e magnético na ausência de matéria.
	\end{corollary}
\end{frame}

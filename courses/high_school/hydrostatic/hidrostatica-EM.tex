\begin{frame}{Variação da pressão com altitude e profundidade}
	\begin{multicols}{2}
		Pela relação da pressão p e força F, $p=\frac{F}{A}$, temos
		\begin{align*}
			F=(p)\times (A),
		\end{align*}
		portanto
		\begin{align*}
			F_1 & = p_1A,\\
			F_2 & = p_2A.\\
		\end{align*}
		Se o cilindro está em repouso, pela segunda Lei de Newton a força resultante deve ser zero, portanto
		\begin{align*}
			F_2 = F_1 + P.
		\end{align*}
		Pela definição de densidade, $m = \rho V$, e sabendo que o volume do cilindro é a base A vezes a altura h temos
		\begin{align*}
			p_2\bcancel{A} & = p_1\bcancel{A} + \rho gh\bcancel{A},\\
				\Aboxed{p_2 & = p_1 + \rho g h.}
		\end{align*}
	\end{multicols}
\end{frame}

\begin{frame}{Variação da pressão com a profundidade}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{block}{Lei de Stevin}
				\justifying Se a superfície de um fluido, cuja densidade é $\rho$, está submetida a uma pressão $p_{atm}$, a pressão p, no interior desse líquido, a uma profundidade h, é dada por
				\[p=p_{atm}+\rho gh\]
			\end{block}
			\begin{corollary}
				\justifying A força da gravidade puxa o fluido para baixo causando uma pressão na base e nas paredes do recipiente.
			\end{corollary}
		\end{column}

\begin{frame}{Variação da pressão com a altitude}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			\centering
			\begin{corollary}
				\justifying Para baixas altitudes ou profundidade a força da gravidade é praticamente constante, portanto a Lei de Stevin pode ser aplicada, mas para altas altitudes a força da gravidade diminui de modo que a pressão do ar varia de maneira praticamente exponencial com a altura.
			\end{corollary}
		\end{column}

\begin{frame}{Experiência de Torricelli}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\justifying Coloca-se mercúrio cuja densidade é conhecida num tubo fino e vira-o de cabeça para baixo. O líquido irá descer e irá preencher o recipiente da parte de baixo. A parte de cima como estava fechada não entrou ar e com a descida do líquido criou-se um vácuo, portanto a pressão da parte de cima será zero. Pela Lei de Stevin temos que a pressão da parte de baixo é dado por
			\begin{align*}
				p_{atm} = \rho g h,
			\end{align*}
			onde h é a coluna de mercúrio (se for medido ao nível do mar h=760mm).
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}[c]
				\includegraphics[scale=0.25]{figuras/experiencia-torricelli.jpg}
				\caption*{\tiny Representação da experiência de Torricelli.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Vasos comunicantes}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{corollary}
				\justifying Pela Lei de Stevin a variação da pressão em um fluido homogêneo ($\rho = constante$) somente depende da profundidade do fluido e independe da posição do líquido ao longo da horizontal, portanto é esperado que a pressão seja a mesma para cada altura independente do recipiente que está contido o fluido.
			\end{corollary}

\begin{frame}{O que é empuxo?}
	\begin{columns}
		\begin{column}{0.65\textwidth}
			\begin{corollary}
				\begin{itemize}
					\item A somatória de todas as forças que o fluido atua nas paredes de um objeto imerso em um fluido é igual a força resultante que atua para cima no intuito de subir o objeto;
					\item Se a força resultante $\vec{E}$ for de mesma intensidade da força peso $\vec{P}$ do volume do fluido deslocado, essa força é chamada de empuxo;
					\item Se o empuxo for maior que a força peso o objeto flutua, e se for menor o objeto afunda.
				\end{itemize}
			\end{corollary}
		\end{column}

\begin{frame}{Relação entre a densidade do fluido, do objeto e o princípio de Arquimedes}
	\begin{columns}[t]
		\begin{column}{0.55\textwidth}
		Pela definição de empuxo E podemos dizer que
		\begin{align*}
			E=m_{fluido}g,
		\end{align*}
		mas pela definição de densidade temos $m_{fluido}=\rho_{fluido}V$, portanto
		\begin{align*}
			E = \rho_{fluido}Vg
		\end{align*}
		O peso P do objeto mergulhado no fluido é dado por $P=m_{obj}g$, portanto se o empuxo for igual ao peso do objeto temos
		\end{column}
		\begin{column}{0.45\textwidth}
		\begin{align*}
			m_{obj}g & = \rho_{fluido}Vg,\\
			\rho_{obj}\bcancel{Vg} & = \rho_{fluido}\bcancel{Vg}.
		\end{align*}
			\begin{corollary}
				\begin{itemize}
					\small 
					\item Se $\rho_{fluido}<\rho_{obj}$, o corpo afundará;
					\item Se $\rho_{fluido}=\rho_{obj}$, o corpo ficará em equilíbrio;
					\item Se $\rho_{fluido}>\rho_{obj}$, o corpo irá flutuar na superfície;
				\end{itemize}
			\end{corollary}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Transformar um número em notação científica}
	\begin{corollary}
		\begin{enumerate}[leftmargin=1.75cm]
			\item[Passo 1:] Escrever o número incluindo a vírgula.
			\item[Passo 2:] Andar com a vírgula até que somente reste um número diferente de zero no lado esquerdo.
			\item[Passo 3:] Colocar no expoente da potência de 10 o número de casas decimais que tivemos que "andar" com a vírgula. Se ao andar com a vírgula o valor do número diminuiu, o expoente ficará positivo, se aumentou o expoente ficará negativo.
		\end{enumerate}
	\end{corollary}

\begin{frame}{Postulados da teoria da relatividade especial}
	\begin{columns}
		\begin{column}{0.7\textwidth}
			\begin{itemize}
				\item \alert{As leis físicas são as mesmas para quaisquer observadores em movimento uniforme:}Todo movimento é relativo, ou seja, não existe na natureza algum referencial privilegiado (como o Éter).

\begin{frame}{Cinemática relativística}
	Einstein percebeu que a única maneira de dois observadores, com um movimento relativo entre si, conseguirem medir o mesmo valor para a velocidade da luz, seria se um deles achasse que a régua ou o relógio do outro não estava coincidindo com o seu.
	\begin{itemize}
		\item O observador parado irá ver o comprimento $L$ do objeto em movimento contraído por um valor $L=\frac{L}{\gamma}'$, onde $L'$ é o seu comprimento próprio.
		\item O objeto em movimento irá ver o tempo do observador andar mais rápido que o seu, segundo a relação $\Delta t=\gamma \Delta t'$.
	\end{itemize}
	\begin{block}{Fator de Lorentz}
		\begin{equation*}
			\gamma = \frac{1}{\sqrt{1-\frac{v^2}{c^2}}},\quad\gamma\geq 1.
		\end{equation*}
	\end{block}
\end{frame}

\begin{frame}{Dinâmica relativística}
	\begin{multicols}{2}
		Na mecânica newtoniana, o momento de um objeto é definido por $\vec{p}=m\vec{v}$, onde $\vec{v}$ é a velocidade e m a massa. Porém, cálculos da velocidade relativa mostra que para a conservação do momento seja satisfeita na relatividade restrita, a massa deve mudar com a velocidade do objeto,
		\begin{equation*}
			\boxed{m(v)=\gamma(v) m_0,}
		\end{equation*}
		onde $m_0$ representa a massa do objeto no seu estado de repouso. Portanto, a expressão que define o momento do objeto também muda, na forma
		\begin{equation*}
			\boxed{\vec{p}=\gamma(v) m_0\vec{v}.}
		\end{equation*}
		Na relatividade, a expressão da energia também depende do fator de Lorentz, na forma
		\begin{equation*}
			\boxed{E=\gamma(v)mc^2.}
		\end{equation*}
	\end{multicols}
\end{frame}

\begin{frame}{Equação geral da relatividade}
	\begin{multicols}{2}
		Considere o momento de um objeto dado pela equação $p=\frac{m_0v}{\left(1-\frac{v^2}{c^2}\right)}$. Elevando ao quadrado temos
		\begin{align*}
			p^2 & = \left(\frac{1}{\sqrt{1-\frac{v^2}{c^2}}}\right)^2m_0^2v^2,\\
			p^2 & = \frac{c^2}{\left(c^2-v^2\right)}m^2_0v^2.
		\end{align*}
		Somando e subtraindo o lado direito por $\frac{m^2_0c^2}{c^2-v^2}$ nos fornece
		\begin{align*}
			p^2 & = c^2\left(\frac{m^2_0v^2}{c^2-v^2}\alert{+\frac{m^2_0c^2}{c^2-v^2}-\frac{m^2_0c^2}{c^2-v^2}}\right),\\
			p^2 & = c^2\left(-m^2_0+\gamma^2 m^2_0\right).
		\end{align*}
		Mas $\gamma^2 m^2_0=\frac{E^2}{c^4}$, portanto
		\begin{equation*}
			\boxed{p^2 = -m^2_0c^2+\frac{E^2}{c^2}.}
		\end{equation*}
	\end{multicols}
\end{frame}

\begin{frame}{Equivalência entre massa e energia}
	\begin{columns}
		\begin{column}{0.5\textwidth}
		Considere a relação $p^2=-m^2_0c^2+\frac{E^2}{c^2}$, se o objeto está em repouso, ou seja v=0, temos p=0, portanto
		\begin{align*}
			\frac{E^2}{c^2} & = m^2_0c^2,\\
			E & = \pm m_0c^2.
		\end{align*}
		\end{column}
		\begin{column}{0.5\textwidth}
		A solução $E=-m_0c^2$ revela que toda matéria de massa $m_0$ possui a sua versão antimatéria, cuja massa tem a mesma quantidade $m_0$.
	\begin{block}{Energia de repouso}
		\begin{equation*}
			E = m_0c^2.
		\end{equation*}
	\end{block}
		\end{column}
	\end{columns}
		\begin{corollary}
			A combinação de matéria e antimatéria acarretaria na transformação de toda massa $2m_0$ numa quantidade de energia equivalente a $E=2m_0c^2$.
		\end{corollary}
\end{frame}

\begin{frame}{Transformar um número em notação científica}
	\begin{corollary}
		\begin{enumerate}[leftmargin=1.75cm]
			\item[Passo 1:] Escrever o número incluindo a vírgula.
			\item[Passo 2:] Andar com a vírgula até que reste somente um número diferente de zero no lado esquerdo.
			\item[Passo 3:] Colocar no expoente da potência de 10 o número de casas decimais que tivemos que "andar" com a vírgula. Se ao andar com a vírgula o valor do número diminuiu, o expoente ficará positivo, se aumentou o expoente ficará negativo.
		\end{enumerate}
	\end{corollary}
	\begin{exampleblock}{Exemplo}
		\begin{equation*}
			6\; 590\; 000\; 000\; 000\; 000,0 = \SI{6,59e15}{}		
		\end{equation*}
	\end{exampleblock}
\end{frame}

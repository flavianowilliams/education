\begin{frame}{Resultado experimental}

	\begin{columns}
		\begin{column}{0.6\textwidth}

				\item[\ding{51}] O sentido da corrente depende se aproximamos ou afastamos o íman da espira e também do sentido das linhas de campo magnético do íman que atravessa a espira.
			\end{enumerate}
		\end{column}

		\begin{figure}

			\caption*{\footnotesize Amperímetro revelando a existência de corrente na espira quando o íman está em movimento em relação a espira.}
		\end{figure}
	\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Definição de fluxo magnético}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			Considere uma pequena área $dA$, o fluxo das linhas de campo magnético que atravessam essa área é dado por
			\begin{equation*}
				d\Phi_B = B\cos\theta dA.
			\end{equation*}
			Somando todas as contribuições do fluxo $d\Phi$ de cada parte $dA$ que pertence a área $A$ inserida no interior de uma espira temos o fluxo magnético total $\Phi$ de $\vec{B}$ que atravessa a espira,
			\begin{equation*}
				\boxed{\Phi_B = \int\vec{B}\cdot\widehat{n} dA.}
			\end{equation*}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
%				\input{tikz/fluxo-magnetico.pgf}
				\centering
				\begin{tikzpicture}[scale=0.75, transform shape, x={(-10:1cm)},y={(90:1cm)},z={(210:1cm)}]

\begin{frame}{Lei de Faraday}
	\small
	\begin{multicols}{2}
		Baseado na observação experimental, podemos concluir que se houver uma variação do fluxo (\alert{que pode ser definido por uma derivada temporal}) do campo magnético que atravessa uma espira fechada, essa variação irá produzir uma f.e.m. induzida $\varepsilon$ ao longo da espira, ou seja,
		\begin{equation*}
			\varepsilon = \frac{d\Phi}{dt},
		\end{equation*}
		No caso de uma bobina contendo N espiras, podemos somar a variação do fluxo em cada espira, o que resulta na expressão
		\begin{equation*}
			\boxed{\varepsilon = N\frac{d\Phi}{dt}.}
		\end{equation*}
		No SI a unidade de medida de fluxo magnético é Tesla-metro quadrado ou Weber (Wb).
	\end{multicols}
	\begin{corollary}
		O módulo da força eletromotriz induzida em uma espira condutora é igual à taxa de variação temporal do fluxo magnético que atravessa a espira.
	\end{corollary}
\end{frame}

\begin{frame}{Sentido da corrente induzida em relação a variação do fluxo}
\begin{columns}
	\begin{column}{0.7\textwidth}
		No caso do íman, aproximando a parte norte na espira, se a corrente tiver sentido oposto ao previsto, pela lei de Ampère criaria um campo magnético induzido no mesmo sentido do campo do íman. Isso atrairia o íman para dentro da espira aumentando a sua energia cinética, e ao mesmo tempo produziria calor por efeito Joule, o que violaria a \alert{lei da conservação da energia}.
	\end{column}
	\begin{column}{0.3\textwidth}
		\begin{figure}
			\centering
			\includegraphics[scale=0.14]{figuras/lei-lenz.jpg}
			\caption*{\footnotesize Resumo da lei de Lenz.}
		\end{figure}
	\end{column}
\end{columns}
	\begin{corollary}
		A conservação da energia exige que a energia cinética dissipada pela corrente por efeito Joule se obtenha a custa de uma redução da energia cinética do íman.
	\end{corollary}
\end{frame}

\begin{frame}{Lei de Lenz e o significado físico do sentido da corrente}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			\begin{block}{Lei de Lenz}
				A corrente induzida em um circuito aparece sempre com um sentido tal que o campo magnético que ela cria tende a contrariar a variação do fluxo magnético que a originou.
			\end{block}
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{figure}
				\centering
				\begin{subfigure}[b]{0.45\textwidth}
					\centering
					\includegraphics[scale=0.2]{figuras/figura-2a}
					\caption*{\footnotesize $\frac{d\Phi}{dt}>0$ no tempo.}
				\end{subfigure}
				\begin{subfigure}[b]{0.45\textwidth}
					\centering
					\includegraphics[scale=0.2]{figuras/figura-2b}
					\caption*{\footnotesize $\frac{d\Phi}{dt}<0$ no tempo.}
				\end{subfigure}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{block}{Lei de Faraday-Lenz}
		\begin{equation*}
			\frac{d\Phi}{dt} = \alert{-}\varepsilon
		\end{equation*}
	\end{block}
\end{frame}

\begin{frame}{Campo elétrico induzido}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Pela definição de força eletromotriz sabemos que $\varepsilon=\oint \vec{E}\cdot d\vec{s}$. Substituindo na definição da Lei de Faraday-Lenz, $\varepsilon = -\frac{d\Phi}{dt}$, temos
			\begin{align*}
				\oint \vec{E}\cdot d\vec{s} = - \frac{d\Phi}{dt}.
			\end{align*}
			\alert{A equação acima nos diz que um campo magnético variável induz um campo elétrico circular}.
		\end{column}

\section{Indutância}

\subsection{}

\begin{frame}{Fluxo magnético no interior de um solenóide}
	\begin{columns}
		\begin{column}[c]{0.7\textwidth}
			Considere um solenóide infinito de área $A$, o fluxo magnético que atravessa cada espira do solenóide é dado por
			\begin{align*}
				\Phi_B = BA.
			\end{align*}
			Pela lei de Ampere, determinamos o campo magnético $\vec{B}$ induzido no interior de um solenóide de densidade n de espiras, dado por	$B = \mu_0 i n$. Substituindo temos
			\begin{align*}
				\Aboxed{\Phi_B = \mu_0 i nA.}
			\end{align*}
		\end{column}
		\begin{column}[c]{0.3\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.3]{figuras/indutancia.png}
				\caption*{\footnotesize Fluxo do campo magnético $\Phi_B$ no interior de um solenóide.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Indutância}
%	\begin{multicols}{2}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			Considere agora um segmento do solenóide que possui um comprimento $l$. O número de espiras desse segmento vale $N=nl$. O fluxo total do campo magnético no seu interior é a soma do fluxo que atravessa cada espira, ou seja,
			\begin{align*}
				N\Phi_B & = (nl)\Phi_B,\\
				N\Phi_B & = (nl)(\mu_0 inA),
			\end{align*}
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{align*}
				N\Phi_B & = (\mu_0 n^2 lA)i.
			\end{align*}
			O termo entre parênteses depende somente da geometria do solenóide, recebendo o nome de indutância L, onde a indutância por comprimento vale
			\begin{align*}
				\Aboxed{\frac{L}{l} = \mu_0n^2A.}
			\end{align*}
		\end{column}		
%	\end{multicols}
	\end{columns}
	\begin{corollary}
		A permeabilidade magnética pode ser representada em unidades de H/m.
	\end{corollary}
\end{frame}

\begin{frame}{Autoindução}
	\begin{columns}
		\begin{column}[c]{0.55\textwidth}
			Como foi mostrado, o fluxo que atravessa um segmento de um solenóide vale $N\Phi_B = Li$. Pela lei de Faraday-Lenz temos que a f.e.m. induzida ao longo do solenóide é dado por
			\begin{align*}
				\varepsilon_L & =-\frac{d(N\Phi_B)}{dt}=-\frac{d(Li)}{dt},\\
				\Aboxed{\varepsilon_L & =-L\frac{di}{dt}.}
			\end{align*}
		\end{column}

\begin{frame}{Conservação da energia em um circuito contendo um indutor}
	\begin{columns}
		\begin{column}[c]{0.7\textwidth}
			Considere um circuito contendo um gerador de corrente alternada e um indutor com indutância $L$. Sabemos que o gerador terá uma potência elétrica dado por $\frac{dW}{dt}=\varepsilon i$, onde $\varepsilon$ é a f.e.m. associada ao gerador. Por conservação de energia, poderemos dizer que neste caso a energia fornecida a cada intervalo de tempo será convertida em energia magnética no indutor, onde pela lei de Faraday-Lenz teremos
			\begin{align*}
				\varepsilon i = -\varepsilon_Li = i\frac{d\Phi_B}{dt},
			\end{align*}
			onde $\varepsilon_L$ é a f.e.m. induzida no solenóide.
		\end{column}

\begin{frame}{Energia armazenada em um campo magnético}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Integrando $\frac{dW}{dt}=\varepsilon i$ teremos
			\begin{align*}
			\Delta U & = \int\limits_{0}^{t} i\frac{d\Phi_B}{dt}dt,\\
				\Delta U & = \int\limits_{0}^{t} Li\frac{di}{dt}dt.
			\end{align*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			Sabendo que $di=\frac{di}{dt}dt$ e considerando que a energia magnética no indutor em t=0 é zero, teremos
			\begin{align*}
				U(I) & = L\int\limits_{0}^{I}idi,\\
				\Aboxed{U(I) & = \frac{1}{2}LI^2.}
			\end{align*}
		\end{column}
	\end{columns}
	\begin{corollary}
		A energia armazenada em um indutor dependerá da corrente I e da sua indutância.
	\end{corollary}
\end{frame}

\begin{frame}{Densidade de energia de um campo magnético}
%	\begin{multicols*}{2}
\begin{columns}
	\begin{column}[t]{0.5\textwidth}
		No solenóide temos $L=\mu_0ln^2A$. Substituindo na expressão anterior teremos
		\begin{align*}
			U & = \frac{1}{2}(\mu_0n^2li^2A),\\
			U & = \frac{1}{2\mu_0}(\mu_0ni)^2Al
		\end{align*}
		O termo entre parêntesis corresponde ao campo $\vec{B}$ no interior do solenóide e
	\end{column}
	\begin{column}[t]{0.5\textwidth}
		 $V=Al$ o seu volume. Portanto
		\begin{align*}
		U = \frac{1}{2\mu_0}B^2V
		\end{align*}
		Dividindo pelo volume temos a densidade de energia magnética no indutor,
		\begin{align*}
			\Aboxed{\frac{U}{V} = u=\frac{1}{2\mu_0}B^2.}
		\end{align*}
	\end{column}
\end{columns}
%	\end{multicols*}
	\begin{corollary}
		A densidade de energia magnética aumenta com o quadrado do campo magnético.
	\end{corollary}
\end{frame}

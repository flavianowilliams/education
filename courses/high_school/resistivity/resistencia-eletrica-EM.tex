\section{Resistência elétrica}

\subsection{}

\begin{frame}{O que é resitência elétrica?}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{corollary}
				Quando uma diferença de potencial V é aplicada nas extremidades de um condutor, estabelecendo nele uma corrente elétrica i, a resistência desse condutor em \textcolor{red}{ohms} ($\Omega$) é dada pela relação
				\begin{align*}
					R=\frac{V}{i}				
				\end{align*}
			\end{corollary}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.33]{figuras/figura-1.png}
				\caption*{\footnotesize Esquema de resistividade em um fio.}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		Quanto maior a temperatura, maior será a oposição que o condutor oferecerá a passagem da corrente e maior será o valor de R.
	\end{corollary}
\end{frame}

\begin{frame}{Resistividade de um material}
	\begin{multicols}{2}
		Foi verificado experimentalmente que a resistência R \textcolor{red}{aumenta proporcionalmente} com o comprimento L do fio condutor, ou seja,
		\begin{align*}
			R\;\alpha\;L.
		\end{align*}
		Verificou-se também que a resistência é \textcolor{red}{inversamente proporcional} a área da seção reta,
		\begin{align*}
			R\;\alpha\;\frac{1}{A},
		\end{align*}
		Reunindo ambas as informações podemos definir a resistência de um fio condutor em função da área e comprimento na forma
		\begin{align*}
			\Aboxed{R=\rho \frac{L}{A},}
		\end{align*}
		onde $\alert{\rho}$ é denominada resistividade do material, que depende das caraterísticas atômicas de cada material.
	\end{multicols}
\end{frame}

\begin{frame}{Fatores que influenciam na resistência}
	\begin{itemize}
		\item A resistência de um condutor é tanto maior quanto maior for seu comprimento;
		\item A resistência de um condutor é tanto maior quanto menor for a área de sua seção reta (quanto mais fino for o condutor);
		\item A resistência de um condutor depende do material de que ele é feito.
	\end{itemize}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{block}{Condutividade elétrica}
				A substância será tanto melhor condutora elétrica quanto menor for a sua resistividade.
			\end{block}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.3]{figuras/figura-2.png}
				\caption*{\footnotesize Resistividade de alguns materiais.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Reostato}
	\begin{corollary}
		Variando o comprimento ou a área de um resistor, podemos variar a sua resistência, e consequentemente a corrente elétrica no circuito. Esse é o princípio de funcionamento do reostato.
	\end{corollary}

\section{A Lei de Ohm}

\subsection{}

\begin{frame}{Resistor ôhmico}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{block}{A Lei de Ohm}
				Para um grande número de condutores (como os metais), o valor da resistência permanece constante, não dependendo da diferença de potencial aplicada ao condutor.
			\end{block}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{align*}
				\frac{(V_{AB})_1}{i_1} & =\frac{(V_{AB})_2}{i_2}=\cdots,\\
				\Aboxed{\frac{V_{AB}}{i} & =R=\text{constante}.}
			\end{align*}
		\end{column}
	\end{columns}
	\begin{figure}
		\centering
		\includegraphics[scale=0.39]{figuras/figura-4a.png}
		\caption*{\footnotesize Aumentanto a tensão a corrente aumenta proporcionalmente}
	\end{figure}
\end{frame}

\begin{frame}{Resistores ligados em paralelo}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			A corrente i que atravessa o fio condutor é dividida nos terminais A e B, o que resulta
			\begin{align*}
				i = i_1+i_2+i_3.
			\end{align*}
			Sabendo que $i=\frac{V}{R}$ temos
			\begin{align*}
				\frac{V_{AB}}{R_{eq}}=\frac{V_{R_1}}{R_1}+\frac{V_{R_2}}{R_2}+\frac{V_{R_3}}{R_3}.
			\end{align*}
		\end{column}
		\begin{column}{0.4\textwidth}

\begin{frame}{Resistores ligados em paralelo}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			mas a tensão entre os terminais A e B da bateria é a mesma nos terminais dos resistores $R_1$, $R_2$ e $R_3$, ou seja, 					\begin{align*}
				V_{AB}=V_{R_1}=V_{R_2}=V_{R_3}=V,
			\end{align*}
			portanto
			\begin{align*}
				\frac{V_{AB}}{R_{eq}} & =\frac{V}{R_{eq}}=\frac{V}{R_1}+\frac{V}{R_2}+\frac{V}{R_3},\\
				\frac{\bcancel{V}}{R_{eq}} & =\bcancel{V}\left(\frac{1}{R_1}+\frac{1}{R_2}+\frac{1}{R_3}\right),
			\end{align*}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{align*}
			\frac{1}{R_{eq}} & =\frac{1}{R_1}+\frac{1}{R_2}+\frac{1}{R_3},\\
				\Aboxed{R_{eq} & =\frac{R_2R_2R_3}{R_1R_3+R_2R_3+R_1R_2}.}
			\end{align*}
			\begin{block}{Resistência equivalente de ligação em paralelo}
				\begin{align*}
					\frac{1}{R_{eq}} =\frac{1}{R_1}+\frac{1}{R_2}+\cdots +\frac{1}{R_N}.
				\end{align*}
			\end{block}
		\end{column}
	\end{columns}	
\end{frame}

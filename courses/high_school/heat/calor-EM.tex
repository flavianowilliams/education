\section{Calor}

\subsection{}

\begin{frame}{O que é calor?}
	\begin{block}{Calor}
		\alert{Energia transferida} de um objeto para outro em virtude, unicamente, de uma diferença de temperatura entre eles (essa energia é chamada de energia interna ou térmica).
	\end{block}
	\begin{figure}[c]
		\includegraphics[scale=0.35]{figuras/figura-1.png}
	\end{figure}
\end{frame}

\section{A teoria do calórico}

\begin{frame}{Caloria como unidade de medida de energia interna}
	\begin{columns}[t]
		\begin{column}{0.6\textwidth}
			\begin{block}{Equivalência entre caloria e Joules}
				Calor é energia em trânsito, portanto possui unidade de energia. A unidade mais utilizada em nutrição é a \alert{caloria}, e seu valor surge da experiência do equivalente mecânico proposto por Thompson em
		\begin{align*}
			\SI{1}{cal} = \SI{4,2}{J}.

	\end{block}

\section{Transferência de Calor}

\subsection{}

\begin{frame}
	A transferência de calor entre dois ou mais objetos pode ocorrer de três maneiras distintas:
	\begin{itemize}
		\item Condução: Ocorre praticamente em sólidos.
		\item Convecção: Ocorre em líquidos e gases.
		\item Radiação: Ocorre em qualquer material, inclusive no vácuo.
	\end{itemize}
	\begin{corollary}
		Radiação é o único processo de transferência de calor que ocorre no vácuo.
	\end{corollary}	
\end{frame}

\begin{frame}{Condução}
	\begin{block}{Processo de transferência por condução}
		Os átomos de uma extremidade da barra após aquecidos vibram com maior intensidade. Essa agitação é transmitida para os átomos que estão próximos e assim sucessivamente até o final da barra. Isso faz com que o calor seja transmitido ao longo da barra através da agitação dos átomos da rede cristalina.
	\end{block}
	\begin{figure}[c]
		\includegraphics[scale=0.35]{figuras/figura-2.png}
	\end{figure}
\end{frame}

\begin{frame}{Convecção}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{block}{Processo de transferência por convecção}
				Quando a parte de baixo é aquecida, as moléculas se agitam mais o que faz a densidade do líquido diminuir. Como a densidade diminui, o líquido fica mais leve e sobe para o ocupar o espaço preenchido pela parte de cima. Enquanto isso, a parte de cima que está mais pesada desce ocupando o espaço vazio abaixo. Esse processo continua, surgindo o que chamamos de \alert{correntes de convecção}.
			\end{block}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}[c]
				\includegraphics[scale=0.35]{figuras/figura-3.png}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Radiação}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{block}{Processo de transferência por radiação}
				Todos os corpos quando aquecidos emitem radiações térmicas que, ao serem absorvidas por outro corpo, podem provocar nele uma elevação da temperatura.
			\end{block}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}[c]
				\includegraphics[scale=0.25]{figuras/figura-4.png}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		O calor que recebemos do sol chega até nós por radiação, uma vez que entre o Sol e a Terra existe vácuo.
	\end{corollary}
\end{frame}

\section{Quantidade de calor}

\subsection{}

\begin{frame}{Capacidade térmica}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{block}{Capacidade térmica}
				Se um objeto recebe uma quantidade de calor $\Delta Q$ e sua temperatura varia de $\Delta T$, a capacidade térmica desse objeto é dada por
				\begin{equation*}
					C=\frac{\Delta Q}{\Delta T}.
				\end{equation*}
			\end{block}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}[c]
				\includegraphics[scale=0.25]{figuras/figura-7.png}
			\end{figure}
		\end{column}
	\end{columns}	
	\vspace*{0.5cm}
	\begin{columns}
		\begin{column}{0.5\textwidth}
		\begin{equation*}
			C_A=\frac{100\;\text{cal}}{20\;^oC}\;\therefore C_A=5,0\;cal/^oC
		\end{equation*}			
		\end{column}
		\begin{column}{0.5\textwidth}
		\begin{equation*}
			C_B=\frac{100\;\text{cal}}{10\;^oC}\;\therefore C_B=10,0\;cal/^oC
		\end{equation*}			
		\end{column}
	\end{columns}	
\end{frame}

\begin{frame}{Calor específico}
	\begin{corollary}
		A razão entre a capacidade térmica e a massa não varia para o mesmo material,
		\begin{equation*}
			\frac{C_1}{m_1}=\frac{C_2}{m_2}=\frac{C_3}{m_3}=\dots=c\;\text{(mesmo material)}.
		\end{equation*}
	\end{corollary}
	\begin{block}{Calor específico}
		Se um objeto de massa m tem uma capacidade térmica C, o calor específico, c, do material que o constitui é dado por
		\begin{equation*}
			c=\frac{C}{m}.
		\end{equation*}
	\end{block}
\end{frame}

\begin{frame}{Cálculo da quantidade de calor cedido e absorvido por um objeto}
	\begin{columns}
		\begin{column}{0.75\textwidth}
			\begin{block}{Quantidade de calor}
				A quantidade de calor, Q ou $\Delta Q$, absorvida ou liberada por um objeto de massa m e calor específico c, quando sua temperatura varia de $\Delta T$, é dada por
				\begin{equation*}
					Q=mc\Delta T.
				\end{equation*}
			\end{block}
		\end{column}
		\begin{column}{0.25\textwidth}
			\begin{figure}[c]
				\begin{footnotesize}
					\begin{table}
						\centering
						\caption*{\footnotesize Calores específicos}
						\vspace*{-0.5cm}
						\begin{tabular}[c]{cc}
							\hline
							Substância & $cal/g\cdot ^{\circ}C$\\
							\hline
							Água & 1,0\\
							Gelo & 0,55\\
							Vapor & 0,5\\
							Chumbo & 0,031\\
							\hline					
						\end{tabular}
					\end{table}				
				\end{footnotesize}
				%\includegraphics[scale=0.19]{figuras/figura-8.png}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}	
		Por conservação de energia, o calor total liberado pelos objetos que se esfriaram é igual ao calor total absorvido pelos objetos que se aqueceram,
		\begin{equation*}
			Q_1+Q_2+Q_3+\dots+Q_n=0.
		\end{equation*}
	\end{corollary}
\end{frame}

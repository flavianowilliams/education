\begin{frame}{A física e os vetores}
	A física inteira é representada por grandezas escalares e vetoriais. Do ponto de vista matemático, as grandezas escalares são representadas por escalares enquanto que as grandezas vetoriais por vetores.
	\begin{exampleblock}{Exemplos de grandezas escalares e vetoriais}
		\begin{itemize}
			\item [\ding{51}] Grandezas escalares: Massa, pressão, energia, trabalho,...
			\item[\ding{51}] Grandezas vetoriais: Força, deslocamento, velocidade, aceleração, torque,...
		\end{itemize}
	\end{exampleblock}
	\begin{corollary}
		Independente se a massa do bloco permanece a mesma, podemos observar que o fenômeno físico observado (seu deslocamento) muda se o empurrarmos da esquerda para a direita ou da direita para a esquerda.
	\end{corollary}
\end{frame}

\begin{frame}{Representação de um vetor numa base ortonormal}
\begin{columns}
	\begin{column}[c]{0.5\textwidth}
		Considere o vetor $\vec{r}$. Desde que a base representada pelo sistema de coordenadas cartesiana seja ortogonal, podemos utilizar relações trigonométricas, e dizer que
		\begin{align*}
		r_x & = rcos\theta,\\
		r_y & = rsen\theta,\\
		  tan\theta & = \frac{r_y}{r_x},\\
		  r & = \sqrt{r^2_x+r^2_y}.
		\end{align*}
	\end{column}

\begin{frame}{Soma de vetores e vetor resultante}
	\begin{columns}
		\begin{column}[c]{0.65\textwidth}
			Considere a equação abaixo
			\begin{equation*}
				\vec{r} = \vec{a} + \vec{b},
			\end{equation*}
			onde $\vec{r}$ é o vetor resultante. Representando $\vec{a}$ e $\vec{b}$ em suas componentes temos
			\begin{align*}
				\vec{r} & = (a_x\hat{i}+a_y\hat{j})+(b_x\hat{i}+b_y\hat{j}),\\
				\vec{r} & = \underbracket{\left(a_x+b_x\right)}_{\tiny\alert{r_x}}\hat{i}+\underbracket{\left(a_y+b_y\right)}_{\tiny\alert{r_y}}\hat{j},\\
				\Aboxed{\vec{r} & = r_x\hat{i}+r_y\hat{j}.}
			\end{align*}

\begin{frame}{Vetor resultante a partir da regra do paralelogramo}
	\begin{columns}
		\begin{column}[c]{0.65\textwidth}
			Ordenando os vetores de modo que o início de um coincida com o fina do outro teremos que
			\begin{align*}
				\vec{a}+\vec{b}+\vec{c}+\vec{r}+\cdots = \vec{0}.
			\end{align*}
			Sabendo que -$\vec{r}$ representa o vetor $\vec{r}$ no sentido oposto teremos
			\begin{align*}
				\boxed{\vec{r} = \vec{a}+\vec{b}+\vec{c}+\cdots.}
			\end{align*}			
			\textit{\small \href{https://phet.colorado.edu/pt_BR/simulations/vector-addition}{Utilize a animação para encontrar o vetor resultante a partir da formação de um polígono}.}

\begin{frame}{Propriedades de adição vetorial}
	Uma grandeza vetorial deve obedecer as seguintes propriedades de adição:
	\begin{align*}
		\vec{r} & = \vec{a}+\vec{b} &\text{(equação vetorial)},\\
		\vec{a}+\vec{b} & = \vec{b}+\vec{a}&\text{(lei comutativa)},\\
		(\vec{a}+\vec{b})+\vec{c} & = \vec{a}+(\vec{b}+\vec{c}) &\text{(lei associativa)},\\
		\vec{r} & = \vec{a}-\vec{b} = \vec{a}+(-\vec{b}) & \text{(subtração de vetores)}.
	\end{align*}
	\begin{corollary}
		A ordem dos vetores na soma não afeta o resultado.
	\end{corollary}
\end{frame}

\begin{frame}{Tipos de multiplicação em cálculo vetorial}
	\begin{itemize}
		\item [\ding{51}] Multiplicação entre um vetor e um escalar $\Rightarrow$ O resultado será outro vetor:
		\begin{align*}
			\vec{F} & = m\vec{a} & \text{(Segunda Lei de Newton)},
		\end{align*}
		\item[\ding{51}] Produto escalar $\Rightarrow$ O resultado será um escalar:
		\begin{align*}
			W & = \vec{F}\cdot d\vec{r} & \text{(Trabalho)},
		\end{align*}
		\item[\ding{51}] Produto vetorial $\Rightarrow$ O resultado será outro vetor diferente dos anteriores:
		\begin{align*}
			\vec{\tau} & = \vec{r}\times\vec{F} & \text{(Torque)}.
		\end{align*}
	\end{itemize}
	\begin{corollary}
		Todas as três maneiras são diferentes entre si e nenhuma é igual à multiplicação algébrica.
	\end{corollary}
\end{frame}

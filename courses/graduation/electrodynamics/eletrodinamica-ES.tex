\section{Corrente elétrica}

\subsection{}

\begin{frame}{Movimento de cargas em um condutor}
	\begin{columns}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.3]{figuras/figura-1.png}
			\end{figure}
		\end{column}
		\begin{column}{0.6\textwidth}
			\begin{itemize}
				\item Em um condutor existe grande número de elétrons que estão fracamente ligados ao núcleo de cada átomo;
				\item Na presença de um campo elétrico, os elétrons livres, sob a ação da força elétrica, entram em movimento ordenado, formando a corrente elétrica;
				\item A corrente de elétrons sempre flui da região de menor potencial para a região de maior potencial elétrico.
			\end{itemize}
		\end{column}
	\end{columns}	
\end{frame}

\begin{frame}{Definição de corrente elétrica}
	\begin{columns}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.45]{figuras/figura-4.png}
			\end{figure}
		\end{column}
		\begin{column}{0.6\textwidth}
			\begin{block}{Intensidade da corrente}
				Quando uma quantidade de carga infinitesimal $dq$ atravessa a seção de um condutor, durante um intervalo de tempo $dt$, a intensidade da corrente i nessa seção é a taxa de variação da carga $q$ no instante $t$,
				\begin{align*}
					i = \lim_{\Delta t\rightarrow\infty}\frac{\Delta q}{\Delta t} = \frac{dq}{dt}.
				\end{align*}
			\end{block}
		\end{column}
	\end{columns}
	\begin{corollary}
		\justifying No SI a unidade de medida da corrente é C/s ou Ampère (A).
	\end{corollary}
\end{frame}

\begin{frame}{Conservação da carga elétrica}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			No regime estacionário, a quantidade de carga que atravessa a seção 0 é a mesma que atravessa as seções i e 2 juntas. Pela conservação da carga elétrica, a soma das correntes nos dois ramos deve ser igual a corrente inicial, ou seja,
			\begin{equation*}
				\boxed{i_0 = i_1+i_2.}
			\end{equation*}
			\begin{corollary}
				As setas são usadas para indicar o sentido de movimento das cargas, já que i é um escalar.
			\end{corollary}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.4]{figuras/corrente-divisao.png}
				\caption*{\footnotesize Corrente inicial $i_0$ separada pelas correntes $i_1$ e $i_2$ após atravessar a ramificação no ponto a.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Densidade de corrente}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			Para descrever o fluxo de cargas usamos a densidade de corrente J, que tem a mesma direção e sentido da velocidade das cargas positivas. Portanto, a corrente que atravessa perpendicularmente o elemento de área dA é dado por
			\begin{equation*}
				\boxed{i = \int \vec{J}\cdot d\vec{A}.}
			\end{equation*}
			Se a corrente é uniforme e paralela a $\widehat{n}$ teremos
			\begin{equation*}
				i = \int JdA = J\int dA = JA.
			\end{equation*}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{equation*}
				\boxed{J = \frac{i}{A}.}
			\end{equation*}

\section{Resistência elétrica}

\subsection{}

\begin{frame}{Visão microscópica da resistividade de um material}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			Os elétrons em um condutor se comportam como as moléculas de um gás livre (gás de elétrons!), se movendo aleatoriamente em todas as direções e colidindo com os átomos da rede cristalina, mudando o movimento em cada colisão. Quando submetido a um campo elétrico $\vec{E}$, os elétrons se movem ligeiramente para o lado, com a velocidade de deriva muito menor que a velocidade efetiva dos elétrons (semelhante a um enxame de abelhas se deslocando de um lado a outro).

\begin{frame}{Visão microscópica da resistividade de um material (continuação)}
	\begin{multicols}{2}
		Se um elétron de massa m é submetido a um campo elétrico $\vec{E}$ ele sofre uma aceleração a dado por
		\begin{equation*}
			a = \frac{F}{m} = \frac{eE}{m}.
		\end{equation*}
		Supondo que o intervalo de tempo médio $\tau$ entre colisões o elétron adquire uma velocidade de deriva dado por $v_d = a\tau$. Portanto
		\begin{align*}
			v_d & = a\tau,\\
			v_d & = \frac{eE\tau}{m}.
		\end{align*}
		Sabendo que $J=nev_d$ temos $\frac{J}{ne}=\frac{eE\tau}{m}$, ou seja,
		\begin{equation*}
			\boxed{E = \rho J,}
		\end{equation*}
		onde $\rho=\frac{m}{e^2n\tau}$. $\rho$ é definido como resistividade do material.
	\end{multicols}
\end{frame}

\begin{frame}{Resistor ôhmico}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			Para um grande número de condutores (como os metais), o valor da resistência permanece constante, não dependendo da diferença de potencial aplicada ao condutor.
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{align*}
				\frac{(\Delta V)_1}{i_1} & =\frac{(\Delta V)_2}{i_2}=\cdots,\\
				\Aboxed{\frac{V}{i} & =R=\text{constante}.}
			\end{align*}
		\end{column}
	\end{columns}
	\begin{figure}
		\centering
		\includegraphics[scale=0.4]{figuras/lei-ohm.png}
		\caption*{\footnotesize Aumentanto a tensão a corrente aumenta proporcionalmente}
	\end{figure}
\end{frame}

\begin{frame}{Resistência a partir da resistividade}
	\begin{multicols}{2}
		Considere um pedaço L de um fio retilíneo que possui uma resistividade $\rho$ e seção reta A, Se o fio for ôhmico a resistência é dado por $R=V/i$. Sabemos que a relação entre o campo elétrico $\vec{E}$ e a ddp entre os terminais do fio é dado por $V=EL$. Sabemos também que a corrente estacionária no fio é dado por $i=JA$. Substituindo
		\begin{equation*}
			R = \frac{V}{i} = \frac{EL}{JA}.
		\end{equation*}
		Mas $E=\rho J$, portanto
		\begin{equation*}
			\boxed{R = \rho\frac{L}{A}.}
		\end{equation*}
	\end{multicols}
	\begin{corollary}
		A resistência é uma propriedade de um dispositivo e a resistividade de um material.
	\end{corollary}
\end{frame}

\begin{frame}{Lei de Ohm}
	\begin{columns}
		\begin{column}[c]{0.7\textwidth}
			\begin{enumerate}
				\item[\ding{51}] A corrente que atravessa um circuito é sempre diretamente proporcional  à diferença de potencial aplicada ao dispositivo;
				\item[\ding{51}] Um dispositivo obedece à lei de Ohm se a resistência não depende da diferença de potencial nem do campo elétrico aplicado;
				\item[\ding{51}] Na prática a resistividade do material aumenta linearmente com a temperatura segundo a relação
				\begin{equation*}
					\rho = \rho_0\alpha (T-T_0).
				\end{equation*}
			\end{enumerate}
		\end{column}
		\begin{column}[c]{0.3\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.4]{figuras/lei-ohm-grafico.png}
				\caption*{\footnotesize Gráfico tensão versus corrente de um resistor ôhmico.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Resistência equivalente em ligações em série}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			Substituindo $V_{AB}$, $V_{BC}$ e $V_{CD}$ temos
			\begin{align*}
				V_{AD} & = V_{AB}+V_{BC}+V_{CD},\\
				V_{AD} & = R_1i+R_2i+R_3i,\\

\begin{frame}{Potência em circuitos elétricos}
%	\begin{multicols}{2}
		Para deslocar uma quantidade de carga $dq$ de um terminal a outro de um circuito é necessário realizar um trabalho $dW$ dado por
		\begin{align*}
			dW = Vdq.
		\end{align*}
		Pela definição de corrente podemos dizer que $dq=idt$, portanto
		\begin{align*}
			dW = Vidt.
		\end{align*}
		Pela definição de potência P, sabemos que $dW=Pdt$, portanto podemos dizer que a taxa de transferência de energia necessária para deslocar uma quantidade de cada por segundo de um terminal a outro é dado por
		\begin{align*}
			\boxed{P = Vi.}
		\end{align*}
%	\end{multicols}
\end{frame}

\begin{frame}{Efeito Joule}
%	\begin{multicols}{2}
		No caso de um resistor ôhmico temos $V=Ri$, substituindo temos a taxa de energia elétrica dissipada a cada segundo em um resistor,
		\begin{align*}
			P & = (Ri)i = Ri^2,\\
			P & = V\left(\frac{V}{R}\right) = \frac{V^2}{R}.
		\end{align*}
		\begin{block}{Efeito Joule}
			Capacidade de um resistor em realizar trabalho convertendo energia elétrica em energia térmica.
		\end{block}
%	\end{multicols}
\end{frame}

\begin{frame}{Força eletromotriz}
	Para que ocorra uma corrente elétrica ao longo do circuito é necessário um dispositivo, como uma bateria ou gerador, que realize um trabalho $\tau$ afim de deslocar uma quantidade de carga $\Delta q$ de um ponto a outro desse circuito. Portanto, o trabalho $\alert{\varepsilon}$ necessário para deslocar cada elemento de carga do terminal positivo para o terminal negativo de uma bateria ou gerador é chamado força eletromotriz da bateria.
	\begin{block}{Força eletromotriz (f.e.m.)}
		\begin{equation*}
			\varepsilon = \lim_{\Delta q\rightarrow\infty}\frac{\Delta W}{\Delta q} = \frac{dW}{dq}.
		\end{equation*}
	\end{block}
	\begin{corollary}
		A unidade de medida de f.e.m. no SI é $\text{J/C}\equiv \text{Volts}$.
	\end{corollary}
\end{frame}

\begin{frame}{Potência fornecida por uma bateria}
	\begin{multicols}{2}
		Vimos que o trabalho realizado por uma bateria para deslocar uma quantidade de carga $dq$ é dado por $dW = \varepsilon dq$. Pela definição de corrente tempos $dq=idt$. Substituindo encontramos
		\begin{equation*}
			dW = \varepsilon idt.
		\end{equation*}
		Pela definição de potência, $P=\frac{dW}{dt}$, encontramos
		\begin{align*}
			dW & = \varepsilon idt,\\
			\Aboxed{P & = \varepsilon i.}
		\end{align*}
		sendo i a corrente que atravessa os terminais da bateria e $\varepsilon$ a f.e.m. da bateria.
	\end{multicols}
	\begin{block}{Potência elétrica de uma bateria}
		\begin{equation*}
			P = \varepsilon\cdot i.
		\end{equation*}	
	\end{block}
\end{frame}

\begin{frame}{A equação do circuito}
	\begin{columns}
		\begin{column}{0.7\textwidth}
   			\footnotesize
			Podemos resumir num circuito ao lado que a cada segundo
			\begin{itemize}
				\item Na bateria, a energia química se transforma em energia elétrica;
				\item No resistor, a energia elétrica se transforma em energia térmica;
				\item No motor, a energia elétrica se transforma em energia mecânica.
			\end{itemize}
			Por conservação da energia, a potência necessária para girar o motor e aquecer o resistor deve ser igual a potência fornecida pela bateria.
			\begin{align*}
				\underbracket{\varepsilon i}_{(Bateria)} & = \underbracket{\varepsilon'i+r'i^2}_{(Motor)}+\underbracket{Ri^2}_{(Resistor)},\\
				\Aboxed{\varepsilon\prime+r\prime i+Ri-\varepsilon & = 0.}
			\end{align*}
		\end{column}
		\begin{column}{0.3\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.35]{figuras/circuito.png}
				\caption*{\footnotesize Circuito fechado.}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{block}{Lei de Kirchhoff}
		A soma dos potenciais em um circuito fechado deve ser zero.
	\end{block}
\end{frame}

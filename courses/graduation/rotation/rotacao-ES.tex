\section{As variáveis de rotação}

\subsection{}

\begin{frame}{Introdução}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			Na mecânica podemos dizer que existem dois tipos de movimentos distintos:
			\begin{enumerate}
				\item [\ding{51}] Translação ou retilíneo: Os objetos se movem ao longo de linhas retas ou curvas;
				\item [\ding{51}] Rotação: Os objetos giram em torno de um eixo.
			\end{enumerate}
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.21]{figuras/rotacao-translacao.jpg}
				\caption*{\footnotesize Movimentos de rotação e translação da Terra.}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		O movimento mais geral de um corpo rígido se compõe de uma translação e uma rotação.
	\end{corollary}
\end{frame}

\begin{frame}{Posição angular}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			A posição angular de uma reta é o ângulo que a reta faz com uma direção fixa, em relação a posição angular zero. Na figura ao lado, a posição angular $\theta$ é medida em relação ao eixo x. De acordo com a figura podemos ver que
			\begin{align*}
				\Aboxed{\theta = \frac{s}{r},}
			\end{align*}
			onde s é o comprimento de um arco de raio r.
		\end{column}

\begin{frame}{Deslocamento angular}
	\begin{columns}
	\begin{column}[c]{0.6\textwidth}
		Se o corpo gira em torno do eixo de rotação variando o seu ângulo de $\theta_1$ para $\theta_2$ nos instante $t_1$ e $t_2$ respectivamente, o corpo sofre um deslocamento angular $\Delta\theta$, dado por
		\begin{equation*}
		\boxed{\Delta\theta=\theta_2-\theta_1.}
		\end{equation*}
		\begin{corollary}
			Por convenção adotamos que um deslocamento angular no sentido anti-horário é positivo, e um deslocamento no sentido horário é negativo.
		\end{corollary}

\begin{frame}{Correspondência entre os movimentos de translação e rotação}
	Em rotações, no caso onde a aceleração angular $\alpha$ for constante, podemos escrever as funções horárias do movimento angular de maneira análoga ao movimento retilíneo.
	\begin{table}
		\centering
		\caption*{\footnotesize Equações de movimento para aceleração linear constante e angular também constante.}
		\begin{tabular}[c]{lclc}
			\hline
			Equação & Variável que falta & Equação & Variável que falta\\
			\hline
			$v=v_0+at$ & $\Delta x$ & $\omega=\omega_0+\alpha t$ & $\Delta \theta$\\
			$\Delta x=v_0t+\frac{1}{2}at^2$ & $v$ & $\Delta \theta=\omega_0 t+\frac{1}{2}\alpha t^2$ & $\omega$\\
			$v^2=v_0^2+2a\Delta x$ & $t$ & $\omega^2=\omega_0^2+2\alpha\Delta \theta$ & $t$\\
			$\Delta x=\frac{1}{2}\left(v_0+v\right)t$ & $a$ & $\Delta \theta=\frac{1}{2}\left(\omega_0+\omega\right)t$ & $\alpha$\\
			$\Delta x=vt-\frac{1}{2}at^2$ & $v_0$ & $\Delta \theta=\omega t-\frac{1}{2}\alpha t^2$ & $\omega_0$\\
			\hline					
		\end{tabular}
	\end{table}				
\end{frame}

\section{Movimento circular}

\subsection{}

\begin{frame}{Relações entre as variáveis lineares e angulares}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Sabemos que uma partícula girando em torno de um eixo com raio $r$ realiza um arco s dado por $s=\theta r$. Se ela realiza um movimento circular, onde $r=cte$, derivando no tempo temos
			\begin{align*}
				\frac{ds}{dt} & = \frac{d(r\theta)}{dt},\\
				\frac{ds}{dt} & = r\frac{d\theta}{dt}.
			\end{align*}
			Pela figura podemos ver que $\frac{ds}{dt}=v$, portanto
			\begin{align*}
				\Aboxed{v=\omega r.}
			\end{align*}
		\end{column}
		\begin{column}[c]{0.4\textwidth}

\begin{frame}{Período de revolução}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Se $\omega =cte$, podemos dizer que o período de revolução $T$ (definido como o tempo gasto para a partícula percorrer uma volta completa) é dado pelo caminho percorrido pela partícula (que corresponde exatamente o comprimento s da circunferência de raio r). Portanto
			\begin{align*}
				T=\frac{2\pi r}{v},
			\end{align*}
			$v$ é conhecido como a velocidade tangencial realizado pela partícula.
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			 Sabendo que $v=\omega r$, temos
			\begin{align*}
				\Aboxed{T = \frac{2\pi \bcancel{r}}{\omega \bcancel{r}} = \frac{2\pi}{\omega}.}
			\end{align*}
			Definindo a frequência $\nu$ como o número de voltas por tempo, temos que
			\begin{align*}
				\nu & = \frac{1}{T},\\
				\Aboxed{\nu & = \frac{\omega}{2\pi}.}
			\end{align*} 
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Acelerações radial e tangencial}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Derivando a função $v=\omega r$ encontramos
			\begin{align*}
				\frac{dv}{dt} & = \frac{d\omega r}{dt},\\
				\frac{dv}{dt} & = r\frac{d\omega}{dt},\\
				\Aboxed{a_t & = \alpha r.}
			\end{align*}
			Sabendo que a aceleração radial $a_r$ corresponde a aceleração centrípeta $a_{cpt}$, temos portanto
			\begin{align*}
				\Aboxed{a_r = \frac{v^2}{r}=\omega^2 r.}
			\end{align*}
		\end{column}

\section{Torque, trabalho e energia cinética}

\subsection{}

\begin{frame}{Energia cinética de rotação}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Considere uma partículas girando em torno de um eixo, mantendo a sua distância r até o eixo constante ao longo do tempo. Sua energia cinética é dada por $K=\frac{1}{2}mv^2$, onde $\vec{v}$ é a sua velocidade tangencial. Sabendo que $v=\omega r$ temos
			\begin{align*}
				K & = \frac{1}{2}mv^2,\\
				K & = \frac{1}{2}m(\omega r)^2,
			\end{align*}
			Definimos $I=mr^2$ como o momento de
		\end{column}
		\begin{column}[t]{0.5\textwidth}
	 		inércia da partícula, onde para um conjunto de N partículas girando em torno de um eixo comum teremos
			\begin{align*}
				\Aboxed{I =\sum_{i}^{N}m_ir^2_i.}
			\end{align*}
			Podemos dizer que a energia cinética de rotação desse conjunto equivale a
			\begin{align*}
				\Aboxed{K =\frac{1}{2}I\omega^2.}
			\end{align*}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Torque}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Considere uma força $\vec{F}$ no ponto P de um objeto à uma distância r do ponto onde esse objeto irá girar. Essa força será responsável por fazê-lo girar em torno do eixo de rotação nessas condições. No entanto podemos decompô-la em
			\begin{enumerate}
				\item [\ding{51}] $\vec{F}_r$ na direção do raio de giro;
				\item [\ding{51}] $\vec{F}_t$ perpendicular ao raio de giro.
			\end{enumerate}
			Percebe-se que somente $\vec{F}_t$ irá contribuir para a rotação, onde
			\begin{align*}
				F_t = F\sin\phi.
			\end{align*}
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.3]{figuras/torque.png}
				\caption*{\footnotesize Rotação no ponto P devido a força $\vec{F}_t$.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Representação vetorial do torque}
\begin{columns}
	\begin{column}[c]{0.7\textwidth}
		A capacidade de $\vec{F}$ de fazer o objeto girar não depende apenas de $\vec{F}_t$, mas também da distância entre o ponto de aplicação P e o ponto O. Para levar em conta os dois fatores definimos uma grandeza chamada torque $\tau$, onde
		\begin{align*}
%			\tau & = (r)(F_t),\\
			\tau & = (r)(F\sin\phi).
		\end{align*}
		Porém, o torque é uma grandeza vetorial, pois a força aplicada em sentido contrário, ou se o ponto O estiver do lado oposto, o objeto poderá girar no sentido horário ou anti-horário. Portanto, é conveniente definir o torque na forma
		\begin{align*}
			\Aboxed{\vec{\tau} & = \vec{r}\times\vec{F}.}
		\end{align*}
	\end{column}

\begin{frame}{Trabalho em rotações}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			A expressão do trabalho $dW$ realizado por uma força $\vec{F}$ ao longo do caminho infinitesimal $ds$ é dado por $dW=\vec{F}\cdot d\vec{s}$. No entanto, no caso do movimento de rotação, percebemos que somente a componente $\vec{F}_t$ realiza efetivamente algum trabalho sobre a partícula, portanto temos um percurso ao longo do caminho c
			\begin{align*}
				W & = \int\limits_cF_t\cos\theta ds,\\
				W & = \int\limits_cF\sin\phi ds.
			\end{align*}
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.3]{figuras/torque.png}
				\caption*{\footnotesize Rotação no ponto P devido a força $\vec{F}_t$.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Trabalho e potência em rotações (continuação)}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Sabemos que o comprimento de um arco s é dado por $s=r\theta$. Considerando que o caminho infinitesimal $ds$ percorrido pela partícula, temos que $ds=rd\theta$. Fazendo uma substituição de variáveis
			\begin{align*}
				dW & = F\sin\phi rd\theta.
			\end{align*}
			Porém, pela definição de torque $\tau=rF\sin\phi$, portanto
			\begin{align*}
				dW & = \tau d\theta.
			\end{align*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			Integrando ao longo do caminho c temos
			\begin{align*}
				\Aboxed{W & = \int\limits_{\theta_1}^{\theta_2}\tau d\theta.}
			\end{align*}
			A potência pode ser determinada através da expressão $P=\frac{dW}{dt}$. Considerando uma rotação onde $\tau=cte$ teremos
			\begin{align*}
				\Aboxed{P & = \frac{\tau d\theta}{dt}=\tau\omega.}
			\end{align*}
		\end{column}
	\end{columns}
\end{frame}

\section{Segunda lei de Newton para rotações}

\subsection{}

\begin{frame}{Momento angular}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
%	\begin{multicols}{2}
		No movimento linear, temos de acordo com a segunda lei de Newton $\vec{F}=\frac{d\vec{p}}{dt}$. Substituindo na definição do torque
		\begin{align*}
			\vec{\tau} & = \vec{r}\times\vec{F},\\
			\vec{\tau} & = \vec{r}\times\frac{d\vec{p}}{dt}.
		\end{align*}
		No entanto, podemos perceber pela propriedade do produto vetorial
		\begin{align*}
			\frac{d(\vec{r}\times\vec{p})}{dt} & =\vec{r}\times\frac{d\vec{p}}{dt}+\frac{d\vec{r}}{dt}\times\vec{p},
%			\vec{r}\times\frac{d\vec{p}}{dt}=\frac{d(\vec{r}\times\vec{p})}{dt}-\frac{d\vec{r}}{dt}\times\vec{p}.
		\end{align*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			onde
		\begin{align*}
			%\frac{d(\vec{r}\times\vec{p})}{dt}=\vec{r}\times\frac{d\vec{p}}{dt}+\frac{d\vec{r}}{dt}\times\vec{p},\\
			\vec{r}\times\frac{d\vec{p}}{dt} & =\frac{d(\vec{r}\times\vec{p})}{dt}-\frac{d\vec{r}}{dt}\times\vec{p}.
		\end{align*}
		Substituindo na expressão do torque
		\begin{align*}
			\vec{\tau} & = \frac{d(\vec{r}\times\vec{p})}{dt}-\overbracket{\frac{d\vec{r}}{dt}}^{\alert{v}}\times\vec{p},\\
			\vec{\tau} & = \frac{d(\vec{r}\times\vec{p})}{dt}-\vec{v}\times m\vec{v}.
		\end{align*}
		onde sabemos que $\vec{v}=\frac{d\vec{r}}{dt}$ e $\vec{p}=m\vec{v}$.
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Momento angular (continuação)}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Porém, pela regra do produto vetorial, temos que $\vec{v}\times m\vec{v}=\vec{0}$, pois o produto de dois vetores com a mesma orientação é igual a zero no produto vetorial, portanto
			\begin{align*}
				\vec{\tau} & = \frac{d(\vec{r}\times\vec{p})}{dt}-\underbracket{\vec{v}\times m\vec{v}}_{\alert{0}},\\
				\vec{\tau} & = \frac{d(\vec{r}\times\vec{p})}{dt},\\
				\vec{\tau} & = \frac{d\vec{l}}{dt}.
			\end{align*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			A equação acima nos fornece uma nova grandeza física associado a rotação chamado momento angular $\vec{l}$, onde
			\begin{align*}
				\vec{l} = \vec{r}\times\vec{p}.
			\end{align*}
			Considerando um movimento circular onde $r=cte$, o módulo de $\vec{l}$ é dado por $l=mrv$, porém $v=\omega r$, substituindo
			\begin{align*}
			 	l & = mr\overbracket{v}^{\alert{\omega r}} = mr^2\omega=I\omega,\\
				\Aboxed{\vec{l} & = I\vec{\omega}.}
			 \end{align*}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Momento angular total de um sistema de partículas}
%	\begin{multicols}{2}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Considerando um sistema de partículas, o momento angular total $\vec{L}$ é a soma vetorial dos momentos angulares $\vec{l}$ de cada partícula do sistema,
			\begin{align*}
				\vec{L} = \vec{l}_1+\vec{l}_2+\dots + \vec{l}_n = \sum\limits_{i=1}^{N}\vec{l}_i.
			\end{align*}
			Derivando no tempo temos
			\begin{align*}
				\frac{d\vec{L}}{dt} & = \frac{d}{dt}\sum\limits_{i=1}^{N}\vec{l}_i=\sum\limits_{i=1}^{N}\frac{d\vec{l}_i}{dt}.
			\end{align*}
%		\end{multicols}
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.55]{figuras/sistema-particulas-girantes.jpeg}
				\caption*{\footnotesize Sistema de duas partículas girando em torno do eixo z.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Conservação do momento angular}
\begin{multicols}{2}
	Aplicando a definição de torque teremos
	\begin{align*}
	\frac{d\vec{L}}{dt} & = \sum\limits_{i=1}^{N}\frac{d\vec{l}_i}{dt},\\
	\frac{d\vec{L}}{dt} & = \sum\limits_{i=1}^{N}\vec{\tau}_i=\vec{\tau}_{\text{res}}.
	\end{align*}
	No caso especial onde o torque externo resultante atuando sobre as partículas for zero, o momento angular total $\vec{L}$ do sistema é conservado,
	\begin{align*}
	\Aboxed{\vec{L}=\text{constante}.}
	\end{align*}
\end{multicols}
\begin{block}{Lei da conservação do momento angular}
	Se a somatória dos torques externos atuando sobre um sistema de partículas é zero, o momento angular total desse sistema não pode variar.
\end{block}
\end{frame}

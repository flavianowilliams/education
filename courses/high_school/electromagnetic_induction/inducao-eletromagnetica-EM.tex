\section{Lei de Faraday}

\subsection{}

\begin{frame}{Definição de fluxo magnético}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			Definimos fluxo do campo magnético $\vec{B}$ que atravessa uma área A como a somatória das linhas de campo magnético que atravessam perpendicularmente essa área.
			\begin{block}{Fluxo do campo magnético}
				\begin{equation*}
					\varphi = BAcos\theta
				\end{equation*}
			\end{block}
		\end{column}
		\begin{column}{0.5\textwidth}
			\begin{figure}
				%\input{tikz/fluxo-magnetico.pgf}
				\begin{tikzpicture}[scale=0.75, transform shape, x={(-10:1cm)},y={(90:1cm)},z={(210:1cm)}]
				
				\filldraw[draw, opacity=0.05] (2,0,0) arc (0:360:2);
				\draw[line width=1pt,dashed] (2,0,0) arc (0:360:2);
				%	 \node[right] at (2,0,0) {$i$};
				\draw[line width=1pt,->] (0,0,0) -- (0,0,1) node[below] {$\hat{n}$};
				
				%vetor B
				\foreach \x in {-1,0,1,2}{
					\foreach \y in {0}{
						\draw[line width=1pt,color=gray] (0,-0.5+\x,-1+\x) -- (0,0+\x,0+\x);
						\draw[line width=1pt,->] (0,0+\x+\y,0+\x) -- (0,0.5+\x+\y,1+\x) node[above] {$\vec{B}$};}}
				
				\draw[->] (2,2,0) node[right] {A} to [out=180, in=90] (1,1,0) ;
				\draw [<->] (0,0.25,0.5) arc (180:200:0.5) node[left] {\footnotesize $\theta$};
				\node at (-1.5,-1.5,1) {$\varphi_{B}=BAcos\theta$};
				\node[text width=6cm] at (1.25,-2,1) {\small $\varphi_{B}$: B por area perpendicular a espira};
				
				\end{tikzpicture}
				\caption*{\footnotesize Linhas de campo magnético atravessando a área A demarcada pela linha tracejada, em um ângulo $\theta$.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Variação do fluxo magnético ao longo de uma espira}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			\begin{block}{Lei de Faraday}
				Sempre que houver uma variação do fluxo magnético através de um circuito, aparecerá, nesse circuito, uma f.e.m. induzida,
				\begin{equation*}
					\left|\frac{\Delta \varphi}{\Delta t}\right| = \varepsilon
				\end{equation*}
			\end{block}
		\end{column}
		\begin{column}{0.5\textwidth}
			\begin{figure}
				%\input{tikz/lei-faraday.pgf}
				\begin{tikzpicture}[scale=0.65, transform shape, x={(-10:1cm)},y={(90:1cm)},z={(210:1cm)}]
				
				\filldraw[draw, opacity=0.2] (2,0,0) arc (0:360:2);
				\draw[line width=1pt,color=red,<-] (2,0,0) arc (0:360:2);
				\node[right] at (2,0,0) {$i$};
				\draw[line width=1pt,->] (0,0,0) -- (0,0,1) node[below] {$\hat{n}$};
				\draw[line width=1pt,->] (0,0,0) -- (0,0.5,1) node[above] {$\vec{B}$};
				\draw[->] (2,2,0) node[right] {A} to [out=180, in=90] (1,1,0) ;
				\draw (0,0.25,0.5) arc (180:200:0.5) node[left] {\footnotesize $\theta$};
				\node at (-1.5,-1.5,1) {$\varphi_{B}=BAcos\theta$};
				\node[text width=6cm] at (1.25,-2,1) {\small $\varphi_{B}$: B por area perpendicular a espira};
				
				\end{tikzpicture}
				\caption*{\footnotesize Sentido da corrente i contornando a área A.}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		No SI a unidade de medida de fluxo magnético é Weber (Wb).
	\end{corollary}
\end{frame}

\section{Lei de Lenz}

\subsection{}

\begin{frame}{Sentido da corrente induzida em relação a variação do fluxo}
	\begin{block}{Lei de Lenz}
		A corrente induzida em um circuito aparece sempre com um sentido tal que o campo magnético que ela cria tende a contrariar a variação do fluxo magnético que a originou.
	\end{block}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{figure}
				\centering
				\begin{subfigure}[b]{0.45\textwidth}
					\centering
					\includegraphics[scale=0.2]{figuras/figura-2a}
					\caption*{\footnotesize $\frac{\Delta \varphi}{\Delta t}>0$ no tempo.}
				\end{subfigure}
				\begin{subfigure}[b]{0.45\textwidth}
					\centering
					\includegraphics[scale=0.2]{figuras/figura-2b}
					\caption*{\footnotesize $\frac{\Delta \varphi}{\Delta t}<0$ no tempo.}
				\end{subfigure}
			\end{figure}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{block}{Lei de Faraday-Lenz}
				\begin{equation*}
					\frac{\Delta \varphi}{\Delta t} = \alert{-}\varepsilon
				\end{equation*}
			\end{block}
		\end{column}
	\end{columns}
\end{frame}

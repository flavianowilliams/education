\section{Temperatura}

\subsection{}

\begin{frame}{Temperatura}
	\begin{itemize}
		\item Usamos a termodinâmica para analisar sistemas contendo uma quantidade muito grande de partículas, moléculas ou átomos;
		\item Para estudar o sistema, usamos conceitos como temperatura, pressão e volume ao invés de força, massa e aceleração;
		\item A temperatura é uma grandeza que caracteriza os estados de agitação dos objetos;
		\item Usamos o termômetro para medir a temperatura dos objetos.
	\end{itemize}
\end{frame}

\begin{frame}{Equilíbrio térmico}
	\begin{corollary}
		\justifying
		Dois ou mais objetos, inicialmente a temperaturas diferentes tendem para um estado final chamado de \alert{equilíbrio térmico}, onde as temperaturas de ambos buscam se igualarem.
	\end{corollary}
	\begin{figure}[c]
		\includegraphics[scale=0.35]{figuras/figura-8.png}
	\end{figure}
\end{frame}

\begin{frame}{Escalas termométricas}
	\begin{itemize}
		\item Assim como usamos uma régua para medir distâncias, para medir temperaturas é necessário graduar o termômetro.
		\item O termômetro mais utilizado é constituído por um tubo fino preenchido por mercúrio, que possui elevado coeficiente de dilatação.
	\end{itemize}
	\begin{block}{Principais escalas de temperatura}
		\begin{itemize}
			\item Celsius ou Centígrada: Baseia-se nos pontos de fusão e ebulição da água pura;
			\item Fahrenheit: Os pontos de fusão e ebulição da água pura estão marcados em $32\;^oF$ e $212\;^oF$;
			\item Kelvin: Baseia-se no zero absoluto (estado de mínima energia possível);
		\end{itemize}
	\end{block}
\end{frame}

\begin{frame}[fragile]{Por que um sólido dilata?}
	\begin{small}
		\begin{itemize}
			\item Os átomos que constituem o sólido distribuem-se ordenadamente, dando origem a uma estrutura denominada rede cristalina do sólido.
			\item A força entre os átomos são como molas que comprimem e esticam.
			\item Quando a temperatura aumenta, há um aumento na agitação dos átomos, que ao vibrarem se afastam mais uns dos outros. Com isso, a distância média entre os átomos torna-se maior.
		\end{itemize}
	\end{small}

\begin{frame}{Dilatação linear}

				\Delta L & \alpha \Delta T,\\
				\Delta L & \alpha L_0.
			\end{align*}
			Portanto, podemos dizer que
			\begin{align*}
				\Delta L & \alpha L_0\Delta T,\\
				\Aboxed{\Delta L & = \alpha L_0\Delta T,}

\begin{frame}{Coeficiente de dilatação linear}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\justifying
			Sabendo que $L=L_0+\Delta L$, temos
			\begin{align*}
				L & = L_0+\Delta L,\\
				L & = L_0+\alpha L_0\Delta T,\\
				L & = L_0(1+\alpha\Delta T),
			\end{align*}
			onde $\alpha$ é a constante de proporcionalidade chamada coeficiente de dilatação linear.
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{table}
				\centering
				\caption*{Coeficiente de dilatação linear.}
				\begin{tabular}{cc}
					\hline
					Substância & $\alpha\;(^{\circ}C^{-1})$\\
					\hline
					\hline
					Alumínio & \SI{23e-6}{}\\
					Cobre & \SI{17e-6}{}\\
					Vidro refratário & \SI{3,2e-6}{}\\
					Diamante & \SI{0,9e-6}{}\\
					\hline
				\end{tabular}
				%\includegraphics[scale=0.3]{figuras/figura-3.png}
			\end{table}
		\end{column}
	\end{columns}
	\begin{corollary}
		A unidade de medida do coeficiente de dilatação linear é $\frac{1}{^oC}=^oC^{-1}$ ou $\frac{1}{K}=K^{-1}$.
	\end{corollary}
\end{frame}

\begin{frame}{Dilatação superficial}

\begin{frame}{Coeficiente de dilatação superficial}
	Sabendo que $A=A_0+\Delta A$, temos
	\begin{align*}
		A & = A_0+\Delta A,\\
		A & = A_0+\beta A_0\Delta T,\\
		A & = A_0(1+\beta\Delta T),
	\end{align*}
	onde $\beta$ é a constante de proporcionalidade chamada coeficiente de dilatação superficial.
	\begin{corollary}
		A relação entre os coeficientes de dilatação linear e superficial é dado por
		\begin{equation*}
			\beta = 2\alpha
		\end{equation*}
	\end{corollary}
\end{frame}

\begin{frame}{Dilatação volumétrica}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			Assim como no caso da barra, se a variação da temperatura não for muito alta, podemos supor que a dilatação de uma placa ($\Delta V$) é proporcional a variação da temperatura e a área da barra ($V_0$),
			\begin{align*}
				\Delta V & \alpha \Delta T,\\
				\Delta V & \alpha V_0.
			\end{align*}
			Portanto, podemos dizer que
			\begin{align*}
				\Aboxed{\Delta V & = \gamma V_0\Delta T,}
			\end{align*}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.35]{figuras/figura-5.png}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Coeficiente de dilatação volumétrica}
	\begin{multicols}{2}
		Sabendo que $V=V_0+\Delta V$, temos
		\begin{align*}
			V & = V_0+\Delta V,\\
			V & = V_0+\gamma V_0\Delta T,\\
			V & = V_0(1+\gamma\Delta T),
		\end{align*}
		onde $\gamma$ é a constante de proporcionalidade chamada coeficiente de dilatação volumétrica. Se $\Delta T$ for consideravelmente pequeno, podemos dizer que $\gamma = 3\alpha$.
	\end{multicols}
	\begin{corollary}
		A relação entre os coeficientes $\alpha$, $\beta$ e $\gamma$ é dado por
		\begin{equation*}
			\gamma = \frac{3}{2}\beta = 3\alpha.
		\end{equation*}
	\end{corollary}
\end{frame}

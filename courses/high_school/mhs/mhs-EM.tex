			\end{figure}
		\end{column}
		\begin{column}{0.6\textwidth}
			\begin{itemize}
				\item k: constante elástica (depende das propriedades do material);
				\item Se $x_{0}=0$, pela Lei de Hooke $\vec{F}=-k\vec{x}$.
			\end{itemize}
			\begin{align*}
				\vec{F} & = m\vec{a},\\
				\vec{a} & = -\frac{k}{m}\vec{x}.
			\end{align*}
			\begin{corollary}
				A aceleração do objeto e a força restauradora possuem sentidos contrários ao deslocamento.
			\end{corollary}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{MHS e movimento circular uniforme}
	\begin{multicols}{2}
		Pela Lei de Hooke, a aceleração máxima do objeto é dado por
		\begin{equation*}
			a_m = \frac{k}{m}x_m.
		\end{equation*}
		Foi mostrado anteriormente que
		\begin{equation*}
			a_{cpt} = \omega^2x_m,
		\end{equation*}
		portanto
		\begin{equation*}
			\boxed{\omega = \sqrt{\frac{k}{m}}.}
		\end{equation*}
		Levando em consideração que $\omega=\frac{2\pi}{T}$.
	\end{multicols}
	\begin{block}{Período de oscilação do sistema massa-mola}
		\[T = 2\pi\sqrt{\frac{m}{k}}.\]
	\end{block}
\end{frame}

		\end{column}
		\begin{column}[c]{0.6\textwidth}
			Se $\theta\ll 1$ temos $sen\left(\theta\right)\approx \theta=\frac{x}{L}$,
			\begin{align*}
				F_R & = -mgsen\left(\theta\right),\\
				\bcancel{m}a & = -\bcancel{m}\frac{g}{L}x,\\
				a & = -\frac{g}{L}x.
			\end{align*}
			Comparando com o sistema massa-mola temos $a=-\omega^{2} x$, ou seja,
			\begin{align*}
				\omega^2 = \frac{g}{L}.
			\end{align*}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Pêndulo simples (continuação)}
	\begin{columns}[c]
		\begin{column}[c]{0.5\textwidth}
			Sabendo que o quadrado da frequência angular de oscilação do pêndulo simples equivale a $\omega^2=\frac{g}{L}$ temos
			\begin{align*}
				\omega = \sqrt{\frac{g}{L}}.
			\end{align*}
			Porém, foi mostrado anteriormente que $\omega=\frac{2\pi}{T}$, portanto
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{align*}
				\frac{2\pi}{T} = \sqrt{\frac{g}{L}}.
			\end{align*}
			\begin{block}{\small Período de oscilação do pêndulo simples}
				\begin{align*}
					T = 2\pi\sqrt{\frac{L}{g}}.
				\end{align*}
			\end{block}
		\end{column}
	\end{columns}
	\begin{corollary}
	\href{https://www.youtube.com/watch?v=Ab9OB9Q6QNw&list=PL1Dg4Oxxk_RI2Ppb541vQyaUbqUuXtiuJ}{\textcolor{blue}{O período de oscilação do pêndulo depende somente do comprimento L do fio.}}
\end{corollary}
\end{frame}

\begin{frame}{Transformar um número em notação científica}
	\begin{corollary}
		\begin{enumerate}[leftmargin=1.75cm]
			\item[Passo 1:] Escrever o número incluindo a vírgula.
			\item[Passo 2:] Andar com a vírgula até que reste somente um número diferente de zero no lado esquerdo.
			\item[Passo 3:] Colocar no expoente da potência de 10 o número de casas decimais que tivemos que "andar" com a vírgula. Se ao andar com a vírgula o valor do número diminuiu, o expoente ficará positivo, se aumentou o expoente ficará negativo.
		\end{enumerate}
	\end{corollary}
	\begin{exampleblock}{Exemplo}
		\begin{equation*}
			6\; 590\; 000\; 000\; 000\; 000,0 = \SI{6,59e15}{}		
		\end{equation*}
	\end{exampleblock}
\end{frame}

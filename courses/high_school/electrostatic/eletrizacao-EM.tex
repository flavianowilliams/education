\section{Condutividade nos sólidos}

\subsection{}

\begin{frame}{O modelo atômico}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			A matéria é constituída por átomos que por sua vez são constituídos pelo núcleo atômico formado por prótons e nêutrons e ao redor orbitam os elétrons em camadas eletrônicas bem definidas. No modelo atômico os prótons e nêutrons estão presos no núcleo enquanto que os elétrons possui mair mobilidade ao redor do núcleo. Os elétrons das camadas mais internas estão mais fortemente ligados ao núcleo, enquanto que os elétrons das camadas mais externas (\alert{valência}) sofrem maiores influências de forças externas.
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.6]{figuras/modelo-atomico.png}
				\caption*{\footnotesize Modelo atômico.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Classificação dos sólidos de acordo com a sua capacidade de conduzir eletricidade}
	\begin{enumerate}
		\item[\ding{51}] \alert{Condutores}: Nos condutores os elétrons possuem liberdade para se moverem ao longo do material.
		\item[\ding{51}] \alert{Isolantes}: Nos isolantes os elétrons estão mais fortemente ligados ao átomos, e assim possuem menos liberdade para se moverem.
		\item[\ding{51}] \alert{Semicondutores}: Nos semicondutores os elétrons podem se mover ao longo do material sob determinadas situações, como aumento da sua temperatura ou a aplicação de um campo elétrico;
		\item[\ding{51}] \alert{Supercondutores}: Nos supercondutores, as cargas elétricas fluem no material sem praticamente nenhuma resistência, e assim sem perdas de energia.
	\end{enumerate}
	\begin{corollary}
		Condutividade é usada para medir o nível de condução de um material.
	\end{corollary}
\end{frame}

\begin{frame}{Mar de elétrons}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Os sólidos cristalinos possuem normalmente poucos elétrons na sua camada de valência, além de estarem pouco atraídos pelo núcleo, o que facilita que esses elétrons da última camada escapem com certa facilidade, tornando elétrons livres que transitam entre os átomos do reticulado. Os átomos que perdem os elétrons se tornam cátions, mas eles podem logo receber elétrons e voltar a se tornar átomos neutros. Esse efeito gera uma mobilidade de cargas no material, o que justifica a sua condutividade elétrica.
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.6]{figuras/eletrons-livres.jpg}
				\caption*{\footnotesize Mar de elétrons livres.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Polarização de um material}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{enumerate}
				\item[\ding{51}] Um material dielétrico é formado por moléculas polares;
				\item[\ding{51}] A carga total de um dielétrico é zero;
				\item[\ding{51}] Naturalmente os dipolos elétricos são orientados de maneira aleatória;
				\item[\ding{51}] Na presença de um objeto eletrizado os dipolos elétricos se alinham e a carga residual é distribuída na superfície;
			\end{enumerate}

\section{Processos de eletrização}

\subsection{}

\begin{frame}{Eletrização por atrito}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{enumerate}
				\item[\ding{51}] Um objeto no seu estado normal possui um número de prótons igual ao número de elétrons;
				\item[\ding{51}] Se perder elétrons ficará com excesso de prótons e estará eletrizado positivamente;
				\item[\ding{51}] Se receber elétrons ficará com excesso de elétrons e estará eletrizado negativamente;
				\item[\ding{51}] A eletrização ocorre entre dois \alert{materiais isolantes} que obedecem a série triboelétrica.
			\end{enumerate}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}[c]
				\includegraphics[scale=0.26]{figuras/serie-triboeletrica.jpg}
				\caption*{\footnotesize Série triboelétrica.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Eletrização por contato}
	\begin{enumerate}
		\item[\ding{51}] A eletrização por contato acontece de maneira efetiva \alert{se os dois objetos forem condutores};
		\item[\ding{51}] Após o contato ambos os objetos ficam eletrizados com cargas de mesmo sinal;
	\end{enumerate}
	\begin{figure}[c]
		\includegraphics[scale=0.35]{figuras/eletrizacao-por-contato.jpg}
		\caption*{\footnotesize Processo de eletrização de um condutor.}
	\end{figure}
\end{frame}

\begin{frame}{Eletrização por indução de um condutor}
	\begin{enumerate}
		\item[\ding{51}] As cargas elétricas se deslocam no \alert{condutor} se localizando em extremidades opostas na superfície;
		\item[\ding{51}] Aterrando o condutor o objeto fica eletrizado.
	\end{enumerate}
	\begin{figure}[c]
		\includegraphics[scale=0.35]{figuras/eletrizacao-por-inducao.png}
		\caption*{\footnotesize Processo de eletrização por indução de um condutor.}
	\end{figure}
\end{frame}

\begin{frame}{Eletrização por indução de um dielétrico}
	\begin{enumerate}
		\item[\ding{51}] Os dipolos se alinham na presença de um objeto eletrizado e a carga residual fica localizado em extremidades opostas na superfície;
		\item[\ding{51}] Rigidez dielétrica é o limite que o dielétrico possui antes de haver condução de elétrons no interior.
	\end{enumerate}

\begin{frame}{Transformar um número em notação científica}
	\begin{corollary}
		\begin{enumerate}[leftmargin=1.75cm]
			\item[Passo 1:] Escrever o número incluindo a vírgula.
			\item[Passo 2:] Andar com a vírgula até que reste somente um número diferente de zero no lado esquerdo.
			\item[Passo 3:] Colocar no expoente da potência de 10 o número de casas decimais que tivemos que "andar" com a vírgula. Se ao andar com a vírgula o valor do número diminuiu, o expoente ficará positivo, se aumentou o expoente ficará negativo.
		\end{enumerate}
	\end{corollary}
	\begin{exampleblock}{Exemplo}
		\begin{equation*}
			6\; 590\; 000\; 000\; 000\; 000,0 = \SI{6,59e15}{}		
		\end{equation*}
	\end{exampleblock}
\end{frame}

\begin{frame}{O que é força?}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			Quando exercemos um esforço para puxar ou empurrar um objeto estamos na verdade interagindo com ele. Essa interação capaz de alterar o seu estado de movimento é chamado de força. Podemos dizer que, para que o efeito de uma força fique totalmente definido, será necessário especificar o seu módulo, direção e sentido.
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.6]{figuras/forca.jpeg}
				\caption*{\footnotesize Todas as forças atuando na caixa.}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		Toda força é uma grandeza vetorial.
	\end{corollary}
\end{frame}

\begin{frame}{Medida de força}
	Um exemplo de força é a "força peso", que representa a interação da Terra atraindo objetos que possuem massa. No caso da força, a unidade de medida escolhida pelos físicos é o peso do quilograma-padrão, ou seja, o peso de um o objeto que possui a massa do quilograma-padrão.
	
	Outra unidade muito usada para medir força é o Newton. Esta unidade é utilizada para satisfazer as unidades de medida do sistema internacional. O seu fator de conversão seria
	\begin{align*}
			\si{1}{kgf} = \si{9,8}{N}.
	\end{align*}
	\begin{corollary}
		Um quilograma-força (1kgf) é o peso do quilograma-padrão, ao nível do mar e a 45 graus de latitude.
	\end{corollary}

\begin{frame}{Inércia}
	Medimos a inércia como a capacidade de um objeto mudar o seu estado de movimento sob a ação de uma força. Ou seja, se estiver em repouso ele permanecerá em repouso, e se estiver em movimento, ele permanecerá com a mesma velocidade.
	\begin{figure}
		\begin{subfigure}[b]{0.45\textwidth}
			\centering
			\includegraphics[scale=0.15]{figuras/inercia-maior.jpg}
			\caption*{\footnotesize Objeto com maior inércia.}
		\end{subfigure}
		\begin{subfigure}[b]{0.45\textwidth}
			\centering
			\includegraphics[scale=0.075]{figuras/inercia-menor.jpg}
			\caption*{\footnotesize Objeto com menor inércia.}
		\end{subfigure}
	\end{figure}
	\begin{corollary}
		Geralmente a inércia aumenta com a massa do objeto.
	\end{corollary}
\end{frame}

\begin{frame}{Primeira Lei de Newton}
	\begin{corollary}
		Se nenhuma força resultante atua sobre um objeto, a velocidade não pode mudar, ou seja, o objeto não pode sofrer aceleração.
	\end{corollary}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Na figura ao lado, se $F_1=F_2$ e  $F_3=F_4$ temos que a força resultante será zero, portanto o objeto permanecerá com a mesma velocidade $\vec{v}$.
			\begin{corollary}
				Mesmo que um objeto esteja submetido a várias forças, se a resultante das forças for zero, o objeto não sofrerá aceleração.
			\end{corollary}
		\end{column}

\begin{frame}{Terceira Lei de Newton - Ação e reação}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			\begin{block}{Terceira Lei de Newton}
				Quando dois objetos (1 e 2) interagem, as forças qua cada objeto exerce sobre o outro são iguais em módulo e têm sentidos opostos,
				\begin{equation*}
					\vec{F}_{12}=-\vec{F}_{21}.
				\end{equation*}
			\end{block}
			\begin{corollary}
				A somatória das forças no sistema é zero.
			\end{corollary}
		\end{column}

	\end{columns}
\end{frame}

\section{Segunda lei de Newton}

\subsection{}

\begin{frame}{Segunda Lei de Newton}
	\begin{corollary}
		A força resultante que age sobre um objeto é igual ao produto da massa do objeto pela aceleração. Em termos matemáticos, podemos escrever como
		\begin{equation*}
			\vec{F}_{\text{res}}=m\vec{a}.
		\end{equation*}
	\end{corollary}
	\begin{columns}
		\begin{column}[c]{0.45\textwidth}
			\begin{corollary}
				No SI a unidade de medida da força é Newton (N),
				\begin{equation*}
					\SI{1}{N}=\SI[per-mode=symbol]{1}{\kilogram\meter\per\square\second}
				\end{equation*}
			\end{corollary}
		\end{column}
		\begin{column}[c]{0.55\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.32]{figuras/segunda-lei-de-newton.jpg}
				\caption*{\footnotesize Relação entre força, massa e aceleração.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Força gravitacional}
	\begin{columns}
		\begin{column}[c]{0.7\textwidth}
			Considere o objeto de massa m caindo em quedra livre. Nesse caso a única força atuando nele é a força da gravidade. Relacionando com a segunda lei de Newton ($\vec{F}=m\vec{a}$) temos
			\begin{equation*}
				\vec{F}=m\vec{g}.
			\end{equation*}
			Considerando que o deslocamento que o objeto realiza é insignificante ao tamanho da Terra, podemos dizer que $\vec{g}$ é praticamente constante.
		\end{column}
		\begin{column}[c]{0.3\textwidth}
			\begin{figure}[c]
				\begin{footnotesize}
					\begin{table}
						\centering

						%						\vspace*{-0.5cm}
						\begin{tabular}[c]{cc}
							\hline
							Latitude & $g(m/s^2)$\\
							\hline
							0 & 9,7803\\
							30 & 9,7932\\
							45 & 9,8017\\
							60 & 9,8191\\
							90 & 9,8322\\
							\hline					
						\end{tabular}
					\end{table}				
				\end{footnotesize}
				%\includegraphics[scale=0.19]{figuras/figura-8.png}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		O peso P de um objeto é igual ao módulo da força gravitacional que age sobre o objeto, aplicando nele uma aceleração igual a g.
	\end{corollary}
\end{frame}

\begin{frame}{Força Normal}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Considere um bloco de massa m pressionando uma mesa para baixo devido a força da gravidade $\vec{F}_g$. Pela terceira lei de Newton, a mesa irá empurrar o bloco para cima aplicando uma força $\vec{F}_N$ com a mesma intensidade.
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.25]{figuras/normal.jpg}
				\caption*{\footnotesize Forças peso $\vec{P}$ e normal $\vec{N}$.}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		\begin{enumerate}
			\item [\ding{51}] Quando um objeto exerce uma força sobre uma superfície, a superfície (ainda que aparentemente rígida) se deforma e empurra o objeto com uma força normal que é perpendicular à superfície;
			\item[\ding{51}] A força normal será sempre perpendicular a superfície de contanto.
		\end{enumerate}
		
	\end{corollary}
\end{frame}

\begin{frame}{Tração}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			Quando uma corda é presa a um objeto e esticada, a corda aplica ao objeto uma força $\vec{T}$ orientada na direção da corda. Essa força é chamada força de tração.
			\begin{enumerate}
				\item [\ding{51}] A corda é frequentemente considerada sem massa;
				\item [\ding{51}] As forças nas duas extremidades da corda são iguais em módulo.
			\end{enumerate}
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.7]{figuras/tracao.JPG}
				\caption*{\footnotesize Tração atuando na corda e nos blocos A e B.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Força de atrito}
	A experiência mostra que, quando um objeto não lubrificado pressiona uma superfície nas mesmas condições, a força de atrito possui três propriedades:
	\begin{enumerate}
		\item [\ding{51}] Se o objeto não se move, a força de atrito se iguala em módulo a força $\vec{F}$;
		\item [\ding{51}] A força de atrito possui valor máximo de $f_{max}=\mu_s N$, onde $\mu_s$ é o coeficiente de atrito estático e N o módulo da força normal;
		\item [\ding{51}] Se o objeto começa a deslizar na superfície, o módulo da força de atrito diminui rapidamente para um valor dado por $f_k=\mu_k N$, onde onde $\mu_k$ é o coeficiente de atrito cinético, $\mu_k<\mu_s$.
	\end{enumerate}
\end{frame}

\begin{frame}{Relação entre força de atrito e a força atuando no objeto (continuação)}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			\begin{enumerate}
				\item [\ding{51}] A força de atrito é zero se não há outras forças atuando no objeto;
				\item [\ding{51}] A força de atrito é igual a força externa se o objeto está em repouso;
				\item [\ding{51}] A força de atrito máxima é igual a $\mu_s N$.
				\item [\ding{51}] Se o objeto está em movimento a força de atrito é igual a $\mu_k N$.
			\end{enumerate}
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{figure}

			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Força de arrasto}
	\begin{columns}
		\begin{column}[c]{0.7\textwidth}
			Quando um objeto se movimenta na presença de um fluido como o ar, ele experimenta uma força de resistência ao seu movimento chamada força de arrasto dado por
			\begin{equation*}
				F_{arrasto}=bv^2,
			\end{equation*}
			onde b  é o coeficiente de arrasto que depende da densidade do ar e da área de contato do objeto. $F_{arrasto}$ aumenta com o quadrado da velocidade, portanto à medida que o objeto acelera $F_{arrasto}$ aumenta.

\begin{frame}{Força de arrasto (continuação)}
	\begin{columns}
		\begin{column}[c]{0.7\textwidth}
			A força de arrasto é contrária a força da gravidade, portanto à medida que $F_{arrasto}$ aumenta a aceleração diminui. Portanto, a velocidade atinge um limite chamado velocidade terminal $v_t$. Aplicando a segunda lei de Newton,
			\begin{equation*}
				bv^2-F_g = 0.
			\end{equation*}
			\begin{block}{Velocidade terminal}
				\begin{equation*}
					v_t=\sqrt{\frac{F_g}{b}}.
				\end{equation*}
			\end{block}
		\end{column}

			\begin{figure}

\begin{frame}{Força centrípeta}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Vimos que um objeto que descreve uma trajetória circular de raio r está sujeito a uma aceleração apontada para o centro da circunferência chamada aceleração centrípeta, $a=v^2/r$. Pelas leis de Newton, a aceleração em um objeto é devido a uma força resultante, portanto podemos dizer que existe uma força sendo aplicada no caso do movimento circular. Essa força é chamada de força centrípeta,
			\begin{equation*}
				\boxed{F=m\frac{v^2}{r}.}
			\end{equation*}
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}

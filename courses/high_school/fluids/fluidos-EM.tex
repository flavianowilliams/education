\section{Definição de fluido}

\subsection{}

\begin{frame}{O que é um fluido?}
	\begin{block}{Fluido}
		\justifying{Líquido incompressível que assume a forma do recipiente que o contém.}
	\end{block}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{itemize}
				\item [\ding{51}] Os sólidos são objetos que possuem forma definida, portanto não são fluidos;
				\item [\ding{51}] O Volume de um fluido não se altera independente da temperatura que se encontra e do recipiente que o contém.
				\item [\ding{51}] Conceitos como densidade e pressão são usados ao invés de massa e força.
			\end{itemize}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}[c]
				\includegraphics[scale=0.25]{figuras/fluido.jpg}
				\caption*{\footnotesize Formato de um fluido em diversos recipientes.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Fluido versus corpo rígido}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			\begin{block}{Corpo rígido}
				\begin{small}
					\begin{itemize}
						\item [\ding{51}] Formato rígido e imutável;
						\item [\ding{51}] Corpos homogêneos ou heterogêneos (a densidade pode variar ao longo da estrutura);
						\item [\ding{51}] Usamos conceitos de massa e força.
					\end{itemize}
				\end{small}
			\end{block}
			\begin{figure}[c]
				\includegraphics[scale=0.18]{figuras/corpo-rigido.png}
				\caption*{\footnotesize Exemplo de um corpo rígido.}
			\end{figure}
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{block}{Fluido}
				\begin{small}
					\begin{itemize}
						\item [\ding{51}] Formato flexível e se adapta ao recipiente;
						\item [\ding{51}] Corpos homogêneos (Densidade é a mesma ao longo do fluido);
						\item [\ding{51}] Usamos conceitos de densidade (massa específica) e pressão.
					\end{itemize}			
				\end{small}
			\end{block}
			\begin{figure}[c]
				\includegraphics[scale=0.1]{figuras/fluido-2.jpg}
				\caption*{\footnotesize Exemplo de um fluido.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\section{Pressão}

\subsection{}

\begin{frame}{O que é Pressão?}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			\begin{block}{Pressão}
				\justifying{A pressão p é a relação entre o módulo da força $\vec{F}$ que atua ao longo da área A.}
				\begin{align*}
				p = \frac{F}{A}.
				\end{align*}
			\end{block}	
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}
				\centering
				\begin{tikzpicture}[scale=0.5,transform shape]
				
				\tkzInit[xmin=-2,xmax=5,ymin=-2,ymax=2.5]
				\tkzClip
				%\tkzGrid
				
				% cubo 1
				\pgfmathsetmacro{\cubex}{5}
				\pgfmathsetmacro{\cubey}{5}
				\pgfmathsetmacro{\cubez}{5}
				
				\draw (0,0,0) -- ++(0,0,\cubez) -- ++(\cubex,0,0) -- ++(0,0,-\cubez) -- cycle;
				\draw[gray,fill=gray!30] (0,0.5,0) -- ++(0,0,\cubez) -- ++(\cubex,0,0) -- ++(0,0,-\cubez) -- cycle;
				\draw[gray,fill=gray!30] (0,0,\cubez) -- ++(\cubex,0,0) -- ++(0,0.5,0) -- ++(-\cubex,0,0) -- cycle;
				\draw[gray,fill=gray!30] (\cubex,0,0) -- ++(0,0,\cubez) -- ++(0,0.5,0) -- ++(0,0,-\cubez) -- cycle;
				
				\node[above,color=red,font=\Large] at (0.5,1.5,0.5) {$\frac{\vec{F}}{A}$};

				\foreach \dx in {1,2,3,4,5}{
					\foreach \dz in {1,2,3,4,5}{
						\draw[blue] (0,0.5,0) -- ++(0,0,\dz) -- ++(\dx,0,0) -- ++(0,0,-\dz) -- cycle;}}
					
				\foreach \dx in {1,2,3,4,5}{
					\foreach \dz in {1,2,3,4,5}{
						\draw[<-,line width=1.5pt,color=red] (-0.5+\dx,0.5,\dz-0.5) --++ (0,1,0);}}
					
				\end{tikzpicture}
				%\includegraphics[scale=0.25]{figuras/pressao.jpg}
				\caption*{\footnotesize Força por unidade de área.}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		Pela definição de pressão (p=F/A), temos que a unidade de medida no SI é o Pascal (Pa), onde a força F e a área A também devem estar no SI,
		\begin{align*}
			\SI{1}{\pascal} = \SI[per-mode=symbol]{1}{\newton\per\square\metre}.
		\end{align*}
	\end{corollary}
\end{frame}

\section{Massa específica}

\subsection{}

\begin{frame}{O que é massa específica?}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{block}{Massa específica}
				\justifying{Massa específica ou densidade absoluta de um objeto é a razão entre sua massa e seu volume.}
				\[\rho = \frac{m}{V}\]
			\end{block}	
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}[c]
				\includegraphics[scale=0.18]{figuras/densidade-iceberg.jpg}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		\small
		Pela definição de densidade ($\rho=m/v$), percebe-se que a sua unidade no SI deve ser dada pela relação da massa m em $kg$ e o volume V em $m^3$.
		\begin{align*}
			1 \frac{g}{cm^3} = \frac{\SI{1e-3}{kg}}{\SI{1e-6}{\cubic\metre}} = 10^3\frac{kg}{m^3}.
		\end{align*}
	\end{corollary}
\end{frame}

\begin{frame}{A experiência de Arquimedes}
	Arquimedes mergulhou em um recipiente contendo água:
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{itemize}
				\item Uma massa de ouro pura igual a massa da coroa;
				\item Uma massa de prata pura igual a massa da coroa;
				\item A coroa em questão;
			\end{itemize}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}[c]
				\includegraphics[scale=0.25]{figuras/arquimedes.jpg}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		\justifying Arquimedes verificou que o volume de água recolhido tinha um valor intermediário entre aqueles recolhidos no caso do ouro e da prata. \alert{Portanto, a coroa não era de ouro puro!}
	\end{corollary}
\end{frame}

\begin{frame}{Transformar um número em notação científica}
	\begin{corollary}
		\begin{enumerate}[leftmargin=1.75cm]
			\item[Passo 1:] Escrever o número incluindo a vírgula.
			\item[Passo 2:] Andar com a vírgula até que somente reste um número diferente de zero no lado esquerdo.
			\item[Passo 3:] Colocar no expoente da potência de 10 o número de casas decimais que tivemos que "andar" com a vírgula. Se ao andar com a vírgula o valor do número diminuiu, o expoente ficará positivo, se aumentou o expoente ficará negativo.
		\end{enumerate}
	\end{corollary}
	\begin{exampleblock}{Exemplo}
		\begin{equation*}
			6\; 590\; 000\; 000\; 000\; 000,0 = \SI{6,59e15}{}		
		\end{equation*}
	\end{exampleblock}
\end{frame}

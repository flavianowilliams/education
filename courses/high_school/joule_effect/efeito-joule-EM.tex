\section{Potência elétrica}

\subsection{}

\begin{frame}{Transformação de energia elétrica em outro tipo de energia}
%	\begin{multicols}{2}
		A definição de potência é a taxa da variação de trabalho $\tau$ realizado por determinado aparelho a cada intervalo de tempo,
		\begin{align*}
			P = \frac{\Delta \tau}{\Delta t}.
		\end{align*}
		Sabemos que um aparelho elétrico realiza trabalho $\tau_{AB}$ de modo a deslocar a quantidade de carga $\Delta q$ do terminal A para o terminal B, ou seja,
		\begin{align*}
			\tau_{AB} = \varepsilon_A-\varepsilon_B,
		\end{align*}
		onde $\varepsilon_{AB}$ representa a energia potencial nos terminais A e B. Pela definição de diferença de potencial nesses terminais, temos de maneira equivalente
		\begin{align*}
			\tau_{AB} = \Delta q\cdot V_{AB}.
		\end{align*}		
%	\end{multicols}
\end{frame}

\begin{frame}{Transformação de energia elétrica em outro tipo de energia}
	\begin{multicols}{2}
		Como trabalho está associado com a variação da energia elétrica $\Delta E$, temos
		\begin{align*}
			\Delta E = \Delta q\cdot V_{AB}.
		\end{align*}
		Mas $P=\frac{\Delta E}{\Delta t}$, portanto
		\begin{align*}
			P = \frac{\Delta E}{\Delta t} = \frac{\Delta q}{\Delta t}\cdot V_{AB}.
		\end{align*}
		Pela definição de corrente temos
		\begin{align*}
			i = \frac{\Delta q}{\Delta t}.
		\end{align*}
		Substituímos na equação acima e obtemos a expressão da potência no circuito.
	\end{multicols}
	\begin{block}{Potência elétrica de um circuito}
		\begin{equation*}
			P = i\cdot V_{AB}.
		\end{equation*}
	\end{block}
	A unidade de medida de potência no SI é Watt (W).
\end{frame}

\section{Efeito Joule}

\subsection{}

\begin{frame}{Transformação de energia elétrica em energia térmica}
	\begin{multicols}{2}
		A potência desenvolvida em um aparelho, pela passagem de uma corrente elétrica i entre os terminais A e B, é dada por $P=iV_{AB}$. Se entre esses terminais estiver um resistor ôhmico de resistência R, onde vale a Lei de Ohm ($V=Ri$), temos
		\begin{align*}
			P & = i\cdot V_{AB},\\
			P & = i\cdot (Ri),\\
			\Aboxed{P & = Ri^2.}
		\end{align*}
	\end{multicols}
	\begin{block}{Efeito Joule}
		O efeito Joule consiste na transformação de \alert{energia elétrica em energia térmica} em um resistor percorrido por uma corrente elétrica i, segundo a relação
		\begin{equation*}
			P = Ri^2.
		\end{equation*}
	\end{block}
\end{frame}

\section{Instrumentos de medida}

\subsection{}

\begin{frame}{Medida de corrente elétrica}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{itemize}
				\item O amperímetro é o instrumento usado para medir corrente elétrica;
				\item Para medir a corrente elétrica que atravessa um fio condutor devemos ligar o amperímetro em série com a resistência do circuito, como mostra a figura ao lado;
				\item A resistência interna de um amperímetro deve ser menor quanto possível, para seu valor não seja acrescentada na resistência do circuito.
			\end{itemize}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.4]{figuras/figura-2.png}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Medida da diferença de potencial}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{itemize}
				\item O voltímetro é o instrumento usado para medir a ddp entre dois terminais de um circuito;
				\item Para medir a ddp devemos ligar o voltímetro em paralelo com a resistência do circuito, como mostra a figura ao lado;
				\item A resistência interna de um voltímetro deve ser maior quanto possível, para que parte da corrente não seja desviada para o aparelho.
			\end{itemize}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.4]{figuras/figura-3.png}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Medida da resistência}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{itemize}
				\item O ohmímetro é o instrumento usado para medir a resistência elétrica de um resistor;
				\item Podemos determinar a resistência R combinando as leituras de um voltímetro e de um amperímetro. Os valores lidos substituímos na Lei de Ohm, $R=\frac{V}{i}$.
			\end{itemize}
		\end{column}
		\begin{column}{0.4\textwidth}

	\end{columns}
\end{frame}

\begin{frame}{Transformar um número em notação científica}
	\begin{corollary}
		\begin{enumerate}[leftmargin=1.75cm]
			\item[Passo 1:] Escrever o número incluindo a vírgula.
			\item[Passo 2:] Andar com a vírgula até que reste somente um número diferente de zero no lado esquerdo.
			\item[Passo 3:] Colocar no expoente da potência de 10 o número de casas decimais que tivemos que "andar" com a vírgula. Se ao andar com a vírgula o valor do número diminuiu, o expoente ficará positivo, se aumentou o expoente ficará negativo.
		\end{enumerate}
	\end{corollary}
	\begin{exampleblock}{Exemplo}
		\begin{equation*}
			6\; 590\; 000\; 000\; 000\; 000,0 = \SI{6,59e15}{}		
		\end{equation*}
	\end{exampleblock}
\end{frame}

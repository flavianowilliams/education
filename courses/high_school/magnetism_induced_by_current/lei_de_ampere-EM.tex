\begin{frame}{Lei de Ampère}
	\begin{columns}
		\begin{column}{0.7\textwidth}
			De acordo com experiências, a corrente que percorre um fio condutor produz um campo magnético circular $\vec{B}$, onde \alert{$B\;\alpha\; i$} e \alert{$B\;\alpha\; \frac{1}{r}$}. Podemos reunir em uma única relação de proporcionalidade,
			\begin{equation*}
				B\; \alpha\; \frac{i}{r}\Rightarrow B = \left(\frac{\mu_0}{2\pi}\right)\frac{i}{r}.
			\end{equation*}
			$\mu_0$ é chamado constante de permeabilidade magnética.
		\end{column}
		\begin{column}{0.3\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.18]{figuras/figura-2.png}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{block}{Lei de Ampère}
		\begin{equation*}
			B = \frac{\mu_0 i}{2\pi r}.
		\end{equation*}
	\end{block}
\end{frame}

\begin{frame}{Força magnética entre dois fios}
	A corrente $i_1$ produzirá um campo magnético $B_1$ no fio 1. A força magnética atuando no fio 2 é $F_m=BiL$. Pela Lei de Àmpere temos $B=\frac{\mu_0i}{2\pi r}$, portanto
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{align*}
				F_m & = BiL,\\
				F_m & = \left(\frac{\mu_0i}{2\pi r}\right)iL.
			\end{align*}
			Se r=1m, i=1A, e L=1m a força encontrada vale $F_m=2\times 10^{-7}\;N$, substituindo
			\begin{equation*}
				\boxed{\mu_0 = 4\pi\times 10^{-7}\;\frac{N}{A^2}.}
			\end{equation*}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.8]{figuras/forca-magnetica.jpg}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\section{Campo magnético numa espira circular}

\subsection{}

\begin{frame}{O que é uma espira circular?}
	\begin{columns}
		\begin{column}{0.7\textwidth}
			Sabemos que a corrente num fio condutor produz linhas de campo magnético circulares, se dobrarmos esse fio afim de possuir um formato circular de raio \alert{r}, as linhas de campo de cada pedaço do fio irão se concentrar no centro do círculo (\alert{chamamos esse círculo de espira circular}). Sabendo que $B\;\alpha\; i$ e $B\;\alpha\; \frac{1}{r}$, podemos dizer que o campo B no centro da espira é proporcional a corrente i e o raio r da espira.
		\end{column}
		\begin{column}{0.3\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.4]{figuras/figura-3.png}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{block}{Campo magnético no centro da espira circular}
		\begin{equation*}
			B\;\alpha\;\frac{i}{r}.
		\end{equation*}
	\end{block}
\end{frame}

\section{Campo magnético num solenóide}

\subsection{}

\begin{frame}{O que é um solenóide?}
	\begin{columns}
		\begin{column}{0.7\textwidth}
			\textit{Um solenóide é um fio enrolado de modo a formar N espiras sucessivas, todas alinhadas em um cilindro de comprimento L}. Sabendo que cada espira produz um campo magnético $\vec{B}$ no seu centro, onde \alert{$B\;\alpha\; i$} e considerando que cada espira possui o mesmo raio, para saber o campo magnético numa posição no interior do solenóide, multiplicamos o campo $\vec{B}$ de cada espira pelo número de espiras por unidade de comprimento do solenóide, \alert{n}.
		\end{column}
		\begin{column}{0.3\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.27]{figuras/figura-4.png}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{block}{Campo magnético no interior do solenóide}
		\begin{equation*}
			B\;\alpha\;ni.
		\end{equation*}
	\end{block}
\end{frame}

\section{Materiais magnéticos}

\subsection{}

\begin{frame}{Tipos de materiais magnéticos}
	Existem três classes de materiais magnéticos:
	\begin{itemize}
		\item Paramagnéticos:\textit{São aqueles que magnetizam fracamente aumentando ligeiramente o campo magnético no local.}
		\item Ferromagnéticos:\textit{São aqueles que magnetizam com muita facilidade e aumentam substancialmente o campo magnético no local.}
		\item Diamagnéticos:\textit{São aqueles que magnetizam fracamente diminuindo ligeiramente o campo magnético no local.}
	\end{itemize}
	\begin{corollary}
		A magnetização do material diminui com a temperatura, pois a elevação da temperatura provoca um aumento da agitação térmica dos átomos, dificultando, então, a sua magnetização.
	\end{corollary}
\end{frame}

\begin{frame}{Materiais paramagnéticos}
	Na presença de um campo magnético, os ímãs elementares do material tendem a se orientarem no \alert{mesmo sentido} do campo magnético. Assim, o campo magnético é somado com o campo dos \alert{ímãs elementares}, fazendo com que ele fique \alert{fracamente aumentado}.
	\begin{figure}[b]
		\begin{subfigure}{0.45\textwidth}
			\Centering
			\includegraphics[scale=0.5]{figuras/figura-5a.png}
			\caption*{\justifying Representação dos ímãs elementares de um material paramagnético.}
		\end{subfigure}
		\begin{subfigure}{0.45\textwidth}
			\Centering
			\includegraphics[scale=0.5]{figuras/figura-5b.png}
			\caption*{\justifying Orientação dos ímãs elementares na presença de um campo magnético $\vec{V}$.}
		\end{subfigure}
	\end{figure}
\end{frame}

\begin{frame}{Materiais ferromagnéticos}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			Assim como no caso dos materiais paramagnéticos, nos materiais ferromagnéticos, os ímãs elementares se orientam no \alert{mesmo sentido} do campo do magnético externo, porém muito mais significativamente. Isso \alert{aumenta consideravelmente} o campo magnético no seu interior.
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.3]{figuras/figura-6.png}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		Existem poucos materiais na natureza que possuem propriedades ferromagnéticas, como o ferro, cobalto e níquel.
	\end{corollary}
\end{frame}

\begin{frame}{Materiais diamagnéticos}
	\begin{columns}
		\begin{column}{0.7\textwidth}
			Nos materiais diamagnéticos, ao contrário dos materias paramagnéticos e ferromagnéticos, os ímãs elementares do material se orientam no sentido contrário ao campo magnético externo, fazendo com que o material se comporte como ímãn com os pólos contrários.
		\end{column}
		\begin{column}{0.3\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.3]{figuras/figura-7.png}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		Geralmente os materiais diamagnéticos sofrem uma força de repulsão na presença de um campo magnético externo.
	\end{corollary}
\end{frame}

\begin{frame}{Transformar um número em notação científica}
	\begin{corollary}
		\begin{enumerate}[leftmargin=1.75cm]
			\item[Passo 1:] Escrever o número incluindo a vírgula.
			\item[Passo 2:] Andar com a vírgula até que reste somente um número diferente de zero no lado esquerdo.
			\item[Passo 3:] Colocar no expoente da potência de 10 o número de casas decimais que tivemos que "andar" com a vírgula. Se ao andar com a vírgula o valor do número diminuiu, o expoente ficará positivo, se aumentou o expoente ficará negativo.
		\end{enumerate}
	\end{corollary}
	\begin{exampleblock}{Exemplo}
		\begin{equation*}
			6\; 590\; 000\; 000\; 000\; 000,0 = \SI{6,59e15}{}		
		\end{equation*}
	\end{exampleblock}
\end{frame}

\section{Transformações cíclicas}

\begin{frame}[fragile]{Diagrama pressão versus volume em processos cíclicos}
	\begin{corollary}
		Definimos como processo cíclico quando o gás retorna para o seu estado inicial.
	\end{corollary}
	\begin{columns}
		\begin{column}{0.7\textwidth}
			A energia interna U(T) de um gás é uma função da temperatura. Se os estados i e f estão em equilíbrio térmico, podemos determinar U(T) sabendo a temperatura nesses estados, os valores \alert{não irão mudar independente do processo termodinâmico} que esse gás poderá sofrer.
		\begin{corollary}
			Durante uma transformação cíclica, a variação da energia interna do gás será zero (\alert{$\Delta U=0$}).
		\end{corollary}

\begin{frame}[fragile]{Trabalho realizado pelo gás em processos cíclicos}
	\begin{columns}
		\begin{column}{0.7\textwidth}
			Podemos definir o trabalho total realizado pelo gás num processo cíclico subtraindo os trabalhos individuais nos processos de i para f e o retorno (f para i),
			\begin{equation*}
				\tau = \tau_{(i\rightarrow f)}-\tau_{(f\rightarrow i)}.
			\end{equation*}
		\begin{corollary}
			Durante uma transformação cíclica, o trabalho realizado pelo gás, ao percorrer o ciclo, é fornecido pela \alert{área entre as curvas}.
		\end{corollary}

\section{Máquinas térmicas}

\subsection{}

\begin{frame}{Definição de máquina térmica}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{block}{Máquina térmica}
				Toda máquina térmica opera num processo cíclico, onde ela recebe calor $\alert{Q_1}$ de uma fonte quente a temperatura $\alert{T_1}$ e parte desse calor ($\alert{Q_2}$) ela devolve para uma fonte fria que está a temperatura $\alert{T_2}$, onde $\alert{T_2<T_1}$.
			\end{block}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.15]{figuras/figura-1.png}
				\caption*{\href{https://www.youtube.com/watch?v=V9GjzQQCDa0}{\textcolor{blue}{Motor a combustão de 4 tempos}}}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		Parte do calor recebido pela fonte quente é convertido em trabalho e o restante é desperdiçado para a fonte fria.
	\end{corollary}
\end{frame}

\begin{frame}[fragile]{Aplicação da Primeira Lei da Termodinâmica em máquinas térmicas}
	\begin{columns}
		\begin{column}{0.65\textwidth}
			O calor total absorvido pelo gás durante o processo será o calor $Q_1$ recebido menos o calor $Q_2$ desperdiçado. Pela Primeira Lei da Termodinâmica temos
			\begin{equation*}
				\Delta U = (Q_1-Q_2)-\tau.
			\end{equation*}
			Mas $\Delta U=0$ num processo cíclico, portanto
			\begin{equation*}
				\boxed{\tau = Q_1-Q_2.}
			\end{equation*}
		\end{column}

\begin{frame}[fragile]{Rendimento de uma máquina térmica}
	\begin{columns}
		\begin{column}{0.7\textwidth}
			Definimos o rendimento R de uma máquina térmica pela \alert{quantidade de calor que ela consegue transformar em trabalho} à partir do calor que recebe da fonte quente,
			\begin{align*}
				R & = \frac{\tau}{Q_1},\\
				R & = \frac{Q_1-Q_2}{Q_1},\\
				\Aboxed{R & = 1-\frac{Q_2}{Q_1}\Rightarrow R<1.}
			\end{align*}
		\end{column}

\begin{frame}[fragile]{Rendimento de uma máquina de Carnot}

			\begin{corollary}
				Nenhuma máquina térmica que opere entre duas fontes às temperaturas $T_1$ e $T_2$, pode ter rendimento maior que uma máquina de Carnot operando entre essas mesmas fontes.

			\begin{block}{Rendimento de uma máquina de Carnot}
				\begin{equation*}
					R=1-\frac{T_2}{T_1}.

\begin{frame}[fragile]{Como funciona um refrigerador?}
\begin{columns}
	\begin{column}{0.65\textwidth}
		O objetivo de um refrigerador é remover calor de uma fonte fria e transferí-la para uma fonte mais quente. Nesse processo inevitavelmente ele realiza trabalho. Portanto, um refrigerador seria uma máquina térmica funcionando no sentido contrário. O rendimento é medido pela capacidade de transferir calor $Q_2$ em relação ao trabalho $\tau$ que ele realiza,
		\begin{NodesList}[margin=3cm]
			\begin{align*}
				R & = \frac{Q_2}{\tau},\AddNode\\
				R & = \frac{Q_2}{Q_1-Q_2}.\AddNode
			\end{align*}
			\LinkNodes{\tiny$(\tau=Q_1-Q_2)$}
		\end{NodesList}
	\end{column}
	\begin{column}{0.35\textwidth}
		\begin{figure}
			\centering
			\vspace*{-0.25cm}
			\begin{tikzpicture}[scale=0.45, transform shape]
			
			\useasboundingbox (5,9) rectangle (-2,-1);
			
			%parametros
			\def\p{8}

\section{Entropia}

\subsection{}

\begin{frame}{O que é entropia?}
	\begin{block}{Entropia}
		A entropia, definida pela letra S, está associada com o grau de desordem de um sistema. No SI a unidade de medida da entropia é Joule por Kelvin (J/K).
	\end{block}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Por exemplo, se o processo ocorre sem variar a sua temperatura (isotérmico) a variação da entropia $\Delta S$ associado ao sistema será
			\begin{align*}
				\Delta S = \frac{Q}{T},
			\end{align*}
			onde Q é a quantidade de calor que o sistema irá receber ou ceder e T a sua temperatura.
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.1]{figuras/entropia.png}
				\caption*{\footnotesize Exemplo de entropia usando tijolos.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Variação da entropia do universo}
	\begin{block}{Segunda lei da termodinâmica}
		Na natureza, a entropia total, \alert{que é a soma da entropia do sistema com a vizinhança}, sempre aumenta.
	\end{block}
	Definindo $\Delta S_u$ como a variação da entropia do universo, $\Delta S_s$ a variação de entropia do sistema e $\Delta S_v$ a variação da entropia da vizinhança, onde
	\begin{align*}
		\Delta S_u & = \Delta S_s + \Delta S_v,
	\end{align*}
	podemos dizer que para qualquer fenômeno que ocorre na natureza $\Delta S_u$ será sempre maior ou no mínimo igual a zero ($\Delta S_u\ge 0$).
\end{frame}

\section{Aplicações}

\subsection{}

\begin{frame}{Entropia e máquinas térmicas}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
%	\begin{multicols}{2}
		No ciclo de Carnot temos que a variação da entropia $\Delta S$ nos processos adiabáticos é zero, pois Q=0. Sabendo que nos processos isotérmicos temos $\Delta S=\frac{Q}{T}$ e que $\Delta S_{\text{Total}}\ge 0$. Se $\Delta S_{\text{sistema}}=0$ temos
		\begin{align*}
			\Delta S_{\text{sistema}} = \Delta S_1 + \Delta S_2 = 0,
		\end{align*}
	\end{column}
		\begin{column}[c]{0.5\textwidth}
		\begin{align*}
			\frac{Q_1}{T_1}\alert{-}\frac{Q_2}{T_2} = 0,\\
			\frac{Q_1}{T_1} = \frac{Q_2}{T_2}.
		\end{align*}
		Sabemos que $T_1\ne 0$ e $T_2\ne 0$, portanto a única maneira de termos $Q_2=0$ é se $Q_1=0$ (\alert{uma máquina que não existe!}).
	\end{column}
%	\end{multicols}
\end{columns}
	\begin{corollary}
		A segunda lei da termodinâmica impede que todo calor $Q_1$ recebido pela máquina térmica seja inteiramente convertido na forma de trabalho.
	\end{corollary}
\end{frame}

\begin{frame}{Transformar um número em notação científica}
	\begin{corollary}
		\begin{enumerate}[leftmargin=1.75cm]
			\item[Passo 1:] Escrever o número incluindo a vírgula.
			\item[Passo 2:] Andar com a vírgula até que reste somente um número diferente de zero no lado esquerdo.
			\item[Passo 3:] Colocar no expoente da potência de 10 o número de casas decimais que tivemos que "andar" com a vírgula. Se ao andar com a vírgula o valor do número diminuiu, o expoente ficará positivo, se aumentou o expoente ficará negativo.
		\end{enumerate}
	\end{corollary}
	\begin{exampleblock}{Exemplo}
		\begin{equation*}
			6\; 590\; 000\; 000\; 000\; 000,0 = \SI{6,59e15}{}

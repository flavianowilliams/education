\section{Estados da matéria}

\subsection{}

\begin{frame}{Estado sólido}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{itemize}
				\item Os átomos estão ligados muito próximos um do outro;
				\item Existe uma ligação eletromagnética relativamente forte entre os átomos;
				\item Os átomos vibram em torno de uma posição média.
				\item Os átomos se organizam em uma estrutura bem definida chamada \alert{rede cristalina}, que se repete ao longo do sólido (\alert{baixa entropia}).
			\end{itemize}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
			 \animategraphics[scale=0.19,controls,loop]{15}{videos/video-3/anim-}{1}{180}

				  \caption*{\footnotesize Rede cristalina em duas dimensões}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Estrutura cristalina e alotropia}
	\begin{corollary}
		A mesma substância pode se apresentar em estruturas cristalinas diferentes.
	\end{corollary}

	\end{block}
\end{frame}

\begin{frame}{Estado líquido}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			\begin{itemize}
				\item Os átomos (ou moléculas) se encontram mais afastados um dos outros que no estado sólido;
				\item O aumento da distância entre os átomos enfraquece a força eletrostática;
				\item Os átomos possuem mais liberdade para se movimentarem.
				\item Os átomos não estão distribuídos ordenadamente (\alert{maior entropia}).
			\end{itemize}
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}
				\animategraphics[scale=0.19,controls,loop]{7}{videos/video-2/anim-}{1}{190}
%				\animategraphics[scale=0.19,controls,loop]{7}{videos/video-2/anim-}{1}{1}
				  \caption*{\footnotesize Água líquida a 298 K e 1 atm.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Estado gasoso}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{itemize}
				\item A separação entre as moléculas é muito maior que no estado líquido;
				\item A força de ligação entre as moléculas é praticamente nula;
				\item As moléculas estão livres para se moverem em todas as direções. Por isso, dizemos que o gás ocupa todo o recipiente que o contém (\alert{entropia elevada}).
			\end{itemize}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\animategraphics[scale=0.19,controls,loop]{15}{videos/video-1/anim-}{1}{180}
%				\animategraphics[scale=0.19,controls,loop]{15}{videos/video-1/anim-}{1}{1}
				  \caption*{\footnotesize Gás de argônio a 298 K e 1000 atm.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Plasma}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			\begin{corollary}
				Um gás superaquecido a temperaturas elevadas pode romper as ligações entre os elétrons e os íons, formando um \alert{gás ionizado}. Portanto, dizemos que o plasma é o estado no qual as partículas que constituem o material se apresentam em forma de íons.
			\end{corollary}
		\end{column}
		\begin{column}{0.5\textwidth}
			\begin{figure}
				\includegraphics[scale=0.4]{figuras/plasma.png}
				  \caption*{\footnotesize Diferença entre gás e plasma \cite{fig:plasma}.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\section{Mudanças de estado}

\subsection{}

\begin{frame}{Mudanças de estado}
	\begin{corollary}
		Quando um objeto recebe calor, o aumento da temperatura causa um aumento na agitação dos átomos, o que modifica a sua organização. Isso faz com que ele \alert{mude o seu estado físico} para qualquer um dos outros quatro estados da matéria.
	\end{corollary}
	\begin{columns}
		\begin{column}[c]{0.3\textwidth}
			\begin{corollary}
				Cada estado da matéria possui características bem definidas que a diferencia das demais.
			\end{corollary}
		\end{column}
		\begin{column}[c]{0.7\textwidth}
	\begin{figure}
	\centering
	\includegraphics[scale=0.4]{figuras/mudanca-fase.png}
	\caption*{\footnotesize Estados da matéria e transições que pode ocorrer com a substância.}
\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Curva de resfriamento}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{itemize}
				\item Durante o resfriamento da água, uma quantidade de calor  $\alert{Q=mc\Delta T}$ é removida, diminuindo a sua temperatura.
				\item Durante a transição de fase, a temperatura permanece constante e uma quantidade de calor $\alert{Q=mL}$ deve ser removida, de modo que a substância possa refazer as ligações químicas e retornar para o seu estado ordenado.
			\end{itemize}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\includegraphics[scale=0.4]{figuras/curva-resfriamento.png}
				  \caption*{\footnotesize Curva de resfriamento da água.}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		Somente válido para substâncias puras.
	\end{corollary}
\end{frame}

\begin{frame}{Diagrama de fases}
	\begin{columns}
		\begin{column}{0.35\textwidth}
			\begin{itemize}
				\item A->B: Fusão.
				\item B->A: Solidificação.
				\item E->D: Sublimação.
				\item D->E: Sublimação.
				\item B->D: Vaporização.
				\item D->B: Condensação.
			\end{itemize}
		\end{column}

\begin{frame}{Diagrama de fases da água}
	\begin{corollary}
		A água é uma das poucas substâncias na natureza cuja densidade no estado sólido é menor que no estado líquido. Além disso, \alert{a temperatura de fusão diminui com o aumento da pressão}, ao contrário do que ocorre com outras substâncias.
	\end{corollary}

\begin{frame}{Algumas considerações sobre o diagrama de fases}
	\begin{columns}
		\begin{column}{0.7\textwidth}
			\begin{itemize}
				\item \alert{Ponto Triplo}: Estado onde os três estados físicos da substância encontram-se simultaneamente em equilíbrio (o ponto triplo da água pura é aproximadamente 0,006 atm e 273 K).
				\item \alert{Temperatura crítica $T_c$}: Temperatura mínima onde o gás não consegue se liquefazer, independente da pressão aplicada sobre ele (chamamos de vapor a substância no estado gasoso abaixo de $T_c$).
				\item \alert{Pressão de vapor}: Pressão mínima aplicada sobre o gás real (diferente de um gás ideal), onde a partir desse valor ele começa a se condensar.
			\end{itemize}
		\end{column}
		\begin{column}{0.3\textwidth}
			\begin{figure}
				\includegraphics[scale=0.25]{figuras/diagrama-agua.png}
				\caption*{\footnotesize Diagrama de fases da água.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\section{Aplicações}

\begin{frame}{Ponto de orvalho}
	\begin{columns}
		\begin{column}{0.65\textwidth}
			Quando a pressão que o ar exerce sobre as moléculas de água atinge a sua pressão de vapor dizemos que a água condensa, ou seja, ela passa do estado gasoso para o líquido, formando as gotas de orvalho.\\
			A razão entre a pressão que o ar exerce sobre as moléculas de água e a pressão de vapor chamamos de umidade relativa do ar, $u_r$,
			\begin{align*}
				u_r= \left(\frac{\text{Pressão parcial do vapor de água}}{\text{Pressão de vapor da água}}\right)\times 100\%
			\end{align*}
			Dizemos que a umidade do ar está saturada quando formam as gotas de orvalho, ou seja, $u_r=100\%$.
		\end{column}
		\begin{column}{0.35\textwidth}
			\begin{footnotesize}

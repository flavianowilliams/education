\begin{frame}{O movimento}
	\begin{itemize}
		\item [\ding{51}] O movimento retilíneo se dá ao longo de uma reta (vertical, horizontal e inclinada);
		\item [\ding{51}] As forças modificam o movimento, tanto em quantidade quanto na sua orientação (Devido a força centrípeta, um carro sofre uma mudança na sua direção quando o mesmo faz uma curva);
		\item[\ding{51}] Se o objeto for uma partícula, todas as partes desse objeto se movem na mesma direção e com a mesma velocidade.
	\end{itemize}
	\begin{corollary}
		A cinemática se preocupa em analisar e classificar o movimento dos objetos, mas sem se preocupar com a causa desse movimento.
	\end{corollary}
\end{frame}

\section{Velocidade}

\subsection{}

\begin{frame}{Velocidade escalar média}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Usa-se a idéia de velocidade para expressar a rapidez do movimento de um objeto. Se quisermos determinar a rapidez de um objeto em percorrer uma certa distância em um intervalo de tempo $\Delta t$, usamos o conceito de velocidade escalar média,
			\begin{equation*}
				\boxed{s_{\text{médio}}=\frac{\text{distância percorrida}}{\Delta t}.}
			\end{equation*}
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}
				\includegraphics[scale=0.23]{figuras/figura-2.jpg}
				\caption*{\footnotesize Trajetórias e distâncias percorridas diferentes para chegar na mesma posição.}
			\end{figure}
		\end{column}
	\end{columns}	
	\begin{corollary}
		A velocidade escalar média é uma grandeza escalar e ela é sempre positiva.
	\end{corollary}
\end{frame}

\begin{frame}{Velocidade média}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Outra maneira de expressar a rapidez de um movimento é através da velocidade média, que é a razão entre o deslocamento $\Delta \vec{x}$ e o intervalo de tempo $\Delta t$,
			\begin{equation*}
			\boxed{\vec{v}_{\text{médio}}=\frac{\Delta \vec{x}}{\Delta t}.}
			\end{equation*}
			A velocidade média é uma grandeza vetorial e portanto depende da orientação.
			\begin{corollary}
				A unidade de medida da velocidade no SI é metro por segundo (m/s).
			\end{corollary}
		\end{column}

\begin{frame}{Velocidade instantânea}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Caso a velocidade varie com o tempo, para determiná-la usamos o conceito de velocidade instantânea. A velocidade instantânea é obtida a partir da velocidade média reduzindo o intervalo de tempo $\Delta t$ a um valor infinitesimal, $\Delta t\rightarrow 0$,
			\begin{equation*}
			\boxed{\vec{v}(t)={\lim\limits_{\Delta t\rightarrow 0}}\frac{\Delta \vec{x}}{\Delta t}=\frac{d\vec{x}}{dt}.}
			\end{equation*}
			\begin{corollary}
				A velocidade instantânea é a taxa de variação da posição em um determinado instante de tempo.
			\end{corollary}
		\end{column}

\begin{frame}{Função horária da posição à partir da velocidade}
%	\begin{multicols}{2}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
		Da definição de velocidade instantânea temos $v(t)=\frac{dx(t)}{dt}$. Integrando os dois lados da equação chegamos a
		\begin{align*}
			\int_{t_0}^{t}v(t)dt = \int_{t_0}^{t}\frac{dx(t)}{dt}dt.
		\end{align*}
		onde podemos usar a substituição infinitesimal $dx=\frac{dx}{dt}dt$,
		\begin{equation*}
			\int_{t_0}^{t}a(t)dt = \int_{t_0}^{t}dx,
		\end{equation*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
		\begin{align*}
			\int_{t_0}^{t}v(t) & = \Delta x,\\
			\int_{t_0}^{t}v(t) & = x(t)-x(t_0).
		\end{align*}
		Se a velocidade for constante e $t_0=0\;\text{s}$ teremos a função horária da posição.
		\begin{block}{Função horária da posição se v = cte}
			\begin{equation*}
				\vec{x}(t)=\vec{x}_0 + \vec{v}t.
			\end{equation*}
		\end{block}
		\end{column}
	\end{columns}
%	\end{multicols}
\end{frame}

\begin{frame}{Função horária da velocidade à partir da aceleração}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Da definição de velocidade instantânea temos $a(t)=\frac{dv(t)}{dt}$. Integrando os dois lados da equação chegamos a
			\begin{equation*}
				\int_{t_0}^{t}a(t)dt = \int_{t_0}^{t}\frac{dv}{dt}dt,
			\end{equation*}
			onde podemos usar a substituição infinitesimal $dv=\frac{dv}{dt}dt$,
			\begin{equation*}
				\int_{t_0}^{t}a(t)dt = \int_{t_0}^{t}dv,
			\end{equation*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			\begin{align*}
				\int_{t_0}^{t}a(t) & = \Delta v,\\
				\int_{t_0}^{t}a(t) & = v(t)-v(t_0).
			\end{align*}
			Se a velocidade for constante e $t_0=0\;\text{s}$ teremos a função horária da posição.
			\begin{block}{Função horária da velocidade se a=cte}
				\begin{equation*}
					\vec{v}(t)=\vec{v}_0 + \vec{a}t.
				\end{equation*}
			\end{block}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Função horária da posição à partir da aceleração}
		\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Se a=cte, definimos a função horária da velocidade como $v(t)=v_0+at$. Integrando novamente teremos
			\begin{equation*}
				\int_{t_0}^{t}v(t)dt = \int_{t_0}^{t}(v_0+at)dt.
			\end{equation*}
			Temos do resultado anterior que $\int_{t_0}^{t}v(t)dt=\Delta x$, portanto
			\begin{equation*}
				\Delta x = \int_{t_0}^{t}(v_0+at)dt.
			\end{equation*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			Integrando o lado direito e considerando que a aceleração é constante teremos
			\begin{equation*}
			\Delta x = v(t_0)t+\frac{1}{2}at^2.
			\end{equation*}
			Considerando $t_0=0\;\text{s}$ teremos a função horária da posição.
			\begin{block}{Função horária da posição se a=cte}
				\begin{equation*}
				\vec{x}(t)=\vec{x}_0+\vec{v}(t_0)t+\frac{1}{2}\vec{a}t^2.
				\end{equation*}
			\end{block}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Equação de Torricelli}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			Considerando as funções horárias da posição $x(t)$ e da velocidade $v(t)$ podemos combiná-las de modo a eliminar a variável tempo. Primeiramente isolamos t em $v(t)$,
			\begin{align*}
				v & = v_0+at,\\
				t & = \frac{v-v_0}{a}.
			\end{align*}
			Substituindo em $x(t)$ temos
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{align*}
				x & = x_0+v_0\left(\frac{v-v_0}{a}\right)+\frac{1}{2}a\left(\frac{v-v_0}{a}\right)^2,\\
				x & = x_0+\frac{v_0\left(v-v_0\right)}{a}+\frac{\bcancel{a}\left(v-v_0\right)^2}{2\bcancel{a^2}},\\
				x & = x_0+\bcancel{\frac{v_0v}{a}}-\frac{v_0^2}{a}+\frac{v^2}{2a}-\bcancel{\frac{v_0v}{a}}+\frac{v_0^2}{2a}.
			\end{align*}
			Somando os termos remanescentes, temos como opção
			\begin{equation*}
				\boxed{v^2=v^2_0+2a\Delta x.}
			\end{equation*}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Forma alternativa da função horária da posição}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			Podemos também eliminar $v_0$ em $v(t)$,
			\begin{align*}
%				v & = v_0+at,\\
				v_0 = v-at.
			\end{align*}
			Substituindo em $x(t)$ temos
			\begin{equation*}
				x = x_0+vt-at^2+\frac{1}{2}at^2.
			\end{equation*}
			Resultando em
			\begin{equation*}
				\boxed{x(t) = x_0+vt-\frac{1}{2}at^2.}
			\end{equation*}
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{table}
				\centering
				\caption*{\footnotesize Equações com aceleração constante.}
				\begin{tabular}[c]{cc}
					\hline
					Equação & Variável que falta\\
					\hline
					$v=v_0+at$ & $\Delta x$\\
					$\Delta x=v_0t+\frac{1}{2}at^2$ & $v$\\
					$v^2=v_0^2+2a\Delta x$ & $t$\\
					$\Delta x=\frac{1}{2}\left(v_0+v\right)t$ & $a$\\
					$\Delta x=vt-\frac{1}{2}at^2$ & $v_0$\\
					\hline					
				\end{tabular}
			\end{table}				
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Transformar um número em notação científica}
	\begin{corollary}
		\begin{enumerate}[leftmargin=1.75cm]
			\item[Passo 1:] Escrever o número incluindo a vírgula.
			\item[Passo 2:] Andar com a vírgula até que reste somente um número diferente de zero no lado esquerdo.
			\item[Passo 3:] Colocar no expoente da potência de 10 o número de casas decimais que tivemos que "andar" com a vírgula. Se ao andar com a vírgula o valor do número diminuiu, o expoente ficará positivo, se aumentou o expoente ficará negativo.
		\end{enumerate}
	\end{corollary}
	\begin{exampleblock}{Exemplo}
		\begin{equation*}
			6\; 590\; 000\; 000\; 000\; 000,0 = \SI{6,59e15}{}		
		\end{equation*}
	\end{exampleblock}
\end{frame}

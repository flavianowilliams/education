\begin{frame}{Sistema termodinâmico}
	\begin{enumerate}
		\item[\ding{51}] Torna-se inviável um tratamento mecânico para um sistema contendo N partículas devido ao elevado grau de liberdade (3N graus de liberdade);
		\item[\ding{51}] Para resolver o sistema é necessário um tratamento  a nível macroscópico através das variáveis pressão, temperatura e volume \alert{(variáveis termodinâmicas)}. O estado do sistema onde as variáveis termodinâmica são conhecidas é chamado \alert{estado termodinâmico};
		\item[\ding{51}] È necessário um descrição estatística para resolver o sistema termodinâmico, onde aplica-se o conceito de limite termodinâmico \alert{(um grande número de elementos, porém N/V é finito)}.
	\end{enumerate}
\end{frame}

\begin{frame}{Temperatura}
	A temperatura é uma medida da média da energia térmica das partículas de uma substância. Uma vez que se trata de um valor médio, não depende do número de partículas do sistema. Nesse sentido, também não depende das dimensões do sistema.
	\begin{corollary}
		A temperatura é considerada uma grandeza intensiva, pois é praticamente a mesma em cada ponto do sistema.
	\end{corollary}
\end{frame}

\begin{frame}{Equilíbrio termodinâmico}
	\begin{enumerate}
		\item[\ding{51}] Um sistema está em equilíbrio termodinâmico quando suas variáveis de estado se mantém inalteradas ao longo do tempo e uniformes através de todo o sistema;
		\item[\ding{51}] no equilíbrio termodinâmico não são observadas mudanças macroscópicas nas variáveis de estado dos sistemas;
		\item[\ding{51}] Equilíbrio térmico é o caso particular do equilíbrio termodinâmico onde a variável a ser analisada é a temperatura.
	\end{enumerate}
	\begin{corollary}
		O estado termodinâmico do sistema é definido somente quando o equilíbrio termodinâmico é atingido.
	\end{corollary}
\end{frame}

\begin{frame}{Escalas termométricas}
	\begin{itemize}
		\item Assim como usamos uma régua para medir distâncias, para medir temperaturas é necessário graduar o termômetro.
		\item O termômetro mais utilizado é constituído por um tubo fino preenchido por mercúrio, que possui elevado coeficiente de dilatação.
	\end{itemize}
	\begin{block}{Principais escalas de temperatura}
		\begin{itemize}
			\item Celsius ou Centígrada: Baseia-se nos pontos de fusão e ebulição da água pura;
			\item Fahrenheit: Os pontos de fusão e ebulição da água pura estão marcados em $32\;^oF$ e $212\;^oF$;
			\item Kelvin: Baseia-se no zero absoluto (estado de mínima energia possível);
		\end{itemize}
	\end{block}
\end{frame}

\begin{frame}[fragile]{Por que um sólido dilata?}

			\item Os átomos que constituem o sólido distribuem-se ordenadamente, dando origem a uma estrutura denominada rede cristalina do sólido.
			\item A força entre os átomos são como molas que comprimem e esticam.
			\item Quando a temperatura aumenta, há um aumento na agitação dos átomos, que ao vibrarem se afastam mais uns dos outros. Com isso, a distância média entre os átomos torna-se maior.
		\end{itemize}

\begin{frame}{Coeficiente de dilatação linear}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\justifying
			Sabendo que $L=L_0+\Delta L$, temos
			\begin{align*}
				L & = L_0+\Delta L,\\
				L & = L_0+\alpha L_0\Delta T,\\
				L & = L_0(1+\alpha\Delta T),
			\end{align*}
			onde $\alpha$ é a constante de proporcionalidade chamada coeficiente de dilatação linear.
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{table}
				\centering
				\caption*{Coeficiente de dilatação linear.}
				\begin{tabular}{cc}
					\hline
					Substância & $\alpha\;(^{\circ}C^{-1})$\\
					\hline
					\hline
					Alumínio & \SI{23e-6}{}\\
					Cobre & \SI{17e-6}{}\\
					Vidro refratário & \SI{3,2e-6}{}\\
					Diamante & \SI{0,9e-6}{}\\
					\hline
				\end{tabular}
				%\includegraphics[scale=0.3]{figuras/figura-3.png}
			\end{table}
		\end{column}
	\end{columns}
	\begin{corollary}
		A unidade de medida do coeficiente de dilatação linear é $\frac{1}{^oC}=^oC^{-1}$ ou $\frac{1}{K}=K^{-1}$.
	\end{corollary}
\end{frame}

\begin{frame}{Relação entre coeficiente dilatação linear e superficial}
%	\begin{multicols}{2}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Sabemos que $A=xy$, onde $\frac{\Delta A}{A_0} = \frac{\Delta (xy)}{(x_0y_0)}.$ Aplicando a regra da diferencial $\Delta (xy)=x\Delta y+y\Delta x$ teremos
			\begin{align*}
				\frac{\Delta A}{A_0} = x\frac{\Delta y}{x_0y_0}+y\frac{\Delta x}{x_0y_0}.
			\end{align*}
			Porém, da dilatação linear temos $\Delta l=\alpha l_0\Delta T$, portanto podemos dizer que
			\begin{align*}
				\Delta x & = \alpha x_0\Delta T,\\
				\Delta y & = \alpha y_0\Delta T.
			\end{align*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			Substituindo
			\begin{align*}
				\frac{\Delta A}{A_0}  & = x\frac{\alpha \bcancel{y_0}\Delta T}{x_0\bcancel{y_0}}+y\frac{\alpha \bcancel{x_0}\Delta T}{\bcancel{x_0}y_0},\\
				\frac{\Delta A}{A_0}  & = \frac{\alpha x\Delta T}{x_0}+\frac{\alpha y\Delta T}{y_0}
			\end{align*}
			onde $\frac{x}{x_0}=\frac{y}{y_0}=1+\alpha\Delta T$, portanto
			\begin{align*}
				\frac{\Delta A}{A_0} = (1+\alpha\Delta T)\alpha\Delta T+(1+\alpha\Delta T)\alpha\Delta T.
			\end{align*}
		\end{column}
	\end{columns}
%	\end{multicols}
\end{frame}

\begin{frame}{Relações entre coeficientes dilatação linear, superficial e volumétrica}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Expandindo os termos entre parêntesis,
			\begin{align*}
				\frac{\Delta A}{A_0} & = \alpha\Delta T+\alpha^2\Delta T^2+\alpha\Delta T+\alpha^2\Delta T^2,\\
				\frac{\Delta A}{A_0} & = 2\alpha\Delta T+2\alpha^2\Delta T^2.
			\end{align*}
			Considerando que $\Delta T\ll 1$, ou seja, uma variação suficientemente pequena da temperatura, podemos dizer que
			\begin{align*}
				\frac{\Delta A}{A_0} & \approx 2\alpha\Delta T.
			\end{align*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			Sabendo que $\Delta A = \beta A_0\Delta T$ temos
			\begin{align*}
				\beta = 2\alpha.
			\end{align*}		
			A mesma análise podemos fazer para a dilatação volumétrica, onde encontraremos $\gamma = 3\alpha$. $\beta$ e $\gamma$ são chamados \alert{coeficientes de dilatação superficial e volumétrica}, respectivamente. Resumindo
			\begin{align*}
				\Aboxed{\gamma = \frac{3}{2}\beta = 3\alpha.}
			\end{align*}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{O que é calor?}
	\begin{columns}

\begin{frame}{Transferência de calor}
	A transferência de calor entre dois ou mais objetos pode ocorrer de três maneiras distintas:
	\begin{itemize}
		\item Condução: Ocorre praticamente em sólidos.
		\item Convecção: Ocorre em líquidos e gases.
		\item Radiação: Ocorre em qualquer material, inclusive no vácuo.
	\end{itemize}
	\begin{corollary}
		Radiação é o único processo de transferência de calor que ocorre no vácuo.
	\end{corollary}	
\end{frame}

\begin{frame}{Quantidade de calor}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			Foi percebido que a quantidade de calor cedido ou absorvido pelo sistema é proporcional a variação de temperatura,
			\begin{align*}
				Q = C\Delta T,
			\end{align*}
			onde $C$ é chamado capacidade térmica. Porém, C depende da massa e das propriedades intrínsecas do material, assim
			\begin{align*}
				\Aboxed{Q = mc\Delta T.}
			\end{align*}
			$c$ é chamado calor específico do material.

\begin{frame}{Trabalho}
	\begin{columns}[c]
		\begin{column}[c]{0.65\textwidth}
			Considere um cilindro de área A, sabemos pela definição de pressão que a força $\vec{F}$ que o gás aplica sobre o pistão é dado por $F=pA$, onde $p$ é a pressão sobre a área A. Se o gás expandir o pistão por um deslocamento $dx$, temos pela definição de trabalho
			\begin{align*}
				dW & = \vec{F}\cdot d\vec{x},\\
				dW & = F_{\text{médio}} dx,\\
				dW & = pA dx.
			\end{align*}

\begin{frame}{Trabalho (continuação)}
	\begin{columns}[c]
		\begin{column}[c]{0.65\textwidth}
			Mas sabemos que o volume deslocado pelo gás é dado por $dV=Adx$, portanto
			\begin{align*}
				dW & = pdV.

\begin{frame}{Convenção de sinais de calor e trabalho na termodinâmica}
	\begin{block}{Convenção de sinais do calor}
		\begin{itemize}
			\item Se o sistema recebe calor da vizinhança então Q é positivo.
			\item Se o sistema cede calor para a vizinhança então Q é negativo.
		\end{itemize}	
	\end{block}
	\begin{block}{Convenção de sinais do trabalho}
		\begin{itemize}
			\item Se o trabalho está sendo realizado sobre o sistema então $\tau$ é negativo.
			\item Se o sistema realiza trabalho sobre a vizinhança então $\tau$ é positivo.
		\end{itemize}	
	\end{block}
\end{frame}

\begin{frame}{Primeira lei da termodinâmica}
	Quando um sistema vai de um estado i para o estado f e troca energia com a vizinhança, a sua energia interna aumenta ou diminui e a sua variação é dado por
	\begin{equation*}
		\Delta U = U_f-U_i=Q.
	\end{equation*}
	Se ele ao mesmo tempo realizar um trabalho infinitesimal $dW$, ou trabalho for feito sobre ele, a quantidade de energia interna que ele recebe ou cede é dado por
	\begin{equation*}
		\boxed{dU = dQ-dW.}
	\end{equation*}
	\begin{corollary}
		A energia interna de um sistema tende a aumentar, se acrescentarmos energia na forma de calor Q, e a diminuir se removermos energia na forma de trabalho W.
	\end{corollary}
\end{frame}

\begin{frame}{Processos termodinâmicos especiais}
	\begin{enumerate}
		\item[\ding{51}] Processos adiabáticos: Aquele que acontece tão depressa ou em um sistema tão bem isolado que não há troca de calor entre o sistema e a vizinhança, portanto $dQ=0$. Substituindo na primeira lei teremos
		\begin{align*}
			dU = -dW.
		\end{align*}
		\item[\ding{51}] Processos isocóricos: Se o volume de um sistema é mantido constante, ele não realiza trabalho, portanto $dW=0$. Substituindo na primeira lei teremos
		\begin{align*}
			dU = dQ.
		\end{align*}
	\end{enumerate}
\end{frame}

\begin{frame}{Processos termodinâmicos especiais (continuação)}
	\begin{enumerate}
		\item[\ding{51}] Processos isotérmicos: A energia interna é uma grandeza que depende da temperatura, se a temperatura é mantida constante, a energia do sistema não muda, portanto $dU=0$. Substituindo na primeira lei teremos
		\begin{align*}
			dQ = dW.
		\end{align*}
		\item[\ding{51}] Expansão livre: São processos nos quais não há troca de calor com a vizinhança, onde $dQ=0$. Ao mesmo tempo nenhum trabalho é realizado sobre ele ($dW=0$).  Substituindo na primeira lei teremos
		\begin{align*}
			dU = 0.
		\end{align*}
	\end{enumerate}
\end{frame}

			\begin{itemize}
				\item k: constante elástica (depende das propriedades do material);
				\item Se $x_{0}=0$, pela Lei de Hooke $\vec{F}=-k\vec{x}$.
			\end{itemize}
			\begin{align*}
				\vec{F} & = m\vec{a},\\
				\vec{a} & = -\frac{k}{m}\vec{x}.
			\end{align*}
			\begin{corollary}
				A aceleração do objeto e a força restauradora possuem sentidos contrários ao deslocamento.
			\end{corollary}

\begin{frame}{Solução do MHS a partir da segunda lei de Newton}
%	\begin{multicols}{2}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			Aplicando a segunda lei de Newton para o sistema massa-mola onde $x_0=0$,
			\begin{align*}
				F & = -kx,\\
				m\frac{d^2x}{dt^2} & = -kx,
			\end{align*}
			Uma possível solução para a equação acima, usando a analogia com o MCU seria uma função do tipo $x(t)=x_m\cos(\omega t)$,
			\begin{align*}
				\frac{dx}{dt} & = -\omega\sin(\omega t),
			\end{align*}
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{align*}
				\frac{d^2x}{dt^2} = -\omega^2 x_m\cos(\omega t),
			\end{align*}
			Substituindo teremos
			\begin{align*}
				-m\omega^2\bcancel{x_m}\bcancel{\cos(\omega t)} & =-\bcancel{x_m}k\bcancel{\cos(\omega t)}.
			\end{align*}
			Vemos que a equação acima é verdadeira somente se
			\begin{align*}
				\Aboxed{\omega=\sqrt{\frac{k}{m}}.}
			\end{align*}
		\end{column}
	\end{columns}
%	\end{multicols}
\end{frame}

\section{Energia do MHS}

\begin{frame}{Energia potencial no MHS}
	\begin{columns}[c]
		\begin{column}[c]{0.5\textwidth}
			Sabendo que a força é conservativa
			\begin{align*}
				U(x) - U(x_{0}) & = -\int^{x}_{x_{0}}\overbracket{F(x)}^{\alert{-kx}}dx\\
				U(x) - U(x_{0}) & = k\int^{x}_{x_{0}}xdx\\
				U(x) - U(x_{0}) & = \frac{kx^{2}}{2}-\frac{kx^{2}_{0}}{2}
			\end{align*}
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			Se $x_{0}=0$ \ensuremath{\Rightarrow} $U(0)=0$, portanto
			\[U(x) = \frac{kx^{2}}{2}\]
			\begin{block}{Energia potencial elástica do MHS}
				\[U(t) = \frac{kx^{2}_{m}}{2}cos^{2}\left(\omega t\right)\]
			\end{block}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Energia cinética no MHS}
	\begin{columns}[c]
		\begin{column}[c]{0.5\textwidth}
			Substituindo a expressão da velocidade
			\[v(t) = -x_{m}\omega sen\left(\omega t\right)\]
			na energia cinética
			\begin{align*}
				K(t) & = \frac{mv^{2}}{2}\\
				K(t) & = \frac{m\left(x_{m}\omega sen\left(\omega t\right)\right)^{2}}{2}
			\end{align*}
		\end{column}
		\begin{column}[c]{0.5\textwidth}
				\[K(t) = \frac{mx^{2}_{m}\omega^{2} sen^{2}\left(\omega t\right)}{2}\]
				mas $\omega^{2}=\frac{k}{m}$,
				\[K(t) = \frac{\alert{m}x^{2}_{m}\left(\frac{k}{\alert{m}}\right) sen^{2}\left(\omega t\right)}{2}\]
			\begin{block}{Energia cinética do MHS}
				\[K(t) = \frac{kx^{2}_{m}}{2}sen^{2}\left(\omega t\right)\]
			\end{block}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Energia total no MHS}
	\begin{columns}[c]
		\begin{column}[c]{0.5\textwidth}
	Sabendo que a energia total é
	\[E=K+U.\]
	Substituindo K e U,
	\begin{align*}
		E & = \frac{kx^{2}_{m}}{2}sen^{2}\left(\omega t\right) + \frac{kx^{2}_{m}}{2}cos^{2}\left(\omega t\right)\\
		E & = \frac{kx^{2}_{m}}{2}\left[\underbracket{sen^{2}\left(\omega t\right) + cos^{2}\left(\omega t\right)}_ {\alert{1}}\right]
	\end{align*}
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{block}{Energia total no MHS}
				\[E = \frac{kx^{2}_{m}}{2}\]
			\end{block}
		\begin{corollary}
			\justifying{A energia total do MHS é invariante no tempo, dependendo apenas da constante elástica e da amplitude de oscilação.}
		\end{corollary}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Solução da equação do oscilador harmônico forçado}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			Para resolver a equação anterior, onde temos explicitamente o termo $F_m\cos(\omega_0 t)$, supomos uma solução do tipo
			\begin{align*}
				x(t) = x_m\cos(\omega_0 t)+x_m'\sin(\omega_0 t).
			\end{align*}
			Calculando a derivada de ordem 2 e substituindo na equação teremos
			\begin{align*}
				\frac{d^2x}{dt^2} = -x_m\omega_0^2\cos(\omega_0 t)-x_m'\omega_0^2\sin(\omega_0 t).
			\end{align*}
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{align*}
				& -x_m\omega_0^2\cos(\omega_0 t)-x_m'\omega_0^2\sin(\omega_0 t)+\\
				 & + \omega^2(x_m\cos(\omega_0 t)+x_m'\sin(\omega_0 t)) =\\
				 & = \frac{F_m}{m}\cos(\omega_0 t).
			\end{align*}
			Reorganizando os termos do lado esquerdo teremos
			\begin{align*}
				& (\omega^2-\omega_0^2)x_m\cos(\omega_0 t)+(\omega^2-\omega_0^2)x_m'\sin(\omega_0 t)\\
				& = \frac{F_m}{m}\cos(\omega_0 t).
			\end{align*}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Ressonância}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Para satisfazer a equação anterior devemos impor a condição $x_m'=0$, para que fique apenas o termo dependente de $\cos(\omega_0 t)$, assim
			\begin{align*}
				(\omega^2-\omega_0^2)x_m\cos(\omega_0 t) = \frac{F_m}{m}\cos(\omega_0 t).
			\end{align*}
			Assim podemos afirmar que $x(t)$ é uma solução para o problema se
			\begin{align*}
				(\omega^2-\omega_0^2)x_m = \frac{F_m}{m},
			\end{align*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			ou seja,
			\begin{align*}
				x_m = \frac{F_m}{m(\omega^2-\omega_0^2)}.
			\end{align*}
			Portanto, podemos concluir que a amplitude $x_m$ da oscilação aumenta à medida que a frequência $\omega_0$ do oscilador se aproxima gradualmente da frequência $\omega$ de oscilação do MHS. \alert{Esse efeito é chamado de ressonância da oscilação}. Caso não houver amortecimento temos que $x_m\rightarrow \infty$ se $\omega_0\rightarrow \omega$.
		\end{column}
	\end{columns}
\end{frame}

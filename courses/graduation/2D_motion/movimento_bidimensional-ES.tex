\begin{frame}{Vetor posição e deslocamento}
	\begin{columns}
		\begin{column}[c]{0.55\textwidth}
			O vetor posição pode ser definido como
			\begin{equation*}
				\vec{r}=x\widehat{i}+y\widehat{j}+z\widehat{k}.
			\end{equation*}
			
%			\begin{block}{Deslocamento}
			O deslocamento da partícula da posição $\vec{r}_0$ para $\vec{r}_1$ é dado por,
				\begin{align*}
					\Delta \vec{r} & = \vec{r}_1-\vec{r}_0,\\
					\Delta \vec{r} & = (x_1\widehat{i}+y_1\widehat{j}+z_1\widehat{k})-(x_0\widehat{i}+y_0\widehat{j}+z_0\widehat{k}),\\
					\Delta \vec{r} & = (x_1-x_0)\widehat{i}+(y_1-y_0)\widehat{j}+(z_1-z_0)\widehat{k},\\
					\Aboxed{\Delta \vec{r} & = \Delta x\widehat{i}+\Delta y\widehat{j}+\Delta z\widehat{k}.}
				\end{align*}
%			\end{block}
		\end{column}

\begin{frame}[fragile]{Velocidade média}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Sabemos que
			\begin{equation*}
				\text{velocidade média} = \frac{\text{deslocamento}}{\text{intervalo de tempo}}.
			\end{equation*}
			Se uma partícula sofre um deslocamento $\Delta \vec{r}$ em um intervalo de tempo $\Delta t$, a velocidade média é dado por
			\begin{align*}
				\vec{v}_{\text{médio}} & = \frac{\Delta \vec{r}}{\Delta t},\\
	 		  	\vec{v}_{\text{médio}} & = \frac{\Delta x}{\Delta t}\widehat{i}+\frac{\Delta y}{\Delta t}\widehat{j}+\frac{\Delta z}{\Delta t}\widehat{k}.
			\end{align*}
		\end{column}

\begin{frame}[fragile]{Velocidade instantânea}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Reduzindo o intervalo de tempo $\Delta t$ a um valor infinitesimal, $\Delta t\rightarrow 0$, temos
			\begin{align*}
				\vec{v}(t)& = {\lim\limits_{\Delta t\rightarrow 0}}\frac{\Delta \vec{r}}{\Delta t}=\frac{d\vec{r}}{dt},\\
				\vec{v}(t) & = \frac{dx}{dt}\widehat{i}+\frac{dy}{dt}\widehat{j}+\frac{dz}{dt}\widehat{k}.
			\end{align*}
			Concluimos assim que
			\begin{equation*}
				\boxed{v_x = \frac{dx}{dt},v_y = \frac{dy}{dt},v_z = \frac{dz}{dt}.}
			\end{equation*}
		\end{column}

\begin{frame}[fragile]{Movimento balístico}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			A velocidade inicial $\vec{v}_0$ de um projétil que faz um ângulo $\theta_0$ com o eixo x pode ser decomposta em suas componentes horizontal e vertical,
			\begin{equation*}
				\vec{v}_0=v_0cos\theta_0\widehat{i}+v_0sen\theta_0\widehat{j}.
			\end{equation*}
			\begin{corollary}
				No movimento balístico, o movimento horizontal e o movimento vertical são independentes, e um não interfere no outro.
			\end{corollary}

\begin{frame}{Movimento vertical}
	\begin{columns}
		\begin{column}[t]{0.4\textwidth}
			No movimento vertical a aceleração é constante, onde
			\begin{equation*}
				\vec{a} = \vec{g}.
			\end{equation*}
			 Portanto é válido as fórmulas do movimento retilíneo uniformemente variado. Sabendo que
			 \begin{equation*}
				 v_{y} = vsen\theta,
			 \end{equation*}
		\end{column}
		\begin{column}[t]{0.6\textwidth}
			temos as equações do movimento vertical,
			\begin{table}
				\centering
				\caption*{\footnotesize Movimento vertical.}
				\begin{tabular}[c]{cl}
					\hline
					Definição & Equação\\
					\hline
					Velocidade & $v_y(t)=v_0sen\theta_0\alert{-}gt$\\
					Torricelli & $v_y^2=(v_0sen\theta_0)^2\alert{-}2g\Delta y$\\
					Posição & $y(t)= y_0+v_0sen\theta_0 t\alert{-}\frac{1}{2}gt^2$\\
					Posição & $y(t)=y_0+\frac{1}{2}\left(v_0sen\theta+v_y\right)t$\\
					Posição & $y(t)=y_0+v_yt\alert{+}\frac{1}{2}gt^2$\\
					\hline					
				\end{tabular}
			\end{table}	
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Equação da trajetória}
	\begin{multicols}{2}
		Podemos determinar o tempo na equação $x=x_0+v_{0}cos\theta_0 t$ no movimento horizontal, na forma
		\begin{equation*}
			t=\frac{x-x_0}{v_0cos\theta_0}.
		\end{equation*}
		Sabendo que o tempo registrado nos dois movimentos (horizontal e vertical) são os mesmos e considerando que $x_0=0$ e $y_0=0$, podemos assim substituir $t$ em $y(t)$,
		\begin{align*}
			y & = y_0+v_0sen\theta_0 t-\frac{1}{2}gt^2,\\
			y & = 0+\bcancel{v}_0sen\theta_0\left(\frac{x-0}{\bcancel{v}_0cos\theta_0}\right)-\\
			 & -\frac{1}{2}g\left(\frac{x-0}{v_0cos\theta_0}\right)^2,\\
			\Aboxed{y(x) & = tan\theta_0 x-\frac{gx^2}{2(v_0cos\theta_0)^2}.}
		\end{align*}
	\end{multicols}
\end{frame}

\begin{frame}{Alcance horizontal}
	\begin{multicols}{2}
		Para determinar a distância horizontal $R$ percorrida pelo projétil ($R=\Delta x$), substituímos R em
		\begin{equation*}
			R=v_0cos\theta_0 t.
		\end{equation*}
		O trecho $R$ corresponde as posições inicial e final onde $y_{\text{final}}=y_0$, portanto
		\begin{equation*}
			\Delta y=v_0sen\theta_0 t-\frac{1}{2}gt^2=0.
		\end{equation*}
		Eliminando t na primeira equação e substituindo na segunda temos
		\begin{align*}
			\Aboxed{R & = \frac{v_0^2}{g}sen2\theta_0.}
		\end{align*}
	\end{multicols}
	\begin{corollary}
		O alcance horizontal R é máximo para um ângulo de lançamento igual a \SI{45}{\degree}.
	\end{corollary}
\end{frame}

\begin{frame}{Movimento circular uniforme (continuação)}
%	\begin{multicols*}{2}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			Usando trigonometria podemos substituir $sen\theta$ por $y/r$ e $cos\theta$ por $x/r$, ou seja,
			\begin{equation*}
				\vec{v} = \left(-\frac{yv}{r}\right)\widehat{i}+\left(\frac{xv}{r}\right)\widehat{j}.
			\end{equation*}
			Considerando $v$ e $r$ constantes, podemos determinar a aceleração $\vec{a}$ derivando $\vec{v}$ em relação ao tempo,
			\begin{align*}
				\vec{a} & = \frac{d\vec{v}}{dt},
			\end{align*}
		\end{column}
		\begin{column}[c]{0.5\textwidth}
		\begin{align*}
			\vec{a} & = \left(-\frac{v}{r}\overbracket{\frac{dy}{dt}}^{\tiny \alert{v_y}}\right)\widehat{i}+\left(\frac{v}{r}\overbracket{\frac{dx}{dt}}^{\tiny\alert{v_x}}\right)\widehat{j},\\
			\vec{a} & = \left(-\frac{v\alert{v_y}}{r}\right)\widehat{i}+\left(\frac{v\alert{v_x}}{r}\right)\widehat{j}.
			\end{align*}
			Mas $v_x=-vsen\theta$ e  $v_y=vcos\theta$,
			\begin{align*}
				\Aboxed{\vec{a} & = \left(-\frac{v^2}{r}cos\theta\right)\widehat{i}+\left(-\frac{v^2}{r}sen\theta\right)\widehat{j}.}
			\end{align*}
		\end{column}
	\end{columns}

\begin{frame}{Transformar um número em notação científica}
	\begin{corollary}
		\begin{enumerate}[leftmargin=1.75cm]
			\item[Passo 1:] Escrever o número incluindo a vírgula.
			\item[Passo 2:] Andar com a vírgula até que reste somente um número diferente de zero no lado esquerdo.
			\item[Passo 3:] Colocar no expoente da potência de 10 o número de casas decimais que tivemos que "andar" com a vírgula. Se ao andar com a vírgula o valor do número diminuiu, o expoente ficará positivo, se aumentou o expoente ficará negativo.
		\end{enumerate}
	\end{corollary}
	\begin{exampleblock}{Exemplo}
		\begin{equation*}
			6\; 590\; 000\; 000\; 000\; 000,0 = \SI{6,59e15}{}		
		\end{equation*}
	\end{exampleblock}

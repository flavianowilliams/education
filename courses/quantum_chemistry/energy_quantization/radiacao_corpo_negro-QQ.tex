\begin{frame}{A antiga teoria quântica}
	\begin{enumerate}
		\item [\ding{51}] \textit{Numa reunião da sociedade alemã de física em 1900, Max Planck apresentou o seu artigo "Sobre a teoria da lei de distribuição de energia do espectro normal". \alert{Esse dia marca o nascimento da física quântica}.}
		\item[\ding{51}] \textit{Até o surgimento da equação de Schroedinger, diversos estudos foram desenvolvidos demonstrando falhas na física clássica. Esses estudos, \alert{chamados de antiga teoria quântica}, marcam os fundamentos da física quântica atual.}
		\item[\ding{51}] \textit{Assim como a teoria da relatividade, a física quântica representa uma generalização da física clássica, que inclui as leis clássicas como casos especiais.}
	\end{enumerate}
	\begin{corollary}
		Os fenômenos ligados a física quântica abrangem todas as áreas da física clássica: mecânica, termodinâmica, ondas, mecânica estatística e eletromagnetismo.
	\end{corollary}
\end{frame}

\begin{frame}{Lei de Stefan-Boltzman}
	\begin{enumerate}
		\item [\ding{51}] \textit{A radiação (\alert{intensidade}) incidente aumenta a vibração dos átomos, aumentando a temperatura do corpo negro;}
		\item [\ding{51}] \textit{A radiação emitida somente depende da temperatura do corpo negro;}
		\item [\ding{51}] \textit{De acordo com a teoria clássica, a radiação aumenta indefinidamente com a frequência da radiação emitida.}
	\end{enumerate}
	Chegando a lei de Stefan-Boltzman que relaciona a radiação emitida por um objeto com a sua temperatura T,
	\begin{align*}
		R(T) = \sigma T^4.
	\end{align*}
	onde $\sigma=\SI{5,6704e-8}{\watt\per\square\meter\kelvin}$
	\textit{\small \href{https://phet.colorado.edu/pt_BR/simulations/vector-addition}{Utilize a animação para ver como a área abaixo da curva da densidade de radiação aumenta com a temperatura}.}
\end{frame}

\begin{frame}{Radiância e radiação emitida pelo material}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
		Pela teoria do eletromagnetismo, a \alert{radiância}, ou seja, radiação emitida por cada onda de frequência $\nu$ ($R(\nu)$), que sai da cavidade de um corpo negro é dado por
		\begin{align*}
			R(\nu) = \frac{c}{4} u(\nu),
		\end{align*}
		onde $u(\nu)$ é a energia armazenada na cavidade por volume. Neste caso, a quantidade de energia na faixa de radiação com frequências entre $\nu$ e $\nu+d\nu$ vale $u(\nu)d\nu$. Portanto, a quantidade de radiação dR
		\end{column}
		\begin{column}[t]{0.5\textwidth}	
			emitida neste intervalo vale
	 		\begin{align*}
				dR = R(\nu)d\nu = \frac{c}{4}u(\nu)d\nu.
			\end{align*}
			Para determinar a radiação total emitida pelo corpo negro, integramos a equação acima (\alert{o que deve estar condizente com a lei de Stefan-Boltzman}),
			\begin{align*}
				R = \frac{c}{4}\int\limits_{0}^{\infty}u(\nu)d\nu.
			\end{align*}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}[fragile]{Relação entre energia e temperatura}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Pela termodinâmica, a energia de agitação dos elétrons por volume está associado estatisticamente com o valor médio da energia E multiplicado pelo número de cargas elétricas oscilantes,
			\begin{align*}
				u(\nu) = \left<E\right>n(\nu),
			\end{align*}
			 No caso acima, podemos dizer que cada elétron ao oscilar emite uma onda eletromagnética de frequência $\nu$. Essa onda eletromagnética confinada em uma cavidade se comporta como uma onda
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			 estacionária contendo um modo normal de vibração específico. Portanto, a onda eletromagnética produzida pelas cargas oscilantes podem escapar da cavidade, sendo em seguida captada pelo sensor como a radiância $R(\nu)$. É possível mostrar que a quantidade de modos de vibração por volume e por frequência é dado por (veja o apêndice D)
			\begin{align*}
				\Aboxed{n(\nu) = \frac{8\pi\nu^2}{c^3}.}
			\end{align*}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Valor médio da energia para uma distribuição contínua de energia}
	\begin{multicols*}{2}
		Para determinar o valor médio da energia $\left<E\right>$, usamos a expressão abaixo
		\begin{align*}
			\left<E\right> & = \frac{\int_{0}^{\infty}Ef(E)dE}{\int_{0}^{\infty}f(E)},\\
			f(E) & = Ae^{-E/k_BT}.
		\end{align*}
		onde $f(E)$ é conhecida como distribuição de Boltzmann e $k_B$ é chamado constante de Boltzmann.	Substituindo na expressão teremos
		\begin{align*}
			\left<E\right> & = \frac{\int_{0}^{\infty}\bcancel{A}Ee^{-E/k_BT}dE}{\int_{0}^{\infty}\bcancel{A}e^{-E/k_BT}dE},\\
			\left<E\right> & = \frac{\int_{0}^{\infty}Ee^{-E/k_BT}dE}{\int_{0}^{\infty}e^{-E/k_BT}dE}.
		\end{align*}
		É possível provar que o resultado da equação acima equivale a
		\begin{align*}
			\Aboxed{\left<E\right> = k_BT.}
		\end{align*}
	\end{multicols*}
\end{frame}

\begin{frame}{A equação de Rayleigh-Jeans}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Considerando uma distribuição contínua de energia para a radiação emitida pelo corpo negro, teremos que a densidade de energia $u(\nu)$ é dado por
			\begin{align*}
				u(\nu) & = \left<E\right>n(\nu);\\
				u(\nu) & =\left(k_BT\right)\left(\frac{8\pi\nu^2}{c^3}\right).
			\end{align*}
			Substituindo na fórmula da radiação,
			\begin{align*}
				R = \frac{c}{4}\int_{0}^{\infty}u(\nu)d\nu.
			\end{align*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
		\begin{align*}
			R & = \frac{\bcancel{c}}{4}\int_{0}^{\infty}\frac{8\pi k_BT}{\bcancel{c}^3}\nu^2d\nu,\\
			R & = \frac{8\pi k_BT}{c^2}\int_{0}^{\infty}\nu^2d\nu,\\
			R & = \left.\frac{8\pi k_BT}{3c^2}\nu^3\right|_{0}^{\infty}=\infty.
		\end{align*}
		Vemos que se considerarmos uma distribuição contínua de energia para $u(\nu)$, quando $\nu\rightarrow \infty$ teremos $R\rightarrow\infty$, o que resulta na catástrofe do ultravioleta.
		\end{column}
	\end{columns}
%	\end{multicols}
\end{frame}

\begin{frame}{A lei de Planck e o nascimento da física quântica}
	Para explicar a catástrofe do ultravioleta, duas causas são possíveis
	\begin{enumerate}
		\item [\ding{51}] \textit{Contagem errada do número de estados $n(\nu)$;}
		\item[\ding{51}] \textit{Valor da energia para cada modo vibracional está errado.}
	\end{enumerate}
	\begin{block}{Hipótese de Planck}
		Como opção, Planck sugeriu que a energia das cargas oscilantes, e consequentemente a radiância emitida por ela, ao invés de assumir qualquer valor, ela deverá ter valores discretos bem definidos, e deve também ser proporcional a frequência da radiação emitida,
		\begin{align*}
			E_n = nh\nu,\; n = 0,1,2,\cdots
		\end{align*}
		h é conhecido como a \alert{constante de Planck}.
	\end{block}
\end{frame}

\begin{frame}{Valor médio da energia para uma distribuição discreta de energia}
%	\begin{multicols*}{2}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
		O valor médio da energia $\left<E\right>$ para uma distrição discreta de energias é dado por
		\begin{align*}
			\left<E\right> & = \frac{\sum_{n=0}^{\infty} E\left(\bcancel{A}e^{-E/k_BT}\right)}{\sum_{n=0}^{\infty}\left(\bcancel{A}e^{-E/k_BT}\right)},\\
			\left<E\right> & = \frac{\sum_{n=0}^{\infty} Ee^{-E/k_BT}}{\sum_{n=0}^{\infty}e^{-E/k_BT}}.
		\end{align*}
		Como temos valores discretos de energia, a integral é substituída por uma somatória de valores de energia $E_n$. É possível provar que o resultado equivale a
		\end{column}
		\begin{column}[c]{0.5\textwidth}
		\begin{align*}
			\Aboxed{\left<E\right> = \frac{h\nu}{e^{\frac{h\nu}{k_BT}}-1}.}
		\end{align*}
		Substituindo em $u(\nu)$ teremos
		\begin{align*}
			\Aboxed{u(\nu) = \frac{8\pi h\nu^3}{c^3\left(e^{\frac{h\nu}{k_BT}}-1\right)}.}
		\end{align*}
		O resultado acima em $R(\nu)$ reproduz perfeitamente os dados experimentais.
		\end{column}
	\end{columns}
%	\end{multicols*}
\end{frame}

\begin{frame}{Lei de Planck para baixas frequências}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
		Pela teoria clássica temos que a teoria se aproxima do resultado experimental quando $\nu <<1$, usando a expansão
		\begin{align*}
			e^{\frac{h\nu}{k_BT}} = 1+\frac{h\nu}{k_BT}+\frac{1}{2}\left(\frac{h\nu}{k_BT}\right)^2+\cdots,
		\end{align*}
		podemos dizer que $e^{\frac{h\nu}{k_BT}}\approx 1$ se $h\nu <<k_BT$. Substituindo em $u(\nu)$ resulta em
		\end{column}
		\begin{column}[t]{0.5\textwidth}
		\begin{align*}
			u(\nu) & \approx \frac{8\pi \bcancel{h}\nu^{\bcancel{3}}T}{c^3\left(\bcancel{1}+\frac{\bcancel{h}\bcancel{\nu}}{k_BT}-\bcancel{1}\right)},\\
			u(\nu) & \approx \frac{8\pi k_BT\nu^2}{c^3},
		\end{align*}
		o que corresponde a lei de Rayleigh-Jeans da teoria clássica da radiação.
		\end{column}
	\end{columns}
	\begin{corollary}
		Para $h\nu/k_BT<<1$ a equação de Planck se resume na lei de Rayleigh-Jeans.
	\end{corollary}
\end{frame}

\begin{frame}{Lei de Planck para frequências elevadas}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Considerando a emissão de radiação de altas frequências, onde $h\nu\gg kT$. Isso faz com que $e^{\frac{h\nu}{k_BT}}\gg 1$, portanto podemos dezprezar o valor 1 do denominador de $u(\nu)$. Temos assim
			\begin{align*}
				u(\nu) & \approx \frac{8\pi h\nu^{3}}{c^3e^{\frac{h\nu}{k_BT}}} \Rightarrow \left(\frac{8\pi h\nu^{3}}{c^3}\right)e^{-\frac{h\nu}{k_BT}}.
			\end{align*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			Pela definição de radiância, onde $R(\nu)=cu(\nu)/4$ temos
			\begin{align*}
				R(\nu) & = \frac{2\pi\nu^3}{c^2}e^{-\frac{h\nu}{k_BT}}.
			\end{align*}
			Sabendo que $e^{-\frac{h\nu}{k_BT}}\rightarrow 0$ para $h\nu\gg kT$, podemos dizer que $R(\nu\rightarrow\infty)\rightarrow 0$, concordando com o resultado experimental.
		\end{column}
	\end{columns}
	\begin{corollary}
		Para $h\nu/k_BT\gg 1$ a equação de Planck se resume ao limite de Wien, onde $R(\nu\rightarrow\infty)\rightarrow 0$.
	\end{corollary}
\end{frame}

\begin{frame}{Apêndice B - Demonstração do valor kT para a energia média}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Considere a energia média $\left<E\right>$ onde definimos $\beta = 1/kT$,
			\begin{align*}
				\left<E\right> & = \frac{\int_{0}^{\infty}Ee^{-\beta E}dE}{\int_{0}^{\infty}e^{-\beta E}dE}
			\end{align*}
			Podemos ver que $-\frac{d}{d\beta}e^{-\beta E}=Ee^{-\beta E}$,
			\begin{align*}
				\left<E\right> & = \frac{\int_{0}^{\infty}\left(-\frac{d}{d\beta}e^{-\beta E}\right)dE}{\int_{0}^{\infty}e^{-\beta E}dE}
			\end{align*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			\begin{align*}
				\left<E\right> & = \frac{-\frac{d}{d\beta}\left(\int_{0}^{\infty}e^{-\beta E}dE\right)}{\int_{0}^{\infty}e^{-\beta E}dE}.
			\end{align*}
			Definindo $F(\beta)=\int_{0}^{\infty}e^{-\beta E}dE$ teremos
			\begin{align*}
				\left<E\right> & = -\frac{\frac{dF(\beta)}{d\beta}}{F(\beta)},
			\end{align*}
			Assim poderemos obter $\left<E\right>$ sem a necessidade de resolver a integral.
		\end{column}
	\end{columns}
\end{frame}
%
\begin{frame}{Apêndice B - Demonstração do valor kT para a energia média}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			Pela propriedade do logaritmo e empregando a regra da cadeia, temos que a sua derivada é dada por
			\begin{align*}
				\frac{d\ln\left[1/F(\beta)\right]}{d\beta} = \left(\frac{d\ln(1/F)}{dF}\right)\left(\frac{dF}{d\beta}\right)
			\end{align*}
			Porém, sabemos que $\frac{d\ln F}{dF}=1/F$, portanto
			\begin{align*}
				\frac{d\ln\left[F(\beta)\right]}{d\beta} = \left(\frac{1}{F}\right)\left(\frac{dF}{d\beta}\right)
			\end{align*}
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			Assim podemos ver que
			\begin{align*}
				\left<E\right> & = -\frac{d}{d\beta}\ln[F(\beta)],\\
				& = -\frac{d}{d\beta}\ln\left(\int_{0}^{\infty}e^{-\beta E}dE\right).
			\end{align*}
			Resolvendo a integral teremos
			\begin{align*}
				\left<E\right> & = \frac{d\ln(\beta)}{d\beta}=1/\beta,\\
				\Aboxed{\left<E\right> & = kT.}
			\end{align*}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}[fragile]{Apêndice C - Demonstração do valor médio da energia segundo Planck}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Segundo Planck, para o cálculo da energia média, devemos adotar a somatória ao invés de uma integral, assim
			\begin{align*}
				\left<E\right> & = -\frac{d}{d\beta}\ln\left(\sum e^{-\beta E}\right),
			\end{align*}
			Levando em consideração que
			\begin{align*}
				 \sum_{n=1}^{\infty}e^{-\beta E} & = \frac{1}{e^{\beta E}-1}
			\end{align*}
			teremos
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			\begin{align*}
				\left<E\right> & = \frac{d}{d\beta}\ln\left(e^{\beta E}-1\right),\\
				& = \frac{d}{dx}\ln x\left(\frac{dx}{d\beta}\right),\\
				& = \frac{E}{e^{\beta E}-1}.
			\end{align*}
			Definindo $E=h\nu$ e $\beta = 1/k_BT$ teremos
			\begin{align*}
				\boxed{\left<E\right> = \frac{h\nu}{e^{h\nu/k_BT}-1}}
			\end{align*}
		\end{column}		
	\end{columns}
\end{frame}

\begin{frame}{Apêndice D - Contagem dos modos normais de vibração em uma cavidade}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			Sabendo que 
			\begin{align*}
				n & = \frac{2L}{c} \nu,\\
				\Delta n & = \frac{2L}{c}\Delta \nu.
			\end{align*}
			podemos dizer que a cada intervalo de frequência $\Delta \nu$ teremos a seguinte quantidade de modos normais
			\begin{align*}
				N(\nu)\Delta \nu = \left(\frac{\pi}{2}\right)\left(\frac{2L\nu}{c}\right)^2\left(\frac{2L\Delta \nu}{c}\right).
			\end{align*}
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{align*}
				N(\nu)\Delta \nu = \frac{4\pi L^3\nu^2}{c^3}\Delta \nu.
			\end{align*}
			Esse valor representa a quantidade de modos de vibração em um intervalo $\Delta \nu$, para saber a quantidade por volume devemos dividir por $L^3$, e considerando o caso mais geral de uma onda não-polarizada devemos multiplicar por 2,
			\begin{align*}
				\boxed{n(\nu)\Delta \nu = \frac{8\pi\nu^2}{c^3}\Delta \nu.}
			\end{align*}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Apêndice E - A hipótese de Planck e a lei de Stefan-Boltzman}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Utilizando a expansão em série da função $1/(e^x+1)$ teremos
			\begin{align*}
				\frac{1}{\left(e^{\frac{h\nu}{k_BT}}-1\right)} & = \sum_{n=1}^{\infty}e^{-\frac{nh\nu}{k_BT}}
			\end{align*}
			Substituindo na fórmula de $u(\nu)$ segundo Planck teremos
			\begin{align*}
			R = \frac{c}{4}\int_{0}^{\infty}u(\nu)d\nu.
			\end{align*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			\begin{align*}
			R & = \frac{2\pi h}{c^2}\int_{0}^{\infty}\nu^3\sum e^{-\frac{nh\nu}{k_BT}}d\nu,\\
			R & = \frac{2\pi h}{c^2}\sum\int_{0}^{\infty}\nu^3e^{-\frac{nh\nu}{k_BT}}d\nu.
			\end{align*}
			Resolvendo a integral através do método de integração por partes chegamos a
			\begin{align*}
				R & = \left(\frac{12\pi k_B^4}{c^2h^3}\sum 1/n^4\right)T^4,\\
				R & = \sigma T^4.
			\end{align*}
		\end{column}
	\end{columns}
	%	\end{multicols}
\end{frame}

\begin{frame}{Apêndice F - A hipótese de Planck e o deslocamento de Wien}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			Considere a expressão da radiância de acordo com a hipótese de Planck. Considerando $c=\lambda\nu$, podemos expressá-la em função de comprimento de onda da radiação,
			\begin{align*}
				R(\lambda)d\lambda = \frac{2\pi hc^2}{\lambda^5\left(e^{\frac{hc}{\lambda k_BT}}-1\right)}d\lambda.				
			\end{align*}
			Considerando a situação onde $hc/k_BT$, podemos dizer que $e^{hc/k_BT}>>1$. Portanto podemos fazer a aproximação
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{align*}
				R(\lambda) \approx 2\pi hc^2\lambda^{-5}e^{-\frac{hc}{\lambda k_BT}}.
			\end{align*}
			O comprimento de onda da máxima radiância é obtida através de $\frac{dR}{d\lambda}=0$,
			\begin{align*}
				\frac{dR}{d\lambda} = \frac{2\pi hce^{-\frac{hc}{\lambda k_BT}}}{\lambda^{5}}\left(-5\lambda^{-1}+\frac{hc}{\lambda^2k_BT}\right) = 0
			\end{align*}
			Isolando $\lambda$ na equação teremos
			\begin{align*}
				\Aboxed{\lambda T = \frac{hc}{5k_B} = \num{2.89e6}\;nm\cdot K.}
			\end{align*}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Vis„o microscůpica da resistividade de um material (continuaÁ„o)}
	\begin{multicols}{2}
		Se um elťtron de massa m ť submetido a um campo elťtrico $\vec{E}$ ele sofre uma aceleraÁ„o a dado por
		\begin{equation*}
			a = \frac{F}{m} = \frac{eE}{m}.
		\end{equation*}
		Supondo que o intervalo de tempo mťdio $\tau$ entre colisűes o elťtron adquire uma velocidade de deriva dado por $v_d = a\tau$. Portanto
		\begin{align*}
			v_d & = a\tau,\\
			v_d & = \frac{eE\tau}{m}.
		\end{align*}
		Sabendo que $J=nev_d$ temos $\frac{J}{ne}=\frac{eE\tau}{m}$, ou seja,
		\begin{equation*}
			\boxed{E = \rho J,}
		\end{equation*}
		onde $\rho=\frac{m}{e^2n\tau}$. $\rho$ ť definido como resistividade do material.
	\end{multicols}
\end{frame}

\begin{frame}{ResistÍncia a partir da resistividade}

\begin{frame}{O conceito de campo elétrico}
	\doubt Se a interação elestrostática é percebida a distância, como uma carga elétrica percebe a presença de outra, a distância, intensidade e sinal da sua carga elétrica?

\begin{frame}{Características do campo elétrico}
	\begin{columns}
		\begin{column}{0.6\textwidth}
	\begin{corollary}
			A direção e o sentido do vetor $\vec{E}$ são dados pela direção e pelo sentido da força que atua na carga de prova positiva, ou seja,
		\begin{itemize}
			\item Se q>0, o campo elétrico $\vec{E}$ e a força $\vec{F}$ tem o mesmo sentido;
			\item Se q<0, o campo elétrico $\vec{E}$ e a força $\vec{F}$ tem sentidos opostos.
		\end{itemize}
	\end{corollary}
		\end{column}

\begin{frame}{Transformar um número em notação científica}

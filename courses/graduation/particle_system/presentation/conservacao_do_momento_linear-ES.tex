\section{Centro de massa}

\subsection{}

\begin{frame}{Centro de massa}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			O centro de massa de um sistema de partículas ou objeto rígido é o ponto que se move como se toda a massa do sistema estivesse concentrada nesse ponto e todas as forças externas estivessem aplicadas nesse ponto.
			\begin{corollary}
				Á partir do centro de massa podemos determinar com mais facilidade o movimento do sistema.
			\end{corollary}
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.2]{figuras/figura-1.png}
				\caption*{\footnotesize Representação do centro de massa de um bastão \cite{halliday}.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Sistema de duas partículas}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Considere um sitema binário (contendo duas partículas) ao longo do eixo x, podemos substituir o conjunto por outro equivalente onde toda a massa está concentrada no seu centro de massa,
			\begin{align*}
				(m_1+m_2)x_{cm}& = m_1x_1+m_2x_2,\\
				\Aboxed{x_{cm} & = \frac{m_1x_1+m_2x_2}{m_1+m_2},}
			\end{align*}
			onde $x_{cm}$ é a localização da massa total do sistema.

\begin{frame}{Sistema de várias partículas}
%	\begin{multicols}{2}
		Para um sistema contendo N partículas, podemos generalizar a idéia anterior na forma
		\begin{align*}
			(m_1+m_2+m_3+\cdots +m_N)x_{cm} & = m_1x_1+m_2x_2+m_3x_3+\cdots+m_Nx_N,\\
			x_{cm} & = \frac{m_1x_1+m_2x_2+m_3x_3+\cdots+m_Nx_N}{m_1+m_2+m_3+\cdots +m_N},\\
			\Aboxed{x_{cm} & = \frac{1}{M} \sum_{i=1}^{N}m_ix_i,}
		\end{align*}
		onde $M=m_1+m_2+m_3+\cdots +m_N$, é a massa total do sistema.
%	\end{multicols}
\end{frame}

\begin{frame}{Objetos maciços}
	No caso de um objeto rígido com uma distribuição contínua de massa, a somatória se transforma em uma integral, resultando em
	\begin{align*}
		x_{cm} = \frac{1}{M}\int xdm,\quad y_{cm} = \frac{1}{M}\int ydm,\quad z_{cm} = \frac{1}{M}\int zdm.
	\end{align*}
	No caso de objetos homogêneos, onde a densidade $\rho$ é constante, temos $\rho=\frac{M}{V}$, onde V é o volume do objeto. Substituindo na equação acima temos
	\begin{align*}
		x_{cm} = \frac{1}{\bcancel{\rho} V}\int x\bcancel{\rho} dV,\quad y_{cm} = \frac{1}{\bcancel{\rho} V}\int y\bcancel{\rho} dV,\quad z_{cm} = \frac{1}{\bcancel{\rho} V}\int z\bcancel{\rho} dV,\\
		\Aboxed{x_{cm} = \frac{1}{V}\int x(V)dV,\quad y_{cm} = \frac{1}{V}\int y(V)dV,\quad z_{cm} = \frac{1}{V}\int z(V)dV.}
	\end{align*}
\end{frame}

\begin{frame}{Outras considerações importantes a respeito do centro de massa}
	\begin{enumerate}
		\item [\ding{51}] O centro de massa de um objeto não precisa estar necessariamente no interior desse objeto;
		\item [\ding{51}] Em um objeto heterogêneo, o centro de massa possui a tendência de estar mais próximo da região que possui maior distribuição de massa;
		\item[\ding{51}] Em objetos que possuem simetria, o centro de massa se torna mais simples, pois nesse caso ele se encontra no \alert{ponto, na linha ou plano de simetria}. Exemplo: Em uma esfera homogênea, o centro de massa se encontra no centro da esfera, em um cilindro homogêneo, o centro de massa se encontra no eixo do cilindro;
		\item[\ding{51}] Em um objeto homogêneo, o centro de massa se encontra no centro geométrico desse objeto.
	\end{enumerate}
\end{frame}

\section{Momento linear}

\subsection{}

\begin{frame}{Segunda Lei de Newton para um sistema de partículas}
	\begin{multicols}{2}
		Sabemos que
		\begin{align*}
			M\vec{r}_{CM} = m_1\vec{r}_1+m_1\vec{r}_2+\cdots+m_N\vec{r}_N,
		\end{align*}
		onde M é a massa total. Derivando duas vezes temos
		\begin{align*}
			M\vec{a}_{CM} = m_1\vec{a}_1+m_1\vec{a}_2+\cdots+m_N\vec{a}_N.
		\end{align*}
		Identificando os termos do lado direito com a segunda Lei de Newton temos
		\begin{align*}
			M\vec{a}_{CM} = \vec{F}_1+\vec{F}_2+\cdots+\vec{F}_N.
		\end{align*}
		Sabendo que a força resultante atuando no conjunto é soma vetorial das forças externas atuando em cada partícula e identificando o termo do lado esquerdo com a segunda Lei de Newton, podemos dizer que a força resultante é dado por
		\begin{align*}
			\vec{F}_{res} & = \vec{F}_1+\vec{F}_2+\cdots+\vec{F}_N,\\
			\Aboxed{\vec{F}_{res} & = M\vec{a}_{CM}.}
		\end{align*}
	\end{multicols}
\end{frame}

\begin{frame}{Momento linear de um sistema de partícula}
	\begin{multicols}{2}
		Sabemos que
		\begin{align*}
			M\vec{r}_{CM} = m_1\vec{r}_1+m_1\vec{r}_2+\cdots+m_N\vec{r}_N,
		\end{align*}
		onde M é a massa total. Derivando temos
		\begin{align*}
			M\vec{v}_{CM} = m_1\vec{v}_1+m_1\vec{v}_2+\cdots+m_N\vec{v}_N,
		\end{align*}
		Podemos definir quantitativamente a quantidade de movimento de uma partícula de massa m e velocidade $\vec{v}$ como
		\begin{align*}
			\Aboxed{\vec{p} = m\vec{v}.}
		\end{align*}
		Temos assim
		\begin{equation*}
			M\vec{v}_{CM} = \vec{p}_1+\vec{p}_2+\cdots +\vec{v}_N.
		\end{equation*}
		Dessa maneira, a quantidade de movimento $\vec{P}$ do centro de massa pode ser definido como
		\begin{align*}
			\Aboxed{\vec{P} = M\vec{v}_{CM}.}
		\end{align*}
	\end{multicols}
\end{frame}

\begin{frame}{Impulso como variação da quantidade de movimento}
	\begin{multicols}{2}
		Derivando a equação anterior temos
		\begin{align*}
			\frac{d\vec{P}}{dt} & = M\frac{d\vec{v}_{CM}}{dt},\\
			\frac{d\vec{P}}{dt} & = M\vec{a}_{CM}.
		\end{align*}
		Sabendo que $\vec{F}_{res} = M\vec{a}_{CM}$ temos
		\begin{align*}
			\Aboxed{\vec{F}_{res} = \frac{d\vec{P}}{dt}.}
		\end{align*}
		Aplicando a regra diferencial temos
		\begin{align*}
			d\vec{P} = \vec{F}_{res} dt.
		\end{align*}
		Integrando durante um intervalo de tempo $\Delta t$, onde $\Delta t=t_f-t_i$ temos
		\begin{align*}
			\vec{J} = \Delta\vec{P} = \int_{t_i}^{t_f}\vec{F}_{res}dt,
		\end{align*}
		onde $\vec{J}$ é chamado impulso e é igual a variação do momento $\Delta \vec{P}$.
	\end{multicols}
\end{frame}

\begin{frame}{Impulso à partir do gráfico Força versus Tempo}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Se $\vec{F}(t)$ é uma função contínua, o impulso sobre a partícula é igual a área abaixo da curva em um gráfico força versus tempo. Além do mais, à partir do teorema do valor médio, o impulso pode ser calculado sabendo o valor médio da força,
			\begin{align*}
				J = F_{\text{médio}}\Delta t,
			\end{align*}
				onde essa expressão é útil para determinar a força média causada pela colisão de N partículas idênticas em um alvo fixo, dado por
				\begin{align*}
					\Aboxed{F_{\text{médio}} = N\Delta p/\Delta t.}
				\end{align*}
		\end{column}

\section{Conservação do momento linear}

\subsection{}

\begin{frame}{Conservação do momento linear}
	\begin{multicols}{2}
		Em um sistema isolado não há forças externas atuando sobre ele ($F_{res}=0$), portanto
		\begin{align*}
			\frac{d\vec{P}}{dt} = \vec{0},
		\end{align*}
		ou seja,
		\begin{align*}
			\Aboxed{P=\text{constante}.}
		\end{align*}
		Em outras palavras, \alert{o momento linear total do sistema permanece invariante sob translação do seu centro de massa}.
	\end{multicols}
	\begin{block}{Lei da conservação do momento linear}
		Se um sistema de partículas não está submetido a forças externas (como no caso de um sistema isolado), o momento linear total desse sistema não pode mudar.
	\end{block}
\end{frame}

\begin{frame}{Demais considerações a respeito da Lei da conservação do momento linear}
	\begin{enumerate}
		\item [\ding{51}] Se uma das componentes da força \textbf{externa} aplicada a um sistema isolado é nula, a componente do momento linear do sistema em questão em relação ao mesmo eixo não pode variar.
		\item[\ding{51}] Se não houver forças externas atuando no sistema teremos
		\begin{align*}
			\vec{F}_{res} = M\vec{a}_{CM} = \vec{0},
		\end{align*}
		ou seja, o momento linear do centro de massa permanece invariante sob translação no espaço,
		\begin{align*}
			\Aboxed{P_{\text{CM}}=\text{constante}.}
		\end{align*}
	\end{enumerate}
\end{frame}

\begin{frame}{Colisões inelásticas em uma dimensão}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Sabemos que
			\begin{align*}
				\left(\frac{\text{momento total }\vec{P}_i}{\text{antes da colisão}}\right)=\left(\frac{\text{momento total }\vec{P}_f}{\text{antes da colisão}}\right).

\begin{frame}{Energia cinética em um sistema de partículas}
	\begin{columns}
		\begin{column}[c]{0.7\textwidth}
			A energia cinética $K$ de um sistema de N partículas vale
			\begin{align*}
				K = \frac{1}{2}m_1v^2_1+\frac{1}{2}m_2v^2_2+\cdots +\frac{1}{2}m_Nv^2_N.
			\end{align*}
			Porém, podemos determinar o movimento do conjunto como a combinação do movimento do centro de massa ($M\vec{v}_{\text{CM}}$) com o movimento de cada partícula em relação ao seu centro de massa ($m\vec{v}'$).
			\begin{align*}
				K & = \frac{1}{2}m_1\left|\vec{v}'_1+\vec{v}_{\text{CM}}\right|^2+\cdots +\frac{1}{2}m_1\left|\vec{v}'_N+\vec{v}_{\text{CM}}\right|^2.
			\end{align*}
			onde $M=m_1+m_2+\cdots +m_N$.
		\end{column}

\begin{frame}{Colisões perfeitamente inelásticas unidimensionais}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Para que a energia cinética seja mínima após a colisão, o movimento das partículas em relação ao centro de massa deve ser zero \alert{(nota-se que não necessariamente o movimento do centro de massa deve ser zero!!!!)}. Para isso, elas devem se mover juntos com a mesma velocidade $\vec{V}$. Se considerarmos que uma das partículas possui velocidade zero antes da colisão, teremos
			\begin{align*}
				m_1\vec{v}_{(1,i)} & =(m_1+m_2)\vec{V},\\
				\vec{V} & = \frac{m_1}{m_1+m_2}\vec{v}_{(1,i)}.
			\end{align*}
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.33]{figuras/colisao-perfeitamente-inelastica.png}
				\caption*{\footnotesize Representação de uma colisão perfeitamente inelástica entre dois objetos \cite{halliday}.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Colisões elásticas em uma dimensão}
	Se durante a colisão a energia cinética total dos objetos envolvidos não é convertida em outra forma de energia, podemos dizer que
	\begin{align*}
		\left(\frac{\text{Energia cinética total }\vec{P}_i}{\text{antes da colisão}}\right)=\left(\frac{\text{Energia cinética total }\vec{P}_f}{\text{antes da colisão}}\right).
	\end{align*}
	Portanto, nas colisões elásticas a energia cinética dos objetos envolvidos na colisão pode mudar, mas a energia cinética total permanece constante. Em suma, podemos dizer que
	\begin{align*}
		m_1\vec{v}_{(1,i)}+m_2\vec{v}_{(2,i)} & = m_1\vec{v}_{(1,f)}+m_2\vec{v}_{(2,f)},\\
		\frac{1}{2}m_1v^2_{(1,i)}+\frac{1}{2}m_2v^2_{(2,i)} & = \frac{1}{2}m_1v^2_{(1,f)}+\frac{1}{2}m_2v^2_{(2,f)}.
	\end{align*}
\end{frame}

\begin{frame}{Colisões elásticas em uma dimensão (alvo estacionário)}
\begin{columns}
	\begin{column}[c]{0.6\textwidth}
		Considerando a partícula 2 como um alvo estacionário teremos $v_{2,i}$, portanto
		\begin{align*}
			m_1v_{1,i} & = m_1v_{1,f}+m_2v_{2,f},\\
			\frac{1}{2}m_1v^2_{(1,i)} & = \frac{1}{2}m_1v^2_{(1,f)}+\frac{1}{2}m_1v^2_{(2,f)}.
		\end{align*}
		Substituindo $v_{1,i}$ da primeira equação teremos
		\begin{align*}
			v_{1,f} & = \frac{m_1-m_2}{m_1+m_2}v_{1,i},\\
			v_{2,f} & = \frac{2m_1}{m_1+m_2}v_{1,i}.
		\end{align*}
	\end{column}
	\begin{column}[c]{0.4\textwidth}
		\begin{figure}
			\centering
			\includegraphics[scale=0.3]{figuras/colisao-inelastica.png}

		\end{figure}
	\end{column}
\end{columns}
\end{frame}

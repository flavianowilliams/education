\section{Vazão ou fluxo}

\subsection{}

\begin{frame}{Definição de vazão ou fluxo}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{block}{Fluxo}
				Volume de fluido que atravessa uma seção transversal do tubo de corrente por unidade de tempo.
				\begin{align*}
					Z=\frac{Volume}{\Delta t}.
				\end{align*}
			\end{block}
			\begin{corollary}
				Pela definição de fluxo, percebe-se que a sua unidade no SI é metro cúbico por segundo ($m^3/s$).
			\end{corollary}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}[c]
				\includegraphics[scale=0.4]{figuras/vazao.jpg}
				\caption*{\footnotesize Fluxo ou vazão de um fluido.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}[fragile]{Equação da continuidade baseada na Lei da conservação da matéria}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			Supondo uma quantidade de fluido que percorre uma distância $S_1$ de área $A_1$ no intervalo de tempo $\Delta t$, o volume ocupado por esse fluido é
			\begin{align*}
				V = \Delta S_1 A_1. 
			\end{align*}
			Mas sabendo que $\Delta S_1 = v_1\Delta t$, onde $v_1$ é a velocidade das moléculas do fluido que percorre esse espaço, temos
			\begin{align*}
				V = v_1 A_1 \Delta t.
			\end{align*}

\begin{frame}{Pressão, velocidade e altura de um fluido em duas regiões distintas (continuação)}
	\begin{multicols}{2}
		\alert{Se não houver perdas de energia}, a energia mecânica do fluido permanece inalterada, de modo que o trabalho total realizado deve ser igual a variação das energias cinéticas e potenciais,
		\begin{align*}
			\tau_1+\tau_2 = \Delta E_m = \Delta E_c+\Delta E_p,
		\end{align*}
		mas
		\begin{align*}
			\Delta E_c & = \frac{1}{2}mv^2_2-\frac{1}{2}mv^2_1,\\
			\Delta E_p & = mgh_2-mgh_1,
		\end{align*}
		portanto
		\begin{align*}
			\tau_1+\tau_2 & = \overbracket{\left(\frac{1}{2}mv^2_2-\frac{1}{2}mv^2_1\right)}^{\alert{\Delta E_c}}+\\
			& +\overbracket{\left(mgh_2-mgh_1\right)}^{\alert{\Delta E_p}},\\
			\tau_1+\tau_2 & = \frac{1}{2}mv^2_2+mgh_2-\frac{1}{2}mv^2_1-mgh_1,\\
			\tau_1+\tau_2 & = \frac{1}{2}mv^2_2+mgh_2-\left(\frac{1}{2}mv^2_1+mgh_1\right).
		\end{align*}
	\end{multicols}
\end{frame}

\begin{frame}{Pressão, velocidade e altura de um fluido em duas regiões distintas (continuação)}
	\begin{columns}
		\begin{column}[t]{0.45\textwidth}
			Separando os termos da região 1 da região 2 temos a equação
			\begin{align*}
				\tau_1 + \frac{1}{2}mv^2_1+mgh_1 & = \alert{-}\tau_2 + \frac{1}{2}mv^2_2+\\
				& + mgh_2.
			\end{align*}
			Mas mostramos que
			\begin{align*}
				\tau_1 & = p_1V,\\
				\tau_2 & = -p_2V.
			\end{align*}
		\end{column}
		\begin{column}[t]{0.55\textwidth}
			Substituindo na equação acima temos
			\begin{align*}
				\alert{p_1V} + \frac{1}{2}mv^2_1+mgh_1 & = -\left(\alert{-p_2V}\right) + \\
				& + \frac{1}{2}mv^2_2 + mgh_2,
			\end{align*}
			mas sabemos que $V=\frac{m}{\rho},$
			\begin{align*}
				p_1\alert{\frac{\bcancel{m}}{\rho}} + \frac{1}{2}\bcancel{m}v^2_1+\bcancel{m}gh_1 & = p_2\alert{\frac{\bcancel{m}}{\rho}} + \frac{1}{2}\bcancel{m}v^2_2 + \bcancel{m}gh_2,\\
				\frac{p_1}{\rho} + \frac{v^2_1}{2}+gh_1 & = \frac{p_2}{\rho} + \frac{v^2_2}{2} + gh_2.
			\end{align*}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Pressão, velocidade e altura de um fluido em duas regiões distintas (continuação)}
	\begin{block}{Equação de Bernoulli}
		Para um fluido não viscoso com escoamento laminar a soma das parcelas hidrostáticas e hidrodinâmicas é a mesma em cada ponto do fluido, no qual vale a equação
		\begin{align*}
			\frac{p_1}{\rho} +gh_1 + \frac{v^2_1}{2} & = \frac{p_2}{\rho} +gh_2 + \frac{v^2_2}{2} = \cdots,\\
			\frac{p}{\rho} +gh + \frac{v^2}{2} & = \text{constante}.
		\end{align*}
	\end{block}
	\begin{corollary}
		A equação de Bernoulli corresponde na hidrodinâmica à Lei de conservação da energia na mecânica.
	\end{corollary}
\end{frame}

\begin{frame}{Analisando os termos da equação de Bernoulli}
	Supondo a densidade constante ao longo de todo o fluido, podemos multiplicar todos os termos da equação por $\rho$ e obter a relação
	\begin{align*}
		\underbracket{\alert{\bcancel{\rho}}\frac{p}{\bcancel{\rho}} +\alert{\rho}gh}_{\text{Lei de Stevin}} + \alert{\rho}\frac{v^2}{2} = \text{constante}.
	\end{align*}
	\begin{corollary}
		\begin{itemize}
			\item Parcela hidrostática \alert{$\left(p +\rho gh\right)$}: Corresponde a pressão hidrostática no fluido;
			\item Parcela fluidodinâmica \alert{$\left(\rho\frac{v^2}{2}\right)$}: Corresponde a pressão hidrodinâmica;
			\item Se o fluido está em repouso $\frac{\rho v^2}{2}=0$ temos a Lei de Stevin.
		\end{itemize}
	\end{corollary}
\end{frame}

\begin{frame}{Transformar um número em notação científica}
	\begin{corollary}
		\begin{enumerate}[leftmargin=1.75cm]
			\item[Passo 1:] Escrever o número incluindo a vírgula.
			\item[Passo 2:] Andar com a vírgula até que somente reste um número diferente de zero no lado esquerdo.
			\item[Passo 3:] Colocar no expoente da potência de 10 o número de casas decimais que tivemos que "andar" com a vírgula. Se ao andar com a vírgula o valor do número diminuiu, o expoente ficará positivo, se aumentou o expoente ficará negativo.
		\end{enumerate}
	\end{corollary}
	\begin{exampleblock}{Exemplo}
		\begin{equation*}
			6\; 590\; 000\; 000\; 000\; 000,0 = \SI{6,59e15}{}		
		\end{equation*}
	\end{exampleblock}
\end{frame}

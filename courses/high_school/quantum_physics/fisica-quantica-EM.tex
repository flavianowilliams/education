\section{Quantização da energia}

\subsection{}

\begin{frame}{Radiação do corpo negro}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Um corpo negro é um material que absorve toda a radiação que incide sobre ele. Após absorver a radiação, o corpo negro aquece e emite radiação própria, que por sua vez depende da temperatura.
			\begin{corollary}
				\begin{equation*}
					\text{Radiação} = \frac{\text{Energia}}{\text{(Área)}\text{(Tempo)}}
				\end{equation*}
			\end{corollary}
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}
				\includegraphics[scale=0.15]{corpo-negro.png}
				\caption*{\footnotesize Representação de um corpo negro.}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		Desde Maxwell consideramos que a luz é uma onda eletromagnética, cuja intensidade é definida como energia por tempo e área;
	\end{corollary}
\end{frame}

\begin{frame}{Discrepância entre a teoria clássica da radiação e a experiência}
	\begin{block}{O que era esperado pela teoria clássica}
		\begin{itemize}
			\item A radiação emitida pelo corpo negro deveria assumir qualquer valor, independente da frequência da onda eletromagnética.
			\item A intensidade da onda eletromagnética é diretamente proporcional ao quadrado da frequência, portanto a intensidade da radiação deveria aumentar com o aumento da frequência.
		\end{itemize}
	\end{block}
	\begin{block}{O que os físicos observaram}
		A radiação emitida pelo corpo negro aumenta até uma certa frequência, atingindo um valor máximo e decaindo a zero em seguida, à medida que a frequência aumenta.
	\end{block}
\end{frame}

\begin{frame}{A Lei de Planck}
%	\begin{corollary}
%		A agitação das cargas elétricas aumenta após absorver a radiação.
%	\end{corollary}
	\begin{block}{Hipótese de Planck}
		A energia das cargas oscilantes era uma variável discreta (quantizada) e seria proporcional a frequência da radiação emitida,
		\begin{align*}
			E_n & = nhf,\quad n=0,1,2,\cdots\\
			h & \approx \SI{6,63e-34}{\joule\cdot\second},\quad\text{(Constante de Planck)}.
		\end{align*}
	\end{block}
	\begin{corollary}
		As idéias de Plank sobre quantização da energia marca o nascimento da Física quântica.
	\end{corollary}
\end{frame}

\section{O modelo atômico}

\subsection{}

\begin{frame}{Modelo planetário do átomo}
	\begin{columns}
		\begin{column}{0.7\textwidth}
			\begin{block}{O que era esperado pela teoria clássica}
				\begin{itemize}
					\item Os elétrons se movem ao redor do núcleo em órbitas circulares;
					\item Pela teoria do eletromagnetismo, cargas em movimento emitem radiação diminuindo sua energia;
					\item À medida que a energia diminui, a órbita do elétron encolhe e ele colapsa para dentro do núcleo.
				\end{itemize}
			\end{block}
		\end{column}
		\begin{column}[c]{0.3\textwidth}
			\begin{figure}
				\Centering
				\includegraphics[scale=0.15]{modelo-atomico-classico.png}
				\caption*{\footnotesize Teoria clássica.}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		Com o colapso do elétron no interior do núcleo, não seria possível a formação de moléculas ou demais combinações envolvendo átomos.
	\end{corollary}
\end{frame}

\begin{frame}{O átomo de Bohr}
	\begin{columns}
		\begin{column}[c]{0.75\textwidth}
			\begin{block}{Postulados de Bohr}
				\begin{itemize}
					\item Os elétrons se movem em certas órbitas bem definidas sem irradiar energia;
					\item O átomo emite radiação quando um elétron faz uma transição de uma órbita para outra;
					\item No limite de grandes órbitas e altas energias, os resultados quânticos devem coincidir com a teoria clássica.
				\end{itemize}
			\end{block}
		\end{column}
		\begin{column}[c]{0.25\textwidth}
			\begin{figure}
				\Centering
				\includegraphics[scale=0.3]{modelo-bohr.jpg}
				\caption*{\footnotesize Modelo de Bohr.}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{block}{Princípio da correspondência}
		Para grandes números quânticos, os cálculos quânticos e os clássicos levam ao mesmo resultado.
	\end{block}
\end{frame}

\begin{frame}{Espectro de linhas do hidrogênio}
	\begin{columns}
		\begin{column}[t]{0.6\textwidth}
			\begin{block}{Níveis de energia do átomo de hidrogênio}
				\begin{align*}
					E_n & = \frac{E_1}{n^2},\quad(n=1,2,3,\cdots),\\
					E_1 & = \SI{-2.18e-18}{J},\quad\text{(estado fundamental)}.
				\end{align*}
			\end{block}
			\begin{corollary}
				O elétron ao redor do átomo adquire valores discretos de energia, e no limite $n\rightarrow\infty$ se aproxima do resultado clássico (energias no continuum).
			\end{corollary}
		\end{column}
		\begin{column}[t]{0.4\textwidth}
			\begin{figure}
				\begin{tikzpicture}[scale=0.18,transform shape,font=\huge]

					\tkzInit[xmin=-5,xmax=12,ymin=-23,ymax=2]

				\caption*{\footnotesize Níveis de energia do hidrogênio.}

		\end{column}
	\end{columns}
\end{frame}

\section{Dualidade onda partícula}

\subsection{}

\begin{frame}{O efeito fotoelétrico}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{block}{O que era esperado pela teoria clássica}
				\begin{itemize}
					\item A energia cinética dos elétrons deveria depender da intensidade da luz;
					\item O efeito fotoelétrico deveria ocorrer com luz de qualquer frequência;
					\item Deveria haver um retardo no tempo, de modo que o elétron absorveria continuamente o feixe de energia.
				\end{itemize}
			\end{block}
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}
				\includegraphics[scale=0.15]{efeito-fotoeletrico.jpg}
				\caption*{\footnotesize Representação de um corpo negro.}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{block}{Hipótese de Einstein}
		A luz é constituída por pacotes de energia $(E=hf)$ chamados fótons.
	\end{block}
\end{frame}

\begin{frame}{Postulado de de Broglie}
	\begin{block}{Hipótese de de Broglie}
		Devido a simetria da natureza, o dualismo onda-partícula é um fenômeno absolutamente geral,
		\begin{align*}
			f & = \frac{E}{h},\quad\text{(Efeito fotoelétrico)},\\
			\lambda & = \frac{h}{p},\quad\text{(Postulado de de Broglie)}.
		\end{align*}
	\end{block}
	\begin{corollary}
			Os elétrons se movem como ondas ao redor do núcleo, o que explica o modelo atômico de Bohr.
	\end{corollary}
\end{frame}

\begin{frame}{Transformar um número em notação científica}
	\begin{corollary}
		\begin{enumerate}[leftmargin=1.75cm]
			\item[Passo 1:] Escrever o número incluindo a vírgula.
			\item[Passo 2:] Andar com a vírgula até que reste somente um número diferente de zero no lado esquerdo.
			\item[Passo 3:] Colocar no expoente da potência de 10 o número de casas decimais que tivemos que "andar" com a vírgula. Se ao andar com a vírgula o valor do número diminuiu, o expoente ficará positivo, se aumentou o expoente ficará negativo.
		\end{enumerate}
	\end{corollary}
	\begin{exampleblock}{Exemplo}
		\begin{equation*}
			6\; 590\; 000\; 000\; 000\; 000,0 = \SI{6,59e15}{}		
		\end{equation*}
	\end{exampleblock}
\end{frame}

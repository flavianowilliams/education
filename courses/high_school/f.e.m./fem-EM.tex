\section{Força eletromotriz}

\subsection{}

\begin{frame}{Força eletromotriz}
	Para que ocorra uma corrente elétrica ao longo do circuito é necessário um dispositivo, como uma bateria ou gerador, que realize um trabalho $\tau$ afim de deslocar uma quantidade de carga $\Delta q$ de um ponto a outro desse circuito. Portanto, o trabalho $\alert{\varepsilon}$ necessário para deslocar cada elemento de carga q do terminal positivo para o terminal negativo de uma bateria ou gerador pode ser dado por $\alert{\varepsilon = \frac{\tau}{\Delta q}}$.
	\begin{block}{Definição de força eletromotriz (f.e.m.)}
		Trabalho que a bateria realiza para transportar cada unidade de carga elétrica, ao longo do circuito, do terminal positivo para o negativo.
	\end{block}
	\begin{corollary}
		A unidade de medida de f.e.m. no SI é $\text{J/C}\equiv \text{Volts}$.
	\end{corollary}
\end{frame}

\begin{frame}{Força contraeletromotriz}
	Um gerador de f.e.m. transfere energia às cargas que passam através dele, transformando energia química em elétrica. \begin{frame}{Potência fornecida por uma bateria}
	\begin{multicols}{2}
		Vimos que o trabalho realizado por uma bateria para deslocar uma quantidade de carga $\Delta q$ é dado por $\tau = \varepsilon \Delta q$. Divindo pelo intervalo de tempo $\Delta t$ que essa carga é transferida tempos
		\begin{equation*}
			\frac{\tau}{\Delta t} = \varepsilon\cdot \frac{\Delta q}{\Delta t}.
		\end{equation*}
		Pela definição de potência temos
		\begin{align*}
			P & = \frac{\tau}{\Delta t},\\
			P & = \varepsilon\cdot \frac{\Delta q}{\Delta t}.
		\end{align*}
		onde usamos a definição $i=\frac{\Delta q}{\Delta t}$.
	\end{multicols}
	\begin{block}{Potência elétrica de uma bateria}
		\begin{equation*}
			P = \varepsilon\cdot i.
		\end{equation*}	
	\end{block}
\end{frame}

\begin{frame}{A Lei de Kirchhoff}
	\begin{columns}
		\begin{column}[c]{0.7\textwidth}
   			\footnotesize
			Podemos resumir num circuito ao lado que a cada segundo
			\begin{itemize}
				\item na bateria, a energia química se transforma em energia elétrica;
				\item no resistor, a energia elétrica se transforma em energia térmica;
				\item no motor, a energia elétrica se transforma em energia mecânica.
			\end{itemize}
			Portanto, pela Lei da conservação da energia, a energia fornecida a cada  segundo para girar o motor e aquecer o resistor deve ser equivalente a potência fornecida pela bateria.
			\begin{align*}
				\underbracket{\varepsilon i}_{(Bateria)} & = \underbracket{\varepsilon'i+r'i^2}_{(Motor)}+\underbracket{Ri^2}_{(Resistor)},\\
			\end{align*}
		\end{column}
		\begin{column}[c]{0.3\textwidth}
			\vspace*{-0.4cm}
			\begin{figure}
				\centering
				\includegraphics[scale=0.35]{figuras/figura-3.png}
				\caption*{\footnotesize Circuito contendo a bateria, motor (contendo resistência interna r') e resistor.}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{block}{Lei de Kirchhoff}
		A soma dos potenciais em um circuito fechado deve ser zero.
	\end{block}
\end{frame}

\begin{frame}{Diferença de potencial nos terminais de uma bateria}
	\begin{columns}
		\begin{column}{0.7\textwidth}
			Suponha uma bateria como na figura ao lado com uma resistência interna r. Aplicando a Lei de Kihrchoff temos
			\begin{align*}
				\varepsilon i & = V_{AB}i+ri^2,\\
				V_{AB}i & = \varepsilon i-ri^2,\\				
				\Aboxed{V_{AB} & = \varepsilon \alert{- ri}.}
			\end{align*}
			Onde $V_{AB}i$ é a potência real da bateria e $V_{AB}$ a sua ddp.
		\end{column}
		\begin{column}{0.3\textwidth}
			\begin{figure}
				\centering

	\begin{corollary}
		Num circuito onde a bateria possui resistência interna considerável, a potência que irá fornecer ao circuito será menor que aquela cuja resistência é praticamente zero.
	\end{corollary}

\begin{frame}{Transformar um número em notação científica}
	\begin{corollary}
		\begin{enumerate}[leftmargin=1.75cm]
			\item[Passo 1:] Escrever o número incluindo a vírgula.
			\item[Passo 2:] Andar com a vírgula até que reste somente um número diferente de zero no lado esquerdo.
			\item[Passo 3:] Colocar no expoente da potência de 10 o número de casas decimais que tivemos que "andar" com a vírgula. Se ao andar com a vírgula o valor do número diminuiu, o expoente ficará positivo, se aumentou o expoente ficará negativo.
		\end{enumerate}
	\end{corollary}
	\begin{exampleblock}{Exemplo}
		\begin{equation*}
			6\; 590\; 000\; 000\; 000\; 000,0 = \SI{6,59e15}{}		
		\end{equation*}
	\end{exampleblock}
\end{frame}

\section{Lei de Faraday}

\subsection{}

\begin{frame}{Definição de fluxo magnético}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			Definimos fluxo do campo magnético $\vec{B}$ que atravessa uma área A como a somatória das linhas de campo magnético que atravessam perpendicularmente essa área.
			\begin{block}{Fluxo do campo magnético}
				\begin{equation*}
					\varphi = BAcos\theta
				\end{equation*}
			\end{block}
		\end{column}
		\begin{column}{0.5\textwidth}
			\begin{figure}
				\input{tikz/fluxo-magnetico.pgf}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Fluxo magnético e a orientação de $\vec{B}$ em relação a área}
	\begin{figure}
		\centering
		\begin{subfigure}[b]{0.45\textwidth}
			\centering
		\input{tikz/fluxo-magnetico-0.pgf}
			\caption*{\footnotesize Fluxo magnético máximo ($\theta = 0^o$)}
		\end{subfigure}
		\begin{subfigure}[b]{0.45\textwidth}
			\centering
		\input{tikz/fluxo-magnetico-90.pgf}
			\caption*{\footnotesize Fluxo magnético zero ($\theta = 90^o$)}
		\end{subfigure}
	\end{figure}
\end{frame}

\begin{frame}{Variação do fluxo magnético ao longo de uma espira}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			\begin{block}{Lei de Faraday}
				Sempre que houver uma variação do fluxo magnético através de um circuito, aparecerá, nesse circuito, uma f.e.m. induzida,
				\begin{equation*}
					\left|\frac{\Delta \varphi}{\Delta t}\right| = \varepsilon
				\end{equation*}
			\end{block}
		\end{column}
		\begin{column}{0.5\textwidth}
			\begin{figure}
				\input{tikz/lei-faraday.pgf}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		No SI a unidade de medida de fluxo magnético é Weber (Wb).
	\end{corollary}
\end{frame}

\section{Lei de Lenz}

\subsection{}

\begin{frame}{Sentido da corrente induzida em relação a variação do fluxo}
	\begin{block}{Lei de Lenz}
		A corrente induzida em um circuito aparece sempre com um sentido tal que o campo magnético que ela cria tende a contrariar a variação do fluxo magnético que a originou.
	\end{block}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{figure}
				\centering
				\begin{subfigure}[b]{0.45\textwidth}
					\centering
					\includegraphics[scale=0.2]{figuras/figura-2a}
					\caption*{\footnotesize $\frac{\Delta \varphi}{\Delta t}>0$ no tempo.}
				\end{subfigure}
				\begin{subfigure}[b]{0.45\textwidth}
					\centering
					\includegraphics[scale=0.2]{figuras/figura-2b}
					\caption*{\footnotesize $\frac{\Delta \varphi}{\Delta t}<0$ no tempo.}
				\end{subfigure}
			\end{figure}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{block}{Lei de Faraday-Lenz}
				\begin{equation*}
					\frac{\Delta \varphi}{\Delta t} = \alert{-}\varepsilon
				\end{equation*}
			\end{block}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Transformar um número em notação científica}
	\begin{corollary}
		\begin{enumerate}[leftmargin=1.75cm]
			\item[Passo 1:] Escrever o número incluindo a vírgula.
			\item[Passo 2:] Andar com a vírgula até que reste somente um número diferente de zero no lado esquerdo.
			\item[Passo 3:] Colocar no expoente da potência de 10 o número de casas decimais que tivemos que "andar" com a vírgula. Se ao andar com a vírgula o valor do número diminuiu, o expoente ficará positivo, se aumentou o expoente ficará negativo.
		\end{enumerate}
	\end{corollary}
	\begin{exampleblock}{Exemplo}
		\begin{equation*}
			6\; 590\; 000\; 000\; 000\; 000,0 = \SI{6,59e15}{}		
		\end{equation*}
	\end{exampleblock}
\end{frame}

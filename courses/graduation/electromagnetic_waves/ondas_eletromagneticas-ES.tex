\section{Transporte de energia}

\subsection{}

\begin{frame}{Energia transportada em um onda eletromagnética}
	\begin{multicols}{2}
		O vetor de Poyting $S$ é definido como a energia transportada pela onda por tempo e área,
		\begin{align*}
			S = \left(\frac{\text{energia/tempo}}{\text{área}}\right).
		\end{align*}
		Porém, energia/área = potência, assim
		\begin{align*}
			S =\left(\frac{\text{potência}}{\text{área}}\right).
		\end{align*}
		A partir do vetor de Poyting podemos encontrar a intensidade de uma onda eletromagnética. Da energia elétrica em um capacitor e em um indutor encontramos as expressões que definem a densidade de energia elétrica $u_E$ e magnética $u_B$ em função dos campos elétrico $E$ e magnético $B$,
		\begin{align*}
			u_E & = \frac{1}{2}\varepsilon_0E^2,\\
			u_B & = \frac{1}{2\mu_0}B^2.
		\end{align*}
	\end{multicols}
\end{frame}

\begin{frame}{Densidade de energia transportada por uma onda eletromagnética}
	\begin{multicols}{2}
		Portanto, a energia total $u$ transportada por área pode ser definido como
		\begin{align*}
			u & = u_E + u_B,\\
			u(x,t) & = \frac{1}{2}\varepsilon_0E(x,t)^2 + \frac{1}{2\mu_0}B(x,t)^2,
		\end{align*}
		onde, pelas equações de Maxwell descobrimos que para uma onda se propagando na direção x teremos
		\begin{align*}
			E(x,t) & = E_m\cos(kx-\omega t),\\
			B(x,t) & = B_m\cos(kx-\omega t).
		\end{align*}
		Substituindo na expressão da energia resulta em
		\begin{align*}
			u(x,t) & = \frac{1}{2}\varepsilon_0[E_m\cos(kx-\omega t)]^2 +\\ 
			& + \frac{1}{2\mu_0}[B_m\cos(kx-\omega t)]^2.
		\end{align*}
	\end{multicols}
\end{frame}

\begin{frame}{Densidade de energia transportada por uma onda eletromagnética (continuação)}
	\begin{multicols}{2}
		Simplificando a expressão anterior obteremos
		\begin{align*}
			u = \frac{1}{2}\left(\varepsilon_0E_m^2+\frac{1}{\mu_0}B_m^2\right)\cos^2(kx-\omega t).
		\end{align*}
		Contudo, sabemos a partir das equações de Maxwell que $B_m=\frac{E_m}{c}$, substituindo
		\begin{align*}
			u = \frac{E_m^2}{2}\left(\varepsilon_0+\frac{1}{\mu_0c^2}\right)\cos^2(kx-\omega t).
		\end{align*}
		Pelas equações de Maxwell também sabemos que $\varepsilon_0=\frac{1}{\mu_0c^2}$, substituímos encontramos
		\begin{align*}
			u & = \frac{E_m^2}{2}\left(\alert{\frac{1}{\mu_0c^2}}+\frac{1}{\mu_0c^2}\right)\cos^2(kx-\omega t),\\
			u & = \frac{E_m^2}{\bcancel{2}}\left(\frac{\bcancel{2}}{\mu_0c^2}\right)\cos^2(kx-\omega t),\\
			\Aboxed{u(x,t) & = \frac{E_m^2}{\mu_0c^2}\cos^2(kx-\omega t).}
		\end{align*}
	\end{multicols}
\end{frame}

\begin{frame}{Vetor de Poyting}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Podemos perceber que
			\begin{align*}
				\left(\frac{\text{energia}}{\text{área}\times\text{ tempo}}\right)=\left(\frac{\text{energia}}{\text{volume}}\times \text{velocidade}\right).
			\end{align*}
			Portanto, no caso de uma onda monocromática se propagando no vácuo, podemos dizer que
			\begin{align*}
				S = u\times c,
			\end{align*}
			ou seja, o vetor de Poyting é justamente a densidade de energia multiplicado pela
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			velocidade da luz. Substituindo $u(x,t)$ teremos
		 	\begin{align*}
				S & = \frac{E_m^2}{\mu_0c}\cos^2(kx-\omega t),
			 \end{align*}
			 Sabendo que $E(x,t)=E_m\cos(kx-\omega t)$, representamos S na forma compacta
			\begin{align*}
				\Aboxed{S(x,t) & = \frac{1}{c\mu_0}E(x,t)^2.}
			\end{align*}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Valor médio do vetor de Poyting}
	\begin{columns}
		\begin{column}[t]{0.65\textwidth}
			Entretando, do ponto de vista prático é mais conveniente obter informações a respeito do valor médio de uma grandeza ao invés de seu valor instantâneo. Dessa maneira definimos a intensidade $I$ como
			\begin{align*}
				I = \left<S(x,t)\right>=\left<\frac{1}{c\mu_0}E(x,t)^2\right>.
			\end{align*}

			\begin{align*}

\begin{frame}{Polarização da luz incidente não polarizada}
%	\begin{multicols}{2}
	\begin{columns}[c]
		\begin{column}[c]{0.5\textwidth}
			Se onda for não polarizada, ou seja, teremos frentes de onda cuja direção do campo elétrico assume todos os valores entre $\theta=\SI{0}{\degree}$ a $\theta=\SI{360}{\degree}$. Sabendo que $E^2_m=E^2_{m,x}+E^2_{m,y}$ teremos
			\begin{align*}
				I & = \frac{1}{2c\mu_0}E^2_m,\\
				I & = \frac{1}{2c\mu_0}E^2_{m,x}+\frac{1}{2c\mu_0}E^2_{m,y}.
			\end{align*}
			A soma das componentes em y e a mesma de z, portanto
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{align*}
				I & = 2\left(\frac{1}{2c\mu_0}E^2_m\right).
			\end{align*}
			 Entretanto, as componentes na direção do eixo x não atravessam, pois $\theta=\SI{90}{\degree}$. Portanto, a intensidade será a metade da onda incidente,
			\begin{align*}
				\Aboxed{I & = \frac{1}{2}I_0.}
			\end{align*}
			onde $I_0$ é a intensidade da onda não polarizada.
		\end{column}
	\end{columns}	
%	\end{multicols}
\end{frame}

\begin{frame}{Redução da velocidade da onda ao atravessar a matéria}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
%	\begin{multicols}{2}
			Considere uma onda atravessando um meio dielétrico neutro e na ausência de corrente elétrica, nesse caso as equações de Maxwell no vácuo ainda são válidas, com a pequena modificação, que o campo elétrico da onda deverá ser reduzido por uma quantidade $\kappa$, onde $\kappa$ é a constante dielétrica do meio, ou seja,
			\begin{align*}
				\vec{E}_{\kappa} = \frac{\vec{E}}{\kappa},\quad(\kappa>1).
			\end{align*}
			Assim, a partir da lei de Ampère-Maxwell
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			\begin{align*}
				\frac{\partial (\kappa E_{\kappa})}{\partial t} & = -\frac{1}{\varepsilon_{0}\mu_{0}}\frac{\partial B}{\partial x},\\
				\frac{\partial E_{\kappa}}{\partial t} & = -\frac{1}{\kappa\varepsilon_{0}\mu_{0}}\frac{\partial B}{\partial x}.
			\end{align*}
			Da lei de Faraday-Lenz temos a equação da luz no vácuo, onde a variação no tempo de $\vec{B}$ induz um campo $\vec{E}$ circular,
			\begin{align*}
				\frac{\partial E_{\kappa}}{\partial x} & = -\frac{\partial B}{\partial t}.
			\end{align*}
		\end{column}
	\end{columns}
%	\end{multicols}
\end{frame}

\begin{frame}{Redução da velocidade da onda ao atravessar a matéria (continuação)}
%	\begin{multicols}{2}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
		Considerando que os campos elétrico e magnético admitem soluções do tipo $A(x,t) = A_{m}cos\left(kx-\omega t\right)$ teremos
		\begin{align*}
			\frac{\partial E_{\kappa}}{\partial x} & = \alert{-E_{\kappa m}k}sen\left(kx -\omega t\right),\\
			\frac{\partial B}{\partial t} & = \alert{B_{m}\omega} sen\left(kx-\omega t\right),\\
			\frac{\partial E_{\kappa}}{\partial t} & = \alert{E_{\kappa m}\omega} sen\left(kx-\omega t\right),\\
			\frac{\partial B}{\partial x} & = \alert{-B_{m}k}sen\left(kx -\omega t\right)
		\end{align*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
		Substituindo nas equações anteriores teremos
		\begin{align*}
			\frac{E_{\kappa m}}{B_{m}} & = \frac{\omega}{k} = v,\\
			\frac{E_{\kappa m}}{B_{m}} & = \frac{k}{\omega\kappa\varepsilon_{0}\mu_{0}} = \frac{1}{v\kappa\varepsilon_{0}\mu_{0}}.
		\end{align*}
			Sabendo que $c^2=\frac{1}{\varepsilon_0\mu_0}$ teremos
			\begin{align*}
				\Aboxed{v = \frac{c}{\sqrt{\kappa}}.}
			\end{align*}
		\end{column}
	\end{columns}

\end{frame}

\begin{frame}{Lei de Snell}
	\begin{multicols}{2}
		Temos assim
		\begin{align*}
			\lambda_1 & = \overline{AC}sen(\theta_1),\\
			\lambda_2 & = \overline{AC}sen(\theta_2).
		\end{align*}
		Dividindo os dois lados da equação temos
		\begin{equation*}
			\frac{\lambda_1}{\lambda_2} = \frac{sen(\theta_1)}{sen(\theta_2)}.
		\end{equation*}
		Sabemos que na onda a relação entre comprimento de onda e velocidade é dado por $v=\lambda f$, onde f é a frequência que não se altera quando a onda atravessa de um meio ao outro, portanto
		\begin{equation*}
			\frac{v_1}{v_2} = \frac{sen(\theta_1)}{sen(\theta_2)}.
		\end{equation*}
		Se o meio 1 for o vácuo ou ar temos $v_1=c$, ou seja,
		\begin{equation*}
			\boxed{\frac{c}{v_2} = \frac{\lambda_1}{\lambda_2} = \frac{sen(\theta_{ar})}{sen(\theta_2)}.}
		\end{equation*}
	\end{multicols}
\end{frame}

\begin{frame}{Índice de refração}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			No caso da luz atravessando do ar para um meio mais refrigente temos a relação
			\begin{equation*}
				\frac{c}{v} = \frac{sen(\theta_{ar})}{sen(\theta)}.
			\end{equation*}
			Se definirmos $n=\sqrt{\kappa}$ teremos $n=c/v$ como a fração da velocidade da luz que diminui ao atravessar o meio, assim
			\begin{equation*}
				n = \frac{sen(\theta_{ar})}{sen(\theta_2)}.
			\end{equation*}
			Para uma onda que atravessa dois meios
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			com índices de refração diferentes temos
			\begin{small}
				\begin{NodesList}[margin=1cm]
					\begin{align*}
						\frac{v_1}{v_2} & = 	\frac{sen(\theta_1)}{sen(\theta_2)},\AddNode\\
						\frac{\left(\frac{\alert{c}}{v_2}\right)}{\left(\frac{\alert{c}}{v_1}\right)} & = 	\frac{sen(\theta_1)}{sen(\theta_2)},\AddNode
					\end{align*}
					\LinkNodes{$\times \alert{\frac{c}{c}}$}
				\end{NodesList}
			\end{small}
			\begin{block}{Lei de Snell}
				\begin{equation*}
					\frac{n_2}{n_1} =\frac{\lambda_1}{\lambda_2} = \frac{sen(\theta_1)}{sen(\theta_2)},\quad (n>1).
				\end{equation*}
			\end{block}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Reflexão total}
	\begin{columns}
		\begin{column}{0.45\textwidth}
			Aplicando a Lei de Snell temos
			\begin{small}
				\begin{align*}
					n_2sen(L) & = n_1\overbracket{sen(90^o)}^{\alert{1}},\\
					sen(L) & = \frac{n_1}{n_2}.
				\end{align*}
			\end{small}

				\begin{equation*}
					L=sen^{-1}\left(\frac{n_1}{n_2}\right),\quad (n_2>n_1).
				\end{equation*}
			\end{block}

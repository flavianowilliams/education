Com o passar dos anos, o óxido de ferro sofria ação do campo magnético da Terra e se alinhava preferencialmente em uma única direção, à medida que a resina ia endurecendo.

\section{Força magnética e a corrente elétrica}

\subsection{}

\begin{frame}{O nascimento do eletromagnetismo}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			Um Físico chamado Hans Christian Oersted mostrou em 1820 que uma corrente que flui ao longo de um circuito é capaz de orientar uma bússola na direção perpendicular ao fio. Como se essa corrente fosse capaz de produzir efeitos magnéticos ao redor do fio. \alert{Essa experiência marca o início da teoria do eletromagnetismo, onde dois fenomenos distintos da física se influenciam mutuamente. Até o momento eles eram tratados como algo totalmente separados um do outro.}
		\end{column}
		\begin{column}{0.5\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.35]{figuras/figura-6.png}
				\caption*{\footnotesize Experiência de Oersted usando uma bússola e um circuito elétrico simples.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{A relação entre eletricidade e magnetismo}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			\begin{corollary}
				Quando duas cargas elétricas estão em movimento, manifestam-se entre elas, além da força eletrostática, uma outra força, denominada força magnética.
			\end{corollary}
		\end{column}
		\begin{column}{0.5\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.45]{figuras/figura-13.png}
				\caption*{Força magnética entre dois fios \cite{forcamanetica}.}
%				\includegraphics[scale=0.35]{figuras/figura-7.png}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		A força magnética que atua nas cargas Q e q possui direção perpendicular ao movimento das cargas.
	\end{corollary}
\end{frame}

\begin{frame}{A força de Lorentz}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			No final do século XIX, Hendrik Lorentz verificou que o módulo da força magnética depende:
			\begin{enumerate}
				\item [\ding{51}] Do valor da carga elétrica q;
				\item [\ding{51}] do módulo da velocidade $\vec{v}$;
				\item [\ding{51}] do ângulo $\theta$ entre a velocidade de q e do campo magnético $\vec{B}$.
			\end{enumerate}
		\end{column}
		\begin{column}{0.5\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.35]{figuras/questao-3.png}
				\caption*{\footnotesize Força magnética $\vec{F}_{\text{mag}}$ atuando em uma carga p em movimento no interior do campo $\vec{B}$.}
			\end{figure}
		\end{column}
	\end{columns}	
\end{frame}

\begin{frame}{Expressão matemática da força de Lorentz}
	\begin{multicols}{2}
		Lorentz obteve as seguintes relações:
		\begin{align*}
			F & \alpha q,\\
			F & \alpha v,\\
			F & \alpha sen(\theta),
		\end{align*}
		que pode ser representado numa única relação de proporcionalidade,
		\begin{equation*}
			F \alpha qvsen(\theta).
		\end{equation*}
		Foi definido que a constante de proporcionalidade é o valor do campo magnético $B$.
	\end{multicols}
	\begin{block}{Força de Lorentz}
		A força magnética que atua numa carga q em movimento num campo magnético B é dado por
		\begin{equation*}
			F = qvBsen(\theta).
		\end{equation*}
	\end{block}
\end{frame}

\begin{frame}{Direção e sentido da força magnética}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			Características da força magnética (força de Lorentz):
			\begin{enumerate}
				\item [\ding{51}] Módulo: $F=qvBsen(\theta)$,
				\item [\ding{51}] Direção: $\vec{F}$ é perpendicular a $\vec{v}$ e $\vec{B}$,
				\item [\ding{51}] Sentido: Dado pela regra da mão direita.
			\end{enumerate}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.24]{figuras/figura-8.png}
				\caption*{\footnotesize Regra da mão direita.}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		Se a carga q for \alert{negativa}, o \alert{sentido da força magnética} será \alert{contrário} àquele que será observado pela carga positiva.
	\end{corollary}
\end{frame}

\begin{frame}{Força magnética em um condutor}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
%	\begin{multicols}{2}
			Considere um campo magnético entrando na tela, perpendicular a corrente elétrica que flui num condutor de comprimento L, a força de Lorentz é dada por
			\begin{equation*}
				F = qvB.
			\end{equation*}
			Se ao invés de uma carga, houver N cargas se movimentando com velocidade $\vec{v}$, a força total no fio será a soma das forças em cada 
carga,
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{align*}
				F_{fio} & = NF,\\
				F_{fio} & = B\alert{Nq}v.
			\end{align*}
			\begin{figure}
				\centering
				\includegraphics[scale=0.3]{figuras/figura-9.png}
				\caption*{\footnotesize Força $\vec{f}$ atuando em cada carga elétrica em movimento que atravessa a seção reta do fio condutor.}
			\end{figure}
		\end{column}
	\end{columns}		
%	\end{multicols}
\end{frame}

\begin{frame}{Força magnética em um condutor}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			Mas pela definição de corrente, $i=\Delta q/\Delta t$, temos
			\begin{equation*}
				Nq = i\Delta t.
			\end{equation*}
			Sabendo que $v=\frac{L}{\Delta t}$ obtemos
			\begin{align*}
				F & = B(i\bcancel{\Delta t})\frac{L}{\bcancel{\Delta t}} = BiL.
			\end{align*}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.25]{figuras/figura-10.png}
				%\caption*{\footnotesize Força magnética $\vec{F}$ atuando em um fio imerso em um campo magnético $\vec{B}$.}
			\end{figure}
		\end{column}
	\end{columns}
	\begin{corollary}
		Se um fio retilíneo, de comprimento L, percorrido por uma corrente i, for colocado em um campo magnético B, sobre esse fio atuará uma força magnética F dada por
		\begin{equation*}
			F = iBLsen(\theta).
		\end{equation*}
	\end{corollary}
\end{frame}

\section{Aplicações do magnetismo}

\subsection{}

\begin{frame}{Campo magnético geográfico da Terra}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			\begin{itemize}
				\item A Terra se comporta como um ímã gigante onde o polo norte magnético encontra-se \alert{próximo} do sul geográfico e vice-versa.
				\item As linhas de indução produzem um campo magnético ao redor da Terra, o que a protege contra radiações vindas do Sol.
				\item As cargas elétricas ao encontrarem o campo magnético sofrem uma força de Lorentz que a deslocam para o polo norte da Terra, causando o fenômeno da aurora Boreal.
			\end{itemize}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.25]{figuras/figura-4.png}
				\caption*{\footnotesize Linhas de campo magnético da Terra.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Motor de corrente contínua}

\begin{frame}{A levitação magnética sem atrito (Maglev)}
	\begin{columns}[t]
		\begin{column}{0.6\textwidth}
			\begin{itemize}
				\small 
				\item O trem levita através da repulsão magnética;
				\item A repulsão magnética pode ser entre um íman e um supercondutor (Maglev Cobra\cite{maglev}) ou através de eletroímans (Transrapid de Xangai, JR-Maglev);
				\item Como o movimento do trem ocorre por levitação, o atrito é praticamente zero, dissipando o mínimo de energia elétrica.
				\item Com a falta de atrito, o trem pode alcançar velocidades supersônicas (JR-Maglev: 603 km/h, Transrapid Xangai: 430 km/h).
			\end{itemize}
		\end{column}
		\begin{column}{0.4\textwidth}
			\begin{figure}
				\centering
				\includegraphics[scale=0.22]{figuras/maglevcobra.jpg}
				\caption*{\footnotesize Projeto Maglev Cobra da UFRJ \cite{maglev}.}
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Transformar um número em notação científica}
	\begin{corollary}
		\begin{enumerate}[leftmargin=1.75cm]
			\item[Passo 1:] Escrever o número incluindo a vírgula.
			\item[Passo 2:] Andar com a vírgula até que reste somente um número diferente de zero no lado esquerdo.
			\item[Passo 3:] Colocar no expoente da potência de 10 o número de casas decimais que tivemos que "andar" com a vírgula. Se ao andar com a vírgula o valor do número diminuiu, o expoente ficará positivo, se aumentou o expoente ficará negativo.
		\end{enumerate}
	\end{corollary}
	\begin{exampleblock}{Exemplo}
		\begin{equation*}
			6\; 590\; 000\; 000\; 000\; 000,0 = \SI{6,59e15}{}

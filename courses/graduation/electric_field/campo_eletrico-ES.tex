\section{Distribuição discreta de cargas}

\subsection{}

\begin{frame}{Distribuição discreta de cargas puntiformes}
	\begin{columns}
		\begin{column}{0.6\textwidth}
			Sabendo que $\vec{E}=\frac{\vec{F}}{q}$, e que  $\vec{F}_p=\vec{F}_{p1}+\vec{F}_{p2}+\cdots$, podemos afirmar que para uma distribuição de cargas teremos
			\begin{align*}
				\frac{\vec{E}_{\text{\tiny p}}}{q_{\text{\tiny prova}}}=\frac{\vec{F}_{p1}}{q_1}+\frac{\vec{F}_{p2}}{q_2}+\frac{\vec{F}_{p3}}{q_3}+\cdots
			\end{align*}
			O campo elétrico $\vec{E}$, existente no ponto p, é dado pela resultante dos campos $\vec{E}_{1}$, $\vec{E}_{2}$,..., produzidos separadamente pelas cargas $q_1$, $q_2$, ...,
			\begin{align*}
				\vec{E}_p = \vec{E}_{1}+\vec{E}_{2}+\vec{E}_{3}+\cdots.
			\end{align*}
		\end{column}

\begin{frame}{Energia potencial de um dipolo elétrico}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Sabemos que a força $\vec{F}$ é conservativa, portanto podemos encontrar a energia potencial através do trabalho realizado pelo torque. Supondo $\theta=180-\theta'$ e $\sin (180-\theta')=-\sin\theta'$ temos
			\begin{NodesList}[margin=1.0cm]
				\begin{align*}
					\Delta U & = -W,\\
					\Delta U & = -\int_{\SI{90}{\degree}}^{\theta}\tau d\theta',\AddNode\\
					\Delta U & = \int_{\SI{90}{\degree}}^{\theta}pEsen\theta' d\theta'.\AddNode
				\end{align*}
				{\tikzset{LabelStyle/.style = {left=-0.25cm,pos=-0.1}}
				\LinkNodes{\tiny $\tau = -pEsen\theta'$}}
			\end{NodesList}
		\end{column}

				U(\theta) & = -pEcos\theta .
			\end{align*}
			Podemos generalizar para a forma vetorial e escrever
			\begin{equation*}
				\boxed{U = -\vec{p}\cdot\vec{E} .}
			\end{equation*}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Campo elétrico de um anel circular}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Considere um anel carregado eletricamente com uma densidade linear de carga $\lambda$. Temos que cada pedaço $ds$ do anel terá uma carga $dq$, onde
			\begin{equation*}
				dq=\lambda ds.
			\end{equation*}
			Portanto, o campo $d\vec{E}$ será dado por
			\begin{equation*}
				d\vec{E}=K\frac{\hat{r}}{r^2}\lambda ds.
			\end{equation*}
			ou na forma das componentes vetoriais,
			\begin{equation*}
				d\vec{E}=dE_x\hat{x}+dE_y\hat{y}+dE_z\hat{z}.
			\end{equation*}
		\end{column}
		\begin{column}[c]{0.4\textwidth}

	\end{columns}
\end{frame}

\begin{frame}{Campo elétrico de um anel circular (continuação)}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			Sabendo que $dE=\frac{K\lambda ds}{r^2}$, substituímos na expressão anterior
			\begin{NodesList}[margin=1.0cm]
				\begin{align*}
					dE_z & = K\frac{\lambda}{r^2}cos\theta ds,\AddNode\\
					dE_z & = K\frac{\lambda}{z^2+R^2}cos\theta ds.\AddNode
				\end{align*}
				\LinkNodes{\tiny $r^2=z^2+R^2$}
			\end{NodesList}
			Integrando ao longo de todo o anel de comprimento $2\pi R$ temos
			\begin{equation*}
				E_z = \int^{2\pi R}_0 K\frac{\lambda}{z^2+R^2}cos\theta ds,
			\end{equation*}
		\end{column}
		\begin{column}[c]{0.5\textwidth}
			\begin{align*}
				E_z & = K\frac{\lambda}{z^2+R^2}cos\theta \int^{2\pi R}_0ds,\\
				E_z & = K\frac{\lambda}{z^2+R^2}cos\theta 2\pi R.
			\end{align*}
			Substituímos o termo $cos\theta = \frac{z}{\left(z^2+R^2\right)^{1/2}}$,
			\begin{align*}
				E_z & = K\left[\frac{\lambda}{z^2+R^2}\right]\left[\frac{z}{\left(z^2+R^2\right)^{1/2}}\right]\left[ 2\pi R\right],\\
				E_z & = K\frac{ \lambda 2\pi Rz}{\left(z^2+R^2\right)^{3/2}}.
			\end{align*}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Campo elétrico de um anel circular (continuação)}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			No entanto, é conveniente definir a solução em termos da carga total ao invés da densidade de carga. Sabemos que $dq=\lambda ds$, integrando ao longo de todo o anel temos
			\begin{align*}
				dq & = \lambda ds,\\
				q & = \int_{0}^{2\pi R}\lambda ds,\\
				\Aboxed{q & = \lambda 2\pi R.}
			\end{align*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			Substituindo na expressão de $E_z$ temos o campo elétrico a uma distância z do centro de um anel circular,
			\begin{equation*}
				\boxed{E_z = K\frac{ qz}{\left(z^2+R^2\right)^{3/2}}.}
			\end{equation*}
			Se $z>>R$ teremos $z^2+R^2\approx z^2$, ou seja,
			\begin{equation*}
				\boxed{E_z = K\frac{q}{z^2}.}
			\end{equation*}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Campo elétrico de um disco circular}
	\begin{columns}
		\begin{column}[c]{0.65\textwidth}
			Considere um disco carregado eletricamente com uma densidade superficial de carga $\sigma$. Podemos considerar que o disco é formado por vários anéis de espessura $dr$ e raio $r$ contendo cargas $dq$, onde
			\begin{align*}
				dq & = \sigma dA,\\
				dq & = \sigma 2\pi rdr.
			\end{align*}

			\begin{equation*}

\begin{frame}{Campo elétrico de um disco circular (continuação)}
\begin{columns}
	\begin{column}[t]{0.5\textwidth}
		Através do campo elétrico $dE_z$ produzido pelo anel temos
		\begin{align*}
			dE_z & = K\frac{z}{\left(z^2+r^2\right)^{3/2}}dq,\\
			dE_z & = \left[K\frac{z}{\left(z^2+r^2\right)^{3/2}}\right]\left[2\pi\sigma rdr\right].
		\end{align*}
		Integrando temos
		\begin{equation*}
			E_z = k\pi\sigma z\int_{0}^{R}\left(z^2+r^2\right)^{-3/2}(2r)dr
		\end{equation*}
	\end{column}
	\begin{column}[t]{0.5\textwidth}
		Para resolver a integral fazemos a substituição $X=z^2+r^2$ e $dX=2rdr$,
		\begin{align*}
			E_z & = k\pi\sigma z\int X^{-3/2}dX,\\
			E_z & = k\overbracket{\pi\sigma}^{\alert{q/R^2}} z\left[\frac{\left(z^2+r^2\right)^{-1/2}}{-1/2}\right]^{R}_{0}.
		\end{align*}
		Chegamos assim na solução final,
		\begin{equation*}
			\boxed{E_z = 2k\frac{q}{R^2}\left(1-\frac{z}{\sqrt{z^2+R^2}}\right).}
		\end{equation*}
	\end{column}
\end{columns}
\end{frame}

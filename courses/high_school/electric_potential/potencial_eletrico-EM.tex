\section{Potencial elétrico}

\subsection{}

\begin{frame}{Trabalho realizado pela força elétrica}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Supondo duas cargas Q e q no espaço, a força $\vec{F}$ entre elas é dado pela lei de Coulomb,
			\begin{equation*}
				F=K\frac{Qq}{r^2}.
			\end{equation*}
			O trabalho necessário para trazer a carga q do ponto A ao ponto B é igual a diferença de energia potencial entre esses pontos,
			\begin{equation*}
				\tau_{AB} = U_{A}-U_{B}.
			\end{equation*}		
		\end{column}
		\begin{column}[c]{0.4\textwidth}
			\begin{figure}
				\centering
				\begin{tikzpicture}[scale=0.5, transform shape]

\end{frame}

\begin{frame}{Energia potencial elétrica}
%	\begin{multicols}{2}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Pela relação de trabalho e força
			\begin{equation*}
				\tau=F\Delta r,
			\end{equation*}
			 onde $\Delta r$ é o deslocamento realizado por q de A até B, $\Delta r = r_A-r_B$. Assim
			\begin{equation*}
				\tau \Rightarrow F_Ar_A-F_Br_B.
			\end{equation*}
			Se $F=K\frac{Qq}{r^2}$, substituímos acima
			\begin{equation*}
				\tau \Rightarrow  \left(\frac{KQq}{\bcancel{r^2_A}}\right)\left(\bcancel{r_A}\right)-\left(\frac{KQq}{\bcancel{r^2_B}}\right)\left(\bcancel{r_B}\right),
			\end{equation*}
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			\begin{equation*}
				\tau \Rightarrow \frac{KQq}{r_A}-\frac{KQq}{r_B}.
			\end{equation*}
			Definimos $U_A$ a energia potencial no ponto A e $U_B$ a energia no ponto B,
			\begin{equation*}
				\tau = U_A-U_B = \frac{KQq}{r_A}-\frac{KQq}{r_B},
			\end{equation*}
			Chegando assim na energia potencial
			\begin{align*}
				\Aboxed{U(r) & = K\frac{Qq}{r}.}
			\end{align*}
		\end{column}
	\end{columns}
%	\end{multicols}
\end{frame}

\begin{frame}{Diferença de potencial}
%	\begin{multicols}{2}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Supondo um conjunto de carga elétrica q, o trabalho necessário para deslocar do ponto A até B cada portador de carga elementar dividimos o trabalho total pela quantidade de carga elétrica q
			\begin{equation*}
				V_{AB} = \frac{\tau_{AB}}{q}.
			\end{equation*}
			Vimos anteriormente que $\tau_{AB}=U_A-U_B$,
		\end{column}
		\begin{column}[t]{0.5\textwidth}
			\begin{equation*}
				V_{AB} = \frac{U_{AB}}{q},
			\end{equation*}
			mas $U=K\frac{Qq}{r}$, portanto
			\begin{align*}
				V_{AB} & = \frac{KQ}{\bcancel{q}r_A}\bcancel{q}-\frac{KQ}{\bcancel{q}r_B}\bcancel{q},\\
				\Aboxed{V_{AB} & = K\frac{Q}{r_A}-K\frac{Q}{r_B}.}
			\end{align*}
		\end{column}
	\end{columns}
%	\end{multicols}
	\begin{block}{Diferença de potencial (d.d.p.)}
		Trabalho necessário para deslocar cada carga elementar de um ponto a outro.
	\end{block}
\end{frame}

\begin{frame}{Potencial elétrico}
	Se trouxermos a carga elementar do infinito até o ponto A teremos
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}

			\end{align*}

		Trabalho necessário para trazer uma carga elementar do infinito até o ponto A.

	\end{corollary}

\section{Potencial e campo elétrico}

\subsection{}

\begin{frame}{Relação entre campo elétrico e potencial}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Para exemplificar a relação entre campo e potencial elétrico usamos duas placas paralelas carregadas eletricamente, de modo a ter um campo elétrico $\vec{E}$ uniforme no interior dessa placa. Se colocarmos uma carga elétrica em A, o trabalho necessário para deslocá-lo até B é dado por
			\begin{equation*}
				\tau_{AB} = qV_{AB}.
			\end{equation*}
			E o trabalho é força F vezes deslocamento d, portanto
			\begin{align*}
			\tau_{AB} = Fd = qV_{AB}.
			\end{align*}
		\end{column}

\begin{frame}{Superfícies equipotenciais}
\begin{columns}
	\begin{column}[c]{0.6\textwidth}
		Pela lei de Coulomb sabemos que a força elétrica é dado por $F=qE$, onde E é o campo elétrico entre as placas, assim
		\begin{align*}
			\bcancel{q}Ed & = \bcancel{q}V_{AB},\\
			\Aboxed{V_{AB} & = Ed.}
		\end{align*}
		\begin{corollary}
			A ligação entre pontos que possuem o mesmo potencial elétrico forma uma superfície chamada superfície equipotencial.
		\end{corollary}
	\end{column}

\begin{frame}{Características de uma superfície equipotencial}
\begin{columns}
	\begin{column}[c]{0.6\textwidth}
		As características de uma superfície equipotencial são:
		\begin{enumerate}
			\item [\ding{51}] Perpendicular ao campo elétrico;
			\item [\ding{51}] A d.d.p. é diferente de zero entre duas superfícies;
			\item [\ding{51}] A d.d.p. é zero entre dois pontos da mesma superfície;
			\item [\ding{51}] Uma carga elétrica irá se deslocar entre superfícies equipotenciais diferentes, ao invés de pontos na mesma superfície.
		\end{enumerate}

\begin{frame}{Transformar um número em notação científica}
	\begin{corollary}
		\begin{enumerate}[leftmargin=1.75cm]
			\item[Passo 1:] Escrever o número incluindo a vírgula.
			\item[Passo 2:] Andar com a vírgula até que reste somente um número diferente de zero no lado esquerdo.
			\item[Passo 3:] Colocar no expoente da potência de 10 o número de casas decimais que tivemos que "andar" com a vírgula. Se ao andar com a vírgula o valor do número diminuiu, o expoente ficará positivo, se aumentou o expoente ficará negativo.
		\end{enumerate}
	\end{corollary}
	\begin{exampleblock}{Exemplo}
		\begin{equation*}
			6\; 590\; 000\; 000\; 000\; 000,0 = \SI{6,59e15}{}		
		\end{equation*}
	\end{exampleblock}
\end{frame}

\begin{frame}[t]{Carga elétrica}
    \begin{itemize}
    \item[\ding{51}] Existem \alert{dois tipos} diferentes de carga elétrica (positivo e negativo);
    \item[\ding{51}] O processo de eletrização não cria cargas, apenas a transfere de um corpo para o outro levando a \alert{lei de conservação da carga elétrica};
    \item[\ding{51}] Acreditava-se que a transferência ocorria pela carga positiva e não pela negativa;
    \item[\ding{51}] Pela experiência realizada por Du Fay cargas de mesmo sinal se repelem e sinais contrários se atraem;
    \end{itemize}
	\begin{block}{Carga elementar}
		A carga elétrica assume \alert{valores discretos}, dados pela carga -e do elétron e +e do próton.
		\[e = 1,602177\times 10^{-19}\;\textrm{C}\]
	\end{block}
\end{frame}

\begin{frame}{Lei de Coulomb}
	\begin{columns}
		\begin{column}{0.5\textwidth}
			\begin{block}{Características}
				\begin{enumerate}
					\item[\ding{51}] É uma força conservativa;
					\item[\ding{51}] Curta distância;
					\item[\ding{51}] Proporcional ao produto das cargas;
					\item[\ding{51}] Inversamente proporcional ao quadrado da distância;
					\item[\ding{51}] Pode ser atrativa ou repulsiva dependendo do produto das cargas;
					\item[\ding{51}] Obedece as Leis de Newton do movimento.
				\end{enumerate}
			\end{block}

\begin{frame}{Leis de Newton e a Força eletrostática}
	\begin{columns}
		\begin{column}[c]{0.5\textwidth}
			Vetorialmente, a lei de Coulomb é definida como
			\[\vec{F}_{Qq} = k\frac{Qq}{r^{2}_{Qq}}\hat{r}_{Qq}\]
			Sabendo que $\vec{r}_{Qq}=-\vec{r}_{qQ}$ temos que a Lei de Coulomb satisfaz a Lei da ação e reação, onde
			\begin{align*}
				\Aboxed{\vec{F}_{Qq}=-\vec{F}_{qQ}}
			\end{align*}
		\end{column}

\begin{frame}{Distribuição discreta de cargas}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
		Dado o conjunto de cargas i distribuídas no espaço, a força resultante atuando em cada carga devido as demais é obtida somando vetorialmente as forças atuando na carga i.
		\begin{align*}
			\vec{F}_{i} = \vec{F}_{i1}+\vec{F}_{i2}+\vec{F}_{i3}+\vec{F}_{i4}+\vec{F}_{i5}+\vec{F}_{i6}.
		\end{align*}
		\begin{block}{Lei de Coulomb em um distribuição de cargas}
			\begin{equation*}
				\vec{F}_{i} = \sum_{j=1}^{N}\vec{F}_{ij}.
			\end{equation*}
		\end{block}
		\end{column}

\section{Material dielétrico}

\subsection{}

\begin{frame}{Força eletrostática no meio dielétrico}
	\begin{columns}
		\begin{column}[c]{0.55\textwidth}
			\begin{corollary}
				\begin{enumerate}
					\item[\ding{51}] A presença de um material dielétrico reduz a força eletrostática entre as cargas;
					\item[\ding{51}] Quanto maior a constante dielétrica do material mais enfraquecido será a força elestrostática entre cargas elétricas;
					\item[\ding{51}] O vácuo possui o menor valor da constante dielétrica (1,0).
				\end{enumerate}
			\end{corollary}
		\end{column}
		\begin{column}[c]{0.45\textwidth}
			\begin{figure}
				\begin{center}
					\includegraphics[scale=0.22]{figuras/nacl.jpg}
					\caption*{\footnotesize Dissolução do NaCl em água.}
				\end{center}		
			\end{figure}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Diminuição da força eletrostática no meio dielétrico}
	\begin{columns}
		\begin{column}{0.55\textwidth}
			\begin{itemize}
				\item A constante dielétrica do ar é praticamente idêntico ao do vácuo;
				\item Duas cargas mergulhadas em óleo possui a força eletrostática 4,6 vezes menor que no vácuo;
				\item A água por ser um dipolo elétrico é altamente polarizável, portanto a constante dielétrica é muito alta.
			\end{itemize}
		\end{column}
		\begin{column}{0.45\textwidth}
			\begin{figure}[c]
				\begin{table}
					\centering
					\caption*{\footnotesize Constante dielétrica de diferentes materiais.}
					\vspace*{-0.5cm}
					\begin{tabular}[c]{cc}
						\hline
						Meio material & Constante dielétrica\\
						\hline
						Vácuo & 1,0\\
						Ar & 1,0005\\
						Benzeno & 2,3\\
						Óleo & 4,6\\
						Água & 81\\
						\hline
					\end{tabular}
				\end{table}			
			\end{figure}
		\end{column}
	\end{columns}
	\begin{block}{Constante dielétrica}
		Fator de redução da força elétrica entre cargas em um meio dielétrico.
	\end{block}
\end{frame}

\section{Energia}

\subsection{}

\begin{frame}{O que é energia}
	Todo fenômeno da natureza está associado a uma energia, onde à partir do trabalho podemos transformar um tipo de energia em outro tipo de energia.
	\begin{figure}
		\includegraphics[scale=1]{figuras/energia.jpg}
		\caption*{\footnotesize Tipos de energia e suas relações com trabalho.}
	\end{figure}
	\begin{corollary}
		No SI a unidade de medida de energia é Joule (J).
	\end{corollary}
\end{frame}

\begin{frame}{Energia cinética}
	\begin{enumerate}
		\item [\ding{51}] A energia cinética é a energia associada ao estado de movimento de um objeto;
		\item[\ding{51}] Quanto mais depressa o objeto se move, maior será a sua energia cinética.
		\item[\ding{51}] Como será mostrado mais a frente, para um objeto de massa m cuja velocidade $\vec{v}$ é muito menor que a velocidade da luz, a energia cinética é dado por
		\begin{equation*}
			\boxed{K = \frac{1}{2}mv^2.}
		\end{equation*}
	\end{enumerate}
	\begin{corollary}
		\begin{equation*}
			\SI{1}{\joule} = \SI{1}{kg\square\meter\per\square\second}
		\end{equation*}
	\end{corollary}
\end{frame}

\begin{frame}{Relação entre trabalho, força, deslocamento e energia cinética}
	\small
	\begin{columns}
		\begin{column}[c]{0.55\textwidth}
			Pela definição de trabalho podemos considerar que o trabalho $W$ realizado por uma força resultante $\vec{F}$ é igual a variação da energia cinética $\Delta K$, onde
			\begin{align*}
				W & = \Delta K,\\
				W & = \frac{1}{2}mv^2_f-\frac{1}{2}mv^2_i.
			\end{align*}
			Se a força for constante, poderemos dizer que a aceleração também será constante.
		\end{column}
		\begin{column}[c]{0.45\textwidth}
			Multiplicando ambos os lados da equação $v^2_f=v^2_i+2a\Delta x$ por $\frac{m}{2}$ teremos
			\begin{align*}
				\frac{m}{2}v^2_f & = \frac{m}{2}v^2_i+\frac{m}{\bcancel{2}}(\bcancel{2}a\Delta x),\\
				\frac{m}{2}v^2_f-\frac{m}{2}v^2_i & = ma\Delta x.
			\end{align*}
			mas $\vec{F}=m\vec{a}$, portanto
			\begin{equation*}
				\boxed{W = \Delta K = \vec{F}\cdot \Delta \vec{x}.}
			\end{equation*}
		\end{column}
	\end{columns}
	\begin{corollary}
		Trabalho é um grandeza escalar assim como a energia cinética.
	\end{corollary}
\end{frame}

\begin{frame}{Trabalho à partir do gráfico força versus deslocamento}

\begin{frame}{Trabalho realizado ao longo de um caminho}
	\begin{columns}
		\begin{column}[c]{0.6\textwidth}
			Cada deslocamento $d\vec{r}$ que a partícula realiza, a componente da força $\vec{F}$ paralela ao deslocamento realiza trabalho $dW$, onde
			\begin{equation*}
				dW = \vec{F}\cdot d\vec{r}.
			\end{equation*}
			Para determinar o trabalho total realizado por $\vec{F}$ durante a trajetória da partícula ao longo do caminho c, somamos cada contribuição $dW$ na forma de integral,
			\begin{equation*}
				\boxed{W = \int_{c} \vec{F}\cdot d\vec{r}.}
			\end{equation*}
		\end{column}

\begin{frame}{Trabalho e variação da energia cinética}
%	\begin{multicols*}{2}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
		Substituindo $\vec{F}=m\frac{d\vec{v}}{dt}$ na expressão do trabalho temos
		\begin{align*}
			W & = \int_{c} \vec{F}\cdot d\vec{r},\\
			W & = m\int_{c} \frac{d\vec{v}}{dt}\cdot d\vec{r}.
		\end{align*}
		No entanto, sabemos que $d\vec{r}=\frac{d\vec{v}}{dt}dt$. Substituindo temos
		\begin{align*}
			W & = \int_{c} \left(\frac{d\vec{v}}{dt}\right)\cdot\left(\frac{d\vec{r}}{dt}\right)dt.
		\end{align*}
	\end{column}
	\begin{column}[t]{0.5\textwidth}
		Mas podemos perceber também que $d\vec{v}=\frac{d\vec{v}}{dt}dt$.
		\begin{equation*}
			W = m\int_{c} \vec{v}\cdot d\vec{v}.
		\end{equation*}
		 Além disso, podemos dizer que $\vec{v}\cdot d\vec{v}\equiv vdv$, portanto
		\begin{align*}
			W & = m\int_{c} vdv,\\
			\Aboxed{W & = \frac{1}{2}mv^2_f-\frac{1}{2}mv^2_i.}
\end{align*}
	\end{column}
\end{columns}
%	\end{multicols*}
\end{frame}

\section{Potência}

\subsection{}

\begin{frame}{Potência média e potência instantânea}
	\begin{columns}
		\begin{column}[t]{0.5\textwidth}
			Definimos como potência instantânea gerada por uma força F como a taxa de variação do trabalho realizado por essa força no instante de tempo t,
			\begin{equation*}
				P(t) = \frac{dW}{dt}.
			\end{equation*}
			Calculando $\frac{dW}{dt}$ teremos
			\begin{equation*}
				\frac{dW}{dt} = \frac{d}{dt}\left(\vec{F}\cdot d\vec{r}\right).
			\end{equation*}
			\end{column}
		\begin{column}[t]{0.5\textwidth}
			Para uma força $\vec{F}$ constante no tempo podemos dizer que
			\begin{align*}
				\frac{dW}{dt} & = \vec{F}\cdot\frac{dr}{dt},\\
				\Aboxed{P(t) & = \vec{F}\cdot\vec{v}(t).}
			\end{align*}
			\begin{corollary}
				No SI a unidade de medida de potência é J/s ou Watt (W).
			\end{corollary}
		\end{column}
	\end{columns}
\end{frame}
